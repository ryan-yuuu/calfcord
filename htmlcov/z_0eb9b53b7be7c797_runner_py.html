1"""CLI entry point for running calfkit agents. 

2 

3Two modes, selected by the positional ``agent`` argument: 

4 

5* ``calfkit-agent <name>`` — run one agent in its own process. One 

6 :class:`calfkit.Worker` with a single :class:`Agent` node, reachable 

7 **by name** on its automatic private input topic. 

8* ``calfkit-agent`` (no argument) — run *every* agent in ``agents/*.md`` on 

9 a single shared :class:`calfkit.Worker` in one process. Each agent is a 

10 separate calfkit node with its own Kafka consumer group (defaulting to 

11 the agent's name) so co-tenant nodes do not contend for partitions. 

12 

13The all-agents mode trades crash isolation for resource simplicity: one 

14Discord webhook client, one Kafka connection, one process to supervise. 

15Concurrent LLM calls across agents are not a problem — they're awaitable 

16network I/O and yield on the event loop. The real tradeoff is *failure 

17domain*: an unhandled exception in any agent's handler tears down the 

18whole Worker. For a small, well-tested fleet that's acceptable; if one 

19agent is materially less stable than the rest, run it as its own 

20process via ``calfkit-agent <name>``. 

21 

22Agents are **name-addressed** (calfkit ADR-0017): there is no per-agent 

23channel state to seed and no addressing gate. The bridge reaches each agent 

24directly by name, and the managed Worker auto-provisions each node's private 

25input + return topics at broker start. Agents advertise their ``AgentCard`` 

26on the native mesh automatically, so the runner wires no presence/departure 

27lifecycle hooks. 

28 

29Run:: 

30 

31 uv run calfkit-agent # all agents in one process 

32 uv run calfkit-agent <agent_name> # one agent in one process 

33"""

103def _resolve_definition(agent_name: str, agents_dir: Path) -> AgentDefinition: 

104 definitions = load_agents_dir(agents_dir) 

105 for d in definitions: 

106 if d.agent_id == agent_name: 

107 return d 

108 known = ", ".join(sorted(d.agent_id for d in definitions)) 

109 raise SystemExit(f"agent {agent_name!r} not found in {agents_dir}. Known: {known or '<none>'}")

112def _resolve_definitions( 

113 agent_name: str | None, 

114 agents_dir: Path, 

115 *, 

116 targets: list[Path] | None = None, 

117) -> list[AgentDefinition]: 

118 """Resolve which agent definitions to run. 

119 

120 Three modes, selected by precedence (``targets`` wins, then ``agent_name``, 

121 then the directory scan): 

122 

123 Targets mode (``targets`` non-empty): one definition per agent resolved 

124 from the explicit ``--target`` file/directory paths via 

125 :func:`load_agent_targets` (which de-duplicates by ``agent_id``). 

126 

127 Single-agent mode (``agent_name`` set, no targets): a list of length 1. 

128 

129 All-agents mode (no targets, ``agent_name`` is ``None``): one definition 

130 per ``agents/*.md`` file. 

131 

132 Name-addressing removed per-agent channel bootstrap, so resolution is pure 

133 ``.md`` loading with no I/O beyond the file read. Any load failure (missing 

134 path, neither file nor directory, malformed ``.md``, duplicate ``agent_id``, 

135 unknown name, empty result) raises :class:`SystemExit` so the CLI exits 

136 cleanly on stderr without a traceback. 

137 """ 

138 if targets: 

139 try: 

140 definitions = load_agent_targets(targets) 

141 except (FileNotFoundError, NotADirectoryError, ValueError) as e: 

142 raise SystemExit(f"failed to load --target paths: {e}") from e 

143 if not definitions: 

144 joined = ", ".join(str(t) for t in targets) 

145 raise SystemExit(f"no agent definitions found in --target paths: {joined}") 

146 return definitions 

147 

148 if agent_name is not None: 

149 return [_resolve_definition(agent_name, agents_dir)] 

150 

151 try: 

152 definitions = load_agents_dir(agents_dir) 

153 except (FileNotFoundError, NotADirectoryError, ValueError) as e: 

154 raise SystemExit(f"failed to load {agents_dir}: {e}") from e 

155 if not definitions: 

156 raise SystemExit(f"no agent definitions found in {agents_dir}") 

157 return definitions

160def _build_node_or_exit(factory: AgentFactory, definition: AgentDefinition) -> Agent: 

161 """Wrap :meth:`AgentFactory.build_node` so any failure exits cleanly. 

162 

163 CLI build: any failure is fatal. The bare-Exception catch is deliberate — 

164 pydantic_ai raises ``UserError(RuntimeError)`` for missing API keys and the 

165 model-client constructors can raise other types we don't want to enumerate. 

166 Convert them all to a clean stderr :class:`SystemExit` so operators don't 

167 see a traceback. 

168 """ 

169 try: 

170 return factory.build_node(definition) 

171 except Exception as e: 

172 raise SystemExit(f"agent {definition.agent_id!r} failed to construct: {e}") from e

175async def _run_worker(worker: Worker) -> None: 

176 """Run ``worker`` until SIGINT/SIGTERM, then drain cleanly. 

177 

178 Delegates to the shared :func:`calfcord._worker_runtime.run_worker_until_signal` 

179 so the shutdown contract (signal-driven drain plus the "clean return 

180 without a signal is a crash" supervisor invariant) is defined in exactly 

181 one place across runners — mirroring :func:`calfcord.tools.runner._run_worker`. 

182 Kept as a thin local wrapper because the runner unit tests reference 

183 ``_run_worker`` by name. 

184 """ 

185 await run_worker_until_signal(worker, drain_label="agents worker")

188async def _prewarm_codex_if_needed(definitions: list[AgentDefinition]) -> None: 

189 """If any definition resolves to the openai-codex provider, prewarm the cache. 

190 

191 Uses :func:`resolve_provider` rather than reading ``definition.provider`` 

192 directly so the ``CALFKIT_AGENT_DEFAULT_PROVIDER`` env-var fallback is 

193 honoured — a bare attribute access would be ``None`` for any agent that 

194 omits ``provider:`` from its frontmatter, even if the operator selected 

195 openai-codex globally via env var. Missing that case would skip prewarm 

196 and crash mid-factory with an opaque ``RuntimeError`` instead of the 

197 actionable :class:`SystemExit` raised here. 

198 

199 Raises: 

200 SystemExit: if the upstream prompt fetch fails AND no cache exists. 

201 Includes a hint pointing the operator at 

202 ``calfkit-auth codex refresh-prompts``. 

203 """ 

204 needs_codex = any(resolve_provider(d) == "openai-codex" for d in definitions) 

205 if not needs_codex: 

206 return 

207 # Lazy import: keeps authlib + openhands-sdk auth machinery out of the 

208 # import graph for deployments that don't use Codex subscription. 

209 from calfcord.providers.codex import ( 

210 CodexPromptsUnavailableError, 

211 prewarm_codex_prompts, 

212 ) 

213 

214 try: 

215 await prewarm_codex_prompts() 

216 except CodexPromptsUnavailableError as exc: 

217 raise SystemExit( 

218 f"openai-codex agents declared but upstream Codex prompts " 

219 f"are unavailable: {exc}. Check internet connectivity, or " 

220 f"run once: uv run calfkit-auth codex refresh-prompts" 

221 ) from exc

224async def _amain(args: argparse.Namespace) -> None: 

225 """Build and run the agent(s). Pure-ish: callers configure logging+env.""" 

226 agents_dir = Path(os.getenv(_AGENTS_DIR_ENV, _AGENTS_DIR_DEFAULT)) 

227 

228 targets = [Path(t) for t in args.targets] if args.targets else None 

229 definitions = _resolve_definitions(args.agent, agents_dir, targets=targets) 

230 await _prewarm_codex_if_needed(definitions) 

231 

232 settings = DiscordSettings() # type: ignore[call-arg] 

233 server_urls = os.getenv("CALF_HOST_URL") or "localhost" 

234 

235 async with ( 

236 DiscordPersonaSender(settings) as persona_sender, 

237 Client.connect(server_urls, provisioning=PROVISIONING) as calfkit_client, 

238 ): 

239 factory = AgentFactory(persona_sender, calfkit_client) 

240 nodes = [_build_node_or_exit(factory, definition) for definition in definitions] 

241 

242 # Managed lifecycle: ``Worker.start()`` (via _run_worker) auto-registers 

243 # handlers and auto-provisions each agent's node topics — its private 

244 # input topic ``agent.{name}.private.input`` (name-addressing) and its 

245 # private return topic — at broker start, and the connect-hook 

246 # auto-provisions the client reply topic. Name-addressed agents carry no 

247 # channel subscriptions and advertise their AgentCard automatically, so 

248 # there are no domain lifecycle hooks to wire here. 

249 worker = Worker(calfkit_client, nodes) 

250 

251 logger.info( 

252 "starting worker with %d agent(s): %s", 

253 len(nodes), 

254 ", ".join(n.node_id for n in nodes), 

255 ) 

256 await _run_worker(worker)
