1"""``calfcord-package-tools`` — build a slim image hosting only specific tools. 

2 

3Walkthrough: 

4 1. Parse args. Names are positional; ``--tag`` is required. 

5 Optional repeatable ``--rename SRC=DST`` aliases a tool to a 

6 new schema name in the resulting image (multi-host deployment 

7 of the same tool — e.g. ``edit_file`` running on a workstation 

8 AND an EU VM, with the EU host renamed to ``edit_file_eu``). 

9 2. Validate the names against the live ``TOOL_REGISTRY``. Unknown 

10 names fail-fast with the full known list (mirrors the 

11 agent-factory's tool-resolution error message). 

12 3. Validate each ``--rename SRC=DST``: ``SRC`` must be a real tool, 

13 ``DST`` must match the tool-name regex and not collide with 

14 another tool or another rename target. 

15 4. Translate positional names through the alias map. An operator 

16 who typed ``edit_file --rename edit_file=edit_file_eu`` means 

17 "host the edit_file tool but expose it as edit_file_eu" — the 

18 filter that lands in the image must be the POST-rename name, 

19 otherwise discovery would drop the alias clone and keep the 

20 original. 

21 5. Generate the Dockerfile via the templater. The image bakes 

22 ``CALFCORD_TOOLS_INCLUDE`` plus an optional 

23 ``CALFCORD_TOOLS_ALIAS`` so the auto-discovery loader narrows 

24 registration to just the listed names (and clones renamed 

25 tools under their new identity). 

26 6. Write the Dockerfile to a tempdir; invoke ``docker buildx build``. 

27 7. On success, print the local tag plus follow-up commands the 

28 operator typically wants. On failure, leave the Dockerfile in 

29 place and print its path. 

30 

31No registry push: the CLI builds locally with ``--tag``; the operator 

32runs ``docker push`` afterward against whichever registry they own. 

33"""

40# Suppress the openhands SDK boot banner during ``--help`` and registry 

41# validation. The CLI is a build-time tool, not a runtime worker — 

42# operators don't care about openhands' banner here. Set BEFORE any 

43# import of ``calfcord.tools`` (which transitively imports 

44# the openhands SDK). Respect an explicit user override so anyone 

45# wanting to see the banner can ``OPENHANDS_SUPPRESS_BANNER=0 …``. 

46os.environ.setdefault("OPENHANDS_SUPPRESS_BANNER", "1") 

47 

48# Strip deploy-time runtime env vars from the operator's shell BEFORE 

49# anything imports ``calfcord.tools``. The CLI validates 

50# ``--rename`` targets against the canonical ``TOOL_REGISTRY``; if the 

51# operator already has ``CALFCORD_TOOLS_ALIAS`` or 

52# ``CALFCORD_TOOLS_INCLUDE`` set in their shell, the registry imported 

53# below would reflect the operator's env rather than the codebase's 

54# canonical surface, and validation would pass against a poisoned 

55# baseline (e.g. ``--rename anything=foo`` would succeed because the 

56# operator's env already cloned ``foo``). The image would then fail on 

57# a fresh host. ``pop`` here is correct: these env vars are 

58# DEPLOY-time runtime configuration that should never apply at 

59# BUILD-time validation. 

60os.environ.pop("CALFCORD_TOOLS_ALIAS", None) 

61os.environ.pop("CALFCORD_TOOLS_INCLUDE", None)

68def _validate_tool_names(names: list[str]) -> list[str]: 

69 """Check each name resolves to a real entry in ``TOOL_REGISTRY``. 

70 

71 Returns the de-duplicated, order-preserving list on success. 

72 Calls :func:`sys.exit` with code 2 (the argparse-standard "usage 

73 error" code) if any name is unknown, after printing the full 

74 known list for forensic value. 

75 

76 Importing :data:`TOOL_REGISTRY` here triggers the auto-discovery 

77 walk in ``tools/__init__.py``; that's fine — the CLI runs 

78 short-lived so the boot cost is paid once per build invocation. 

79 """ 

80 # Local import keeps ``--help`` cheap (no openhands-tools / smolagents 

81 # imports just to display usage text). 

82 from calfcord.tools import TOOL_REGISTRY 

83 

84 known = sorted(TOOL_REGISTRY) 

85 unknown = [name for name in names if name not in TOOL_REGISTRY] 

86 if unknown: 

87 sys.stderr.write( 

88 f"error: unknown tool(s) {unknown!r}\n" 

89 f" Known tools: {', '.join(known)}\n" 

90 ) 

91 sys.exit(2) 

92 # De-dupe while preserving caller's order (deterministic Dockerfile 

93 # output regardless of how the user typed the args). 

94 seen: set[str] = set() 

95 result: list[str] = [] 

96 for name in names: 

97 if name not in seen: 

98 seen.add(name) 

99 result.append(name) 

100 return result

103def _parse_and_validate_renames(rename_args: list[str]) -> dict[str, str]: 

104 """Parse ``--rename SRC=DST`` flags into a ``{src: dst}`` dict. 

105 

106 Each entry must: 

107 * Split cleanly on the first ``=`` into non-empty SRC and DST. 

108 * Have SRC in the live ``TOOL_REGISTRY`` (catches typos at 

109 build time rather than at deploy time). 

110 * Have DST matching :data:`TOOL_NAME_REGEX` (shared char set 

111 between Anthropic and OpenAI; the length bound is Anthropic's 

112 128 — operators targeting OpenAI-only deployments should keep 

113 targets ≤ 64 chars). 

114 * SRC != DST (no-op aliasing). 

115 * DST not colliding with another tool name in ``TOOL_REGISTRY`` 

116 (would shadow the audited tool — refuse). 

117 * Each SRC and each DST appearing at most once across all 

118 ``--rename`` flags (v1 supports one alias per source and one 

119 source per target; multi-region from a single image is a 

120 future feature). 

121 

122 Any failure prints an actionable message and exits 2. Returns the 

123 parsed map on success. 

124 """ 

125 from calfcord.tools import TOOL_REGISTRY 

126 

127 result: dict[str, str] = {} 

128 used_targets: set[str] = set() 

129 known = sorted(TOOL_REGISTRY) 

130 for raw in rename_args: 

131 if "=" not in raw: 

132 sys.stderr.write( 

133 f"error: --rename expects SRC=DST, got {raw!r}\n" 

134 ) 

135 sys.exit(2) 

136 src, _, dst = raw.partition("=") 

137 src = src.strip() 

138 dst = dst.strip() 

139 if not src or not dst: 

140 sys.stderr.write( 

141 f"error: --rename {raw!r} has empty SRC or DST\n" 

142 ) 

143 sys.exit(2) 

144 if not TOOL_NAME_REGEX.match(dst): 

145 sys.stderr.write( 

146 f"error: --rename target {dst!r} is not a valid tool name; " 

147 f"must match {TOOL_NAME_REGEX.pattern}\n" 

148 ) 

149 sys.exit(2) 

150 if src not in TOOL_REGISTRY: 

151 sys.stderr.write( 

152 f"error: --rename source {src!r} is not a known tool\n" 

153 f" Known tools: {', '.join(known)}\n" 

154 ) 

155 sys.exit(2) 

156 if src == dst: 

157 sys.stderr.write( 

158 f"error: --rename {raw!r} aliases a tool to itself; " 

159 f"either drop the flag or pick a distinct target name\n" 

160 ) 

161 sys.exit(2) 

162 if dst in TOOL_REGISTRY: 

163 sys.stderr.write( 

164 f"error: --rename target {dst!r} collides with an existing " 

165 f"tool name. Renaming over an audited tool would silently " 

166 f"shadow it; pick a unique target.\n" 

167 ) 

168 sys.exit(2) 

169 if src in result: 

170 sys.stderr.write( 

171 f"error: --rename source {src!r} is aliased multiple times; " 

172 f"only one alias per source is supported\n" 

173 f" All --rename args: {rename_args!r}\n" 

174 ) 

175 sys.exit(2) 

176 if dst in used_targets: 

177 sys.stderr.write( 

178 f"error: --rename target {dst!r} is used by multiple " 

179 f"renames; only one source may alias to a given target\n" 

180 f" All --rename args: {rename_args!r}\n" 

181 ) 

182 sys.exit(2) 

183 result[src] = dst 

184 used_targets.add(dst) 

185 return result

188def _check_aliases_referenced( 

189 aliases: dict[str, str], include_tools: list[str] 

190) -> None: 

191 """Refuse builds where a --rename source isn't in the positional list. 

192 

193 The CLI's positional-name translation maps each include name 

194 through the alias map before baking ``CALFCORD_TOOLS_INCLUDE``. 

195 If the operator typed ``--rename edit_file=edit_file_eu`` but 

196 didn't add ``edit_file`` to the positional list, the alias bakes 

197 into the image's ``CALFCORD_TOOLS_ALIAS`` env BUT the include 

198 filter doesn't reference ``edit_file_eu`` (the post-rename name) 

199 — discovery at boot would add the clone, then immediately drop 

200 it via the filter. The image then has rename env doing nothing 

201 (visible in ``docker inspect`` as live config but inert at 

202 runtime) — silent dead config. 

203 

204 Refusing at build time short-circuits the "deploy, debug, why 

205 didn't it work" cycle. 

206 """ 

207 unused = sorted(set(aliases) - set(include_tools)) 

208 if unused: 

209 sys.stderr.write( 

210 f"error: --rename source(s) {unused!r} not in positional tool " 

211 f"list {include_tools!r}.\n" 

212 f" An alias whose source isn't in the include filter is dead " 

213 f"config — the clone gets registered then dropped at boot. Add " 

214 f"the source(s) to the positional list, or remove the unused " 

215 f"--rename flag(s).\n" 

216 ) 

217 sys.exit(2)

220def main(argv: list[str] | None = None) -> int: 

221 parser = make_parser( 

222 prog="calfcord-package-tools", 

223 item_kind="tool", 

224 item_examples="shell grep", 

225 ) 

226 # The ``--rename`` flag is specific to tool images — agent .md 

227 # files carry identity-bound fields (slash command, display name) 

228 # that don't safely rename at deploy time, so the shared 

229 # ``make_parser`` deliberately doesn't carry this flag. 

230 parser.add_argument( 

231 "--rename", 

232 action="append", 

233 default=[], 

234 metavar="SRC=DST", 

235 help=( 

236 "Rename a tool's schema name at deploy time. The same Python " 

237 "tool body registers under DST instead of (or in addition to) " 

238 "SRC; the Kafka topics it subscribes to are renamed in lockstep. " 

239 "Use for multi-host deployments of the same tool. Pair with the " 

240 "positional name list (which must reference DST, not SRC) for " 

241 "true rename behavior on the target host. Repeatable." 

242 ), 

243 ) 

244 args = parser.parse_args(argv) 

245 

246 aliases = _parse_and_validate_renames(args.rename) 

247 include_tools = _validate_tool_names(args.names) 

248 # Cross-validate: every alias src must appear in the positional 

249 # include list. Otherwise the alias env would bake into the 

250 # image but the include filter wouldn't reference the post- 

251 # rename name, leaving the clone dropped at boot — silent dead 

252 # config. 

253 _check_aliases_referenced(aliases, include_tools) 

254 # Translate the positional include list through the alias map. 

255 # An operator who types ``calfcord-package-tools edit_file --rename 

256 # edit_file=edit_file_eu`` means "host the edit_file tool body but 

257 # expose it as edit_file_eu" — the filter env var baked into the 

258 # image must reference the POST-rename name, otherwise discovery 

259 # would drop the alias clone (filter excludes ``edit_file_eu``) 

260 # and keep the original (filter includes ``edit_file``) — the 

261 # opposite of intent. The dict.get fallback preserves names that 

262 # weren't renamed. 

263 include_tools = [aliases.get(n, n) for n in include_tools] 

264 dockerfile_content = render_tools_dockerfile( 

265 include_tools=include_tools, 

266 aliases=aliases or None, 

267 ) 

268 

269 context = args.context if args.context is not None else repo_root() 

270 return run_build( 

271 dockerfile_content=dockerfile_content, 

272 tag=args.tag, 

273 context=context, 

274 dry_run=args.dry_run, 

275 verbose=args.verbose, 

276 )
