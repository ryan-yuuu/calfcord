1"""Publish helpers for the control plane. 

2 

3These functions reach into ``client._connection`` (a FastStream KafkaBroker) 

4because calfkit's public ``Client`` API exposes only ``send`` / ``start`` / ``execute``, 

5all of which are agent invocations -- not what we want for plain control-plane 

6messages. The private-attribute access is documented here in one place so a 

7future calfkit upgrade that exposes a public ``Client.publish`` is a single- 

8file swap. 

9 

10The same broker is used for calfkit's own agent invocation traffic; both flows 

11coexist on different topics. 

12 

13Partition keys 

14-------------- 

15All per-agent publishes (state events, departures, control commands) use the 

16``agent_id`` as the Kafka partition key. This pins every message for a given 

17agent to a single partition, which is the only way Kafka guarantees ordering 

18across multiple events. Without a key, a multi-partition deployment could 

19deliver an agent's ``command_applied`` state event AFTER its departure event, 

20leaving the bridge with a stale resurrected entry. Discovery pings are 

21broadcast and not keyed -- each agent is in its own consumer group, so every 

22agent reads every partition of ``bridge.discovery`` regardless. 

23"""

26import uuid 

27from datetime import UTC, datetime

29from calfkit.client import Client

31from calfcord.control_plane.schema import (

37) 

38from calfcord.control_plane.topics import ( 

39 AGENT_STATE_TOPIC, 

40 BRIDGE_DISCOVERY_TOPIC,

42)

45def _agent_key(agent_id: str) -> bytes: 

46 """Encode ``agent_id`` as a Kafka partition key. 

47 

48 Kafka requires keys to be bytes (or null). Encoding here so callers don't 

49 have to repeat themselves; UTF-8 is the conventional choice and matches 

50 how the agent_id is serialized inside the payload itself. 

51 """ 

52 return agent_id.encode("utf-8") 

53 

54 

55async def publish_control_command( 

56 client: Client, agent_id: str, command: AgentControlCommand 

57) -> None: 

58 """Publish a targeted control command to the agent's control topic. 

59 

60 Fire-and-forget: returns when the broker has accepted the message; 

61 does not wait for the agent to consume or apply. 

62 

63 Partition-keyed by ``agent_id`` so that, on a multi-partition deployment 

64 of ``agent.<id>.control.in``, successive operator commands stay ordered. 

65 """ 

66 envelope = AgentControlEnvelope(command=command) 

67 await client._connection.publish( 

68 envelope.model_dump(mode="json"), 

69 topic=control_topic_for(agent_id), 

70 key=_agent_key(agent_id), 

71 )

74async def publish_discovery_ping(client: Client) -> None: 

75 """Broadcast a discovery ping. Every running agent's control sink re-announces. 

76 

77 Not partition-keyed: this is broadcast traffic, and each agent's control 

78 sink is the sole member of its own consumer group, so it consumes from 

79 every partition regardless of how the producer routed the message. 

80 """ 

81 envelope = AgentControlEnvelope( 

82 command=DiscoveryPingOp( 

83 issued_at=datetime.now(UTC), 

84 request_id=str(uuid.uuid4()), 

85 ), 

86 ) 

87 await client._connection.publish( 

88 envelope.model_dump(mode="json"), 

89 topic=BRIDGE_DISCOVERY_TOPIC, 

90 )

93async def publish_state_event(client: Client, event: AgentStateEvent) -> None: 

94 """Agent publishes its current state on startup, on each applied command, 

95 or in response to a discovery ping. 

96 

97 Partition-keyed by ``event.agent_id`` so the bridge consumes all events 

98 for a given agent in publish order. Without this, a multi-partition 

99 ``agent.state`` topic could deliver a ``command_applied`` event AFTER 

100 the agent's departure event, leaving the bridge with a stale entry. 

101 """ 

102 await client._connection.publish( 

103 event.model_dump(mode="json"), 

104 topic=AGENT_STATE_TOPIC, 

105 key=_agent_key(event.agent_id), 

106 ) 

107

109async def publish_departure(client: Client, agent_id: str) -> None: 

110 """Best-effort graceful goodbye on agent shutdown. 

111 

112 Bridge consumes from agent.state, dispatches on the ``kind`` discriminator, 

113 and removes the agent from its registry projection. Crashes / SIGKILL leave 

114 the bridge with a stale entry until the agent next restarts. 

115 

116 Partition-keyed by ``agent_id`` -- same partition as the agent's state 

117 events so the bridge sees the departure AFTER any prior state events, 

118 not before or interleaved. 

119 """ 

120 event = AgentDepartureEvent( 

121 agent_id=agent_id, 

122 departed_at=datetime.now(UTC), 

123 ) 

124 await client._connection.publish( 

125 event.model_dump(mode="json"), 

126 topic=AGENT_STATE_TOPIC, 

127 key=_agent_key(agent_id), 

128 )
