1"""First-reply detection over the Discord outbox (host-agnostic, §4.6 / §12.2). 

2 

3The init wizard's live finish needs to confirm — end to end, over the real 

4broker — that the agent it just started actually answers. After the user types 

5``@<agent> hello`` in Discord, that agent's ``ReturnCall`` lands on 

6``discord.outbox`` (the same topic the bridge posts replies from). This module 

7watches that topic for the FIRST reply from the *target* agent and returns 

8``True``, or returns ``False`` on a clean timeout. 

9 

10Why a calfkit consumer, not a raw subscriber 

11-------------------------------------------- 

12The replying-agent identity is NOT in the envelope JSON body. ``emitter_node_id`` 

13/ ``emitter_node_kind`` are ``PrivateAttr`` on :class:`SessionRunContext`, 

14stamped from the Kafka headers ``x-calf-emitter`` / ``x-calf-emitter-kind`` by 

15the consumer handler's ``_stamp_transport`` and explicitly excluded from 

16serialization. A raw ``broker.subscriber`` that only decodes the body therefore 

17*cannot* recover which agent replied. So — exactly like the bridge's own outbox 

18consumer (:func:`calfcord.bridge.outbox.build_outbox_consumer`) — we register a 

19:class:`~calfkit.ConsumerNode`, whose handler reads the headers and projects a 

20:class:`~calfkit.ConsumerContext` with ``emitter_node_id`` / ``emitter_node_kind`` 

21populated. 

22 

23The match field (mirrors the bridge's gate set): ``emitter_node_kind == "agent"`` 

24AND ``emitter_node_id == agent_id`` (an agent's ``node_id`` == its ``agent_id`` 

25== its ``.md`` ``name``), behind the same ``final_output_parts`` non-empty gate 

26the outbox uses to skip intermediate hops (tool completions, mid-loop 

27transitions). 

28 

29Isolation from the bridge 

30-------------------------- 

31The watcher runs in its OWN Kafka consumer group (a per-call uuid node_id) at 

32``auto_offset_reset="latest"`` (the Worker default), so it never disturbs the 

33bridge's outbox group nor replays backlog — it sees only replies that arrive 

34after it joins, which is precisely the reply to the user's ``@<agent> hello``. 

35 

36Separation of concerns 

37---------------------- 

38:func:`make_first_reply_node` is the pure, stateless matcher (a 

39``ConsumerNode`` factory) — unit-testable by driving its handler directly, 

40no broker. :func:`wait_for_first_reply` is the thin orchestrator that wires the 

41node onto a transient managed Worker (whose start ensurer provisions the node's 

42``discord.outbox`` subscribe topic for it) and bounds the wait. One-shot 

43semantics (stop on first match) live in the orchestrator, keeping the node a 

44reusable matcher. 

45"""

68def make_first_reply_node( 

69 agent_id: str, 

70 *, 

71 on_match: Callable[[], None], 

72 on_match_correlation: Callable[[str], None] | None = None, 

73 node_id: str | None = None, 

74) -> ConsumerNode[str]: 

75 """Build the consumer node that fires ``on_match`` on the target agent's reply. 

76 

77 Pure and stateless: it carries no "have I matched yet" flag — every envelope 

78 that satisfies the gate + emitter checks invokes ``on_match``. The caller 

79 (the orchestrator) owns first-only semantics by, e.g., setting an 

80 :class:`asyncio.Event` in ``on_match`` and tearing the Worker down once it 

81 fires. 

82 

83 Args: 

84 agent_id: The agent whose reply we are watching for. An agent's 

85 ``node_id`` equals its ``agent_id`` equals its ``.md`` ``name``, so 

86 this is matched directly against ``ConsumerContext.emitter_node_id``. 

87 on_match: Invoked (no args) when a reply from ``agent_id`` lands. Kept 

88 argument-free so the common case (set an Event) is trivial. 

89 on_match_correlation: Optional secondary callback receiving the matching 

90 envelope's ``correlation_id`` — lets a caller tie the live reply 

91 back to the inbound wire's ``event_id`` (the bridge's idiom). Called 

92 after ``on_match``. 

93 node_id: Optional consumer node id override (the Kafka ``group_id``). 

94 Defaults to a per-instance uuid so the watcher always reads in its 

95 own group at ``latest`` and never disturbs the bridge's outbox group. 

96 

97 Returns: 

98 A :class:`ConsumerNode` ready to register on a :class:`Worker`. 

99 """ 

100 

101 def _final_output_parts_gate(ctx: SessionRunContext) -> bool: 

102 # Same gate the bridge outbox uses: skip intermediate hops (tool 

103 # completions, mid-loop transitions) that carry no final output. 

104 return bool(ctx.state.final_output_parts) 

105 

106 def _consume(result: ConsumerContext[str]) -> None: 

107 # Identity lives in the Kafka headers, recovered by the consumer handler 

108 # into these fields — a non-agent emitter (tool/client) or a different 

109 # agent (multi-agent org) is not the reply we're confirming. 

110 if result.emitter_node_kind != _AGENT_EMITTER_KIND: 

111 return 

112 if result.emitter_node_id != agent_id: 

113 return 

114 on_match() 

115 if on_match_correlation is not None: 

116 on_match_correlation(result.correlation_id) 

117 

118 return ConsumerNode[str]( 

119 node_id=node_id or f"calfcord-firstreply-{uuid.uuid4().hex}", 

120 subscribe_topics=DISCORD_OUTBOX_TOPIC, 

121 consume_fn=_consume, 

122 agent_output_type=str, 

123 gates=[_final_output_parts_gate], 

124 )

127async def wait_for_first_reply( 

128 server_urls: str, 

129 *, 

130 agent_id: str, 

131 timeout_s: float = _DEFAULT_REPLY_TIMEOUT_S, 

132 client: Client | None = None, 

133 ready: asyncio.Event | None = None, 

134) -> bool: 

135 """Watch ``discord.outbox`` for the first reply from ``agent_id``. 

136 

137 Host-agnostic and bridge-independent: connects a transient client (its own 

138 consumer group at ``latest``), registers a :func:`make_first_reply_node` 

139 consumer on a Worker, and awaits an :class:`asyncio.Event` set by the node's 

140 ``on_match``, bounded by ``timeout_s``. 

141 

142 Returns: 

143 ``True`` if a reply from ``agent_id`` arrived within ``timeout_s``; 

144 ``False`` on a clean timeout (the wizard then downgrades to a 

145 try-it-yourself hint rather than promising more than it detected). 

146 

147 Args: 

148 server_urls: Kafka bootstrap URL(s) — the same value passed to 

149 ``Client.connect`` elsewhere (calfkit 0.6.0 removed 

150 ``Client.server_urls``, so the URL is threaded through explicitly for 

151 provisioning). 

152 agent_id: The agent whose reply confirms the org is live. 

153 timeout_s: Bounded wait before giving up. 

154 client: Injected transient client for tests (so a unit/integration test 

155 can supply a fake or a pre-wired connection). Defaults to a fresh 

156 :meth:`Client.connect` with the shared opt-in provisioning policy. 

157 ready: Optional caller-owned event set the instant the consumer group has 

158 JOINED (right after ``worker.start()`` returns), BEFORE the match 

159 wait. The init wizard awaits it so it only prompts the human to send 

160 ``@<agent> hello`` once this ``latest``-offset watcher is listening — 

161 closing the race where a fast human posts before the group joins and 

162 the reply lands before the watcher is subscribed. Set even when the 

163 wait later times out: readiness is the join, not the reply. 

164 """ 

165 matched = asyncio.Event() 

166 node = make_first_reply_node(agent_id, on_match=matched.set) 

167 

168 owns_client = client is None 

169 transient = client if client is not None else Client.connect(server_urls, provisioning=PROVISIONING) 

170 try: 

171 worker = Worker(transient, [node]) 

172 # ``discord.outbox`` is this node's SUBSCRIBE topic, and Worker is a 

173 # MANAGED surface: its ``start()`` ensurer auto-provisions the node's 

174 # subscribe topics (via ``topics_for_nodes``) in calfkit's single 

175 # pre-start pass — so no separate ``provision_extra_topics`` is needed 

176 # here. This matches the tools/router/agents managed runners, which all 

177 # trust the managed start to create their node topics (see 

178 # ``calfcord._provisioning``). Embedded managed lifecycle (signals OFF), 

179 # like the bridge: start() joins the consumer group, then we block on the 

180 # match Event with a hard timeout. ``stop()`` in the finally always 

181 # drains the broker — a no-op if start() never ran. 

182 await worker.start() 

183 # Signal readiness ONLY after the group has joined: the caller gates its 

184 # "now send @agent hello" prompt on this, so any reply the human triggers 

185 # afterward lands while we are already subscribed at ``latest``. 

186 if ready is not None: 

187 ready.set() 

188 try: 

189 await asyncio.wait_for(matched.wait(), timeout=timeout_s) 

190 return True 

191 except TimeoutError: 

192 return False 

193 finally: 

194 await worker.stop() 

195 finally: 

196 if owns_client: 

197 # Graceful client shutdown: cancels the (unused) reply dispatcher's 

198 # pending futures and stops the broker. Idempotent with the 

199 # ``worker.stop()`` above (both stop the same broker connection). 

200 await transient.close()
