1"""Best-effort Discord typing indicator for the bridge. 

2 

3Discord's only live-activity affordance is the channel typing indicator 

4("*X is typing…*"), triggered by ``POST /channels/{channel_id}/typing`` 

5(:meth:`discord.http.HTTPClient.send_typing`). It expires after ~10 seconds and 

6has no "stop" counterpart, so the bridge simply (re)fires it on each unit of 

7agent work for a correlation and lets it lapse on its own. Because replies post 

8via webhooks (a different identity than the bot user that types), nothing the 

9bridge posts can actively clear the indicator, so it persists up to ~10s from 

10the last fire — a short, accepted tail once the reply lands. 

11 

12:class:`TypingNotifier` wraps that one call with the project's 

13fire-and-forget + swallow-Discord-errors discipline: 

14 

15* **Fire-and-forget.** :meth:`fire` is synchronous and returns immediately, 

16 scheduling the REST call on a detached :class:`asyncio.Task`. This is 

17 deliberate: under the 0.12 caller surface the bridge's progress renderer 

18 drains the run's ``stream()`` and processes hops *serially* on the bridge's 

19 event loop (there are no consumers and no ``agent.steps`` topic), so an 

20 awaited ``send_typing`` that hit discord.py's rate limiter (which 

21 ``asyncio.sleep``s inside the request when ``max_ratelimit_timeout`` is 

22 unset — and the shared REST client leaves it unset) would stall the 

23 live-progress UI for **every** channel. Detaching the call keeps any 

24 rate-limit sleep off the renderer's critical path. Detached tasks are held 

25 in a set (with a done-callback that discards them) so they are not 

26 garbage-collected mid-flight. 

27* **Best-effort.** Typing is purely cosmetic; a failure must never propagate or 

28 crash a hop. Every error is logged and swallowed. ``CancelledError`` is a 

29 ``BaseException`` (not caught), so shutdown cancellation stays clean. 

30 

31**Identity & permissions.** The indicator is triggered as the **bot user**, not 

32as the agent persona — Discord webhooks (which the personas use) cannot type, 

33so the indicator always shows the bot's own name. Triggering typing also 

34requires the bot user to hold ``Send Messages`` in the channel (and 

35``Send Messages in Threads`` for ``/task`` threads), which is **separate** from 

36the ``Manage Webhooks`` permission the persona sender relies on. A bot granted 

37only ``Manage Webhooks`` gets ``403 Forbidden`` on every typing call; that is 

38swallowed (the indicator simply never shows), with the first occurrence logged 

39at WARNING so the misconfiguration is discoverable. 

40""" 

41from __future__ import annotations 

42 

43import asyncio 

44import contextlib 

45import logging 

46 

47import discord 

48 

49logger = logging.getLogger(__name__) 

50 

51 

52class TypingNotifier: 

53 """Fire-and-forget Discord typing indicators, best-effort. 

54 

55 Construct once per bridge process from a started, REST-capable 

56 :class:`discord.Client` (e.g. :attr:`DiscordPersonaSender.client`) and call 

57 :meth:`fire` with the id of the channel (or thread) the indicator should 

58 appear in. See the module docstring for the fire-and-forget rationale and 

59 the bot-user permission requirement. 

60 """ 

61 

62 def __init__(self, client: discord.Client) -> None: 

63 self._client = client 

64 self._tasks: set[asyncio.Task[None]] = set() 

65 # Channels we've already warned about a typing 403 in — warn once per 

66 # channel (bounded by the guild's channel count) so a per-channel 

67 # permission override stays discoverable, without per-hop log spam. 

68 self._forbidden_channels: set[int] = set() 

69 logger.info("typing indicator enabled") 

70 

71 def fire(self, channel_id: int) -> None: 

72 """Schedule a typing indicator for ``channel_id`` and return immediately. 

73 

74 Synchronous and non-blocking: the REST call runs on a detached task, so 

75 a rate-limit sleep can never stall the caller (notably the serial 

76 progress renderer). Safe to call from inside a ``try`` whose 

77 ``except`` would otherwise misread a typing error as a handler failure — 

78 any error lives on the detached task, not the caller. 

79 """ 

80 try: 

81 task = asyncio.create_task(self._send(channel_id)) 

82 except RuntimeError: 

83 # No running event loop to schedule onto. Keeps the "fire never 

84 # raises into its caller" contract unconditional (every real call 

85 # site runs inside the bridge's loop, so this is belt-and-braces). 

86 logger.debug("typing: no running event loop; skipping channel_id=%d", channel_id) 

87 return 

88 self._tasks.add(task) 

89 task.add_done_callback(self._tasks.discard) 

90 

91 async def _send(self, channel_id: int) -> None: 

92 """Issue the single ``POST /channels/{id}/typing`` call, swallowing errors. 

93 

94 Tiered so a *cosmetic* failure never crashes a hop, yet a real bug is 

95 not buried: permission denials warn once per channel, transient Discord 

96 errors stay at DEBUG, and anything unexpected (a mis-wired client, a 

97 programming error) is logged with a stack trace. 

98 """ 

99 try: 

100 # discord.py exposes no public one-shot "trigger typing" for a bare 

101 # channel id (``Messageable.typing()`` is a repeating context 

102 # manager); ``http.send_typing`` is the intended primitive, and is 

103 # stable across discord.py 2.x. 

104 await self._client.http.send_typing(channel_id) 

105 except discord.Forbidden: 

106 # Missing Send Messages / Send Messages in Threads (separate from the 

107 # Manage Webhooks the personas use). Operator-actionable, so warn — 

108 # but once per channel, since typing fires on every hop. 

109 if channel_id not in self._forbidden_channels: 

110 self._forbidden_channels.add(channel_id) 

111 logger.warning( 

112 "typing indicator forbidden in channel_id=%d; the bot user needs " 

113 "Send Messages (and Send Messages in Threads for task threads) here, " 

114 "separate from Manage Webhooks. Typing is skipped where denied.", 

115 channel_id, 

116 ) 

117 else: 

118 logger.debug("typing: forbidden channel_id=%d (suppressed)", channel_id) 

119 except discord.DiscordException as e: 

120 # Transient Discord-side failure (5xx, rate-limit exhausted, channel 

121 # gone). Cosmetic and self-resolving, and typing fires per hop, so 

122 # DEBUG keeps the noise down. 

123 logger.debug("typing: send_typing failed channel_id=%d: %s", channel_id, e) 

124 except Exception: 

125 # Not a Discord API error — a mis-wired client, a session closed at 

126 # teardown, or a genuine bug. Swallowed so a cosmetic call can't 

127 # crash a hop, but logged loudly so it is never invisible. 

128 logger.exception("typing: unexpected error firing typing for channel_id=%d", channel_id) 

129 

130 async def aclose(self) -> None: 

131 """Cancel any in-flight typing tasks. Idempotent; called on bridge teardown.""" 

132 tasks = list(self._tasks) 

133 for task in tasks: 

134 task.cancel() 

135 for task in tasks: 

136 with contextlib.suppress(asyncio.CancelledError): 

137 await task
