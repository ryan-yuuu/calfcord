1"""Discord auto-discovery for the ``calfcord init`` wizard (design §4.5). 

2 

3This module is a *library* — it has no CLI subcommand of its own. The init wizard composes it to 

4replace the old "paste a numeric ID" prompts with discovery: validate the bot token the instant it 

5is pasted, hand the user a ready-made invite link, wait until the bot actually appears in a server, 

6then offer pick-lists of the user's guilds and *postable* channels. 

7 

8Why "postable", not "visible": a bot can *see* a channel (View Channel) yet be unable to say 

9anything in it (no Send Messages, or no Manage Webhooks — which the persona webhooks require). A 

10pick-list filtered by visibility would happily seat the default agent in a channel where it can 

11never reply, producing the worst onboarding outcome: a green light that lies. So the channel filter 

12computes the bot's *effective* permission for each channel using Discord's documented overwrite 

13algorithm and keeps only channels where it can both send and manage webhooks. Channels the bot can 

14see but not post in are surfaced separately so the wizard can explain the gap instead of hiding it. 

15 

16Secrets: the bot token is handled exactly as ``doctor`` handles it — it rides only in the 

17``Authorization`` header and is NEVER placed in a return value, a message, or an exception. The 

18underlying httpx exception text (which could echo a request) is never propagated; failures are 

19re-raised as this module's own typed errors with fixed, token-free messages so the wizard can branch 

20(rejected vs. rate-limited vs. unreachable vs. join-timeout) without risk of leaking the token. 

21 

22The httpx client is injected via ``client_factory`` (as in ``doctor._discord_username``) so tests 

23drive it with ``httpx.MockTransport`` and no real Discord is contacted; ``poll_until_joined`` injects 

24``clock``/``sleep`` (as in ``supervisor/lifecycle.py``) so its backoff/timeout runs instantly. 

25"""

44# Permission bits (https://discord.com/developers/docs/topics/permissions). Only the handful we 

45# reason about are named; the rest of the 64-bit mask is irrelevant to "can this bot post here". 

46_VIEW_CHANNEL = 1 << 10 

47_SEND_MESSAGES = 1 << 11 

48_ADMINISTRATOR = 1 << 3 

49_MANAGE_WEBHOOKS = 1 << 29 

50 

51# What the persona-reply path actually requires in a channel: View Channel (without it the bot can't 

52# see the channel at all — and Send/Manage are dead bits), Send Messages (the bot user speaks), AND 

53# Manage Webhooks (the bridge posts agent replies as persona webhooks). All three, not any subset: 

54# a channel granting Send|Manage while denying View is one the bot can never reply in, so classifying 

55# it postable would be a green light that lies. 

56_POST_REQUIRED = _VIEW_CHANNEL | _SEND_MESSAGES | _MANAGE_WEBHOOKS 

57 

58# Permission overwrite target types in the channel object (`permission_overwrites[].type`). 

59_OVERWRITE_ROLE = 0 

60_OVERWRITE_MEMBER = 1 

61 

62# Channel types we treat as message targets. Only standard guild text (0) is offered; voice, 

63# categories, threads, forums, etc. are not where a default agent should be seated. 

64_TEXT_CHANNEL = 0 

65 

66# The invite bitmask granted by the canonical invite link (kept in lock-step with 

67# docs/discord-setup.md's `permissions=...`). It is a superset of `_POST_REQUIRED`; a guard test

183def _raise_for_discord_status(resp: httpx.Response, path: str) -> None: 

184 """Map an HTTP status to a typed, token-free error. 2xx returns; everything else raises.""" 

185 code = resp.status_code

187 return 

188 if code in (401, 403): # token not accepted -> fatal; re-prompting won't help

255def poll_until_joined( 

256 token: str, 

257 *, 

258 client_factory: Callable[[], httpx.Client] | None = None, 

259 clock: Callable[[], float] | None = None, 

260 sleep: Callable[[float], None] | None = None, 

261 timeout_s: float = _DEFAULT_POLL_TIMEOUT_SECONDS, 

262 interval_s: float = _DEFAULT_POLL_INTERVAL_SECONDS, 

263) -> list[Guild]: 

264 """Poll ``GET /users/@me/guilds`` until the bot is in ≥1 guild, returning them. 

265 

266 The Discord detour is the one place the wizard must wait on the *user* (open browser, pick a 

267 server, click Authorize), so this blocks-and-polls and auto-advances the instant the bot 

268 appears. Resilience matches that intent: a 429 backs off by Retry-After (inheriting doctor's 

269 rate-limit tolerance) and a transient transport blip is retried — neither aborts the wait. A 

270 *rejected token*, by contrast, is fatal and fails fast (spinning would never recover). If the 

271 bot never joins within ``timeout_s``, raises :class:`DiscordJoinTimeoutError` (the user likely never 

272 authorized) so the wizard can surface the common causes rather than hanging forever. 

273 

274 ``clock``/``sleep`` are injected (default ``time.monotonic``/``time.sleep``) so tests run the 

275 backoff/timeout in virtual time. 

276 """ 

277 clock = clock or time.monotonic 

278 sleep = sleep or time.sleep 

279 deadline = clock() + timeout_s 

280 

281 while True: 

282 try: 

283 guilds = _parse_guilds(_get("/users/@me/guilds", token, client_factory=client_factory)) 

284 except DiscordRateLimitedError as exc: 

285 wait = _seconds_from(exc.args, _DEFAULT_RETRY_AFTER_SECONDS) 

286 except DiscordUnavailableError: 

287 # A transient blip (connect reset, brief 5xx) during a multi-minute wait is expected; 

288 # keep polling at the normal cadence rather than aborting the whole detour. 

289 wait = interval_s 

290 else: 

291 if guilds: 

292 return guilds 

293 wait = interval_s # joined zero guilds yet -> keep waiting at the normal cadence 

294 

295 # Stop *before* sleeping past the deadline, so the budget is honored to the second and we 

296 # never sleep through a timeout we've already crossed. 

297 if clock() + wait > deadline: 

298 raise DiscordJoinTimeoutError( 

299 f"bot did not join a server within {timeout_s:g}s — " 

300 "open the invite link, pick a server, and click Authorize" 

301 ) 

302 sleep(wait)

346def list_postable_channels( 

347 token: str, 

348 guild_id: str | int, 

349 *, 

350 client_factory: Callable[[], httpx.Client] | None = None, 

351) -> ChannelListing: 

352 """List a guild's text channels, split by whether the bot can actually *post* in each. 

353 

354 "Postable" means the bot's *effective* permission for the channel includes both Send Messages 

355 and Manage Webhooks (or it is the guild owner / has Administrator, which bypass everything). 

356 Effective permission is computed from the bot's guild-level base permission plus the channel's 

357 overwrites, following Discord's documented overwrite order. Non-text channels are excluded 

358 entirely (they are not message targets); text channels the bot can see but not post in land in 

359 ``unpostable`` so the gap is explainable. 

360 

361 Raises :class:`DiscordUnavailableError` if the bot is not in / cannot see the given guild (it would 

362 otherwise compute permissions against an absent base). 

363 """ 

364 guild_id = str(guild_id) 

365 

366 # The bot's user id — needed to apply the member-specific channel overwrite (the highest- 

367 # precedence overwrite), and it's the cheapest call that also re-validates the token. 

368 bot_user_id = verify_bot_identity(token, client_factory=client_factory).id 

369 

370 # Base (guild-level) permissions + owner flag come from the partial guild object; this is 

371 # Discord's already-computed base permission, so we don't re-fetch+fold roles ourselves. 

372 guild = _find_guild(list_guilds(token, client_factory=client_factory), guild_id) 

373 

374 # The bot's role ids in this guild — needed to apply role-level channel overwrites. 

375 # Use the bot's explicit user id: GET /guilds/{id}/members/@me is a USER-OAuth route 

376 # (needs the guilds.members.read scope) and returns HTTP 400 for a BOT token — only 

377 # GET /guilds/{id}/members/{user.id} works with a bot token (verified against live Discord). 

378 member = _get(f"/guilds/{guild_id}/members/{bot_user_id}", token, client_factory=client_factory) 

379 role_ids = _member_role_ids(member) 

380 

381 channels = _get(f"/guilds/{guild_id}/channels", token, client_factory=client_factory) 

382 postable: list[PostableChannel] = [] 

383 unpostable: list[PostableChannel] = [] 

384 for raw in _iter_dicts(channels): 

385 if raw.get("type") != _TEXT_CHANNEL: 

386 continue # only standard text channels are message targets 

387 effective = _effective_channel_permissions( 

388 base=guild.base_permissions, 

389 owner=guild.owner, 

390 guild_id=guild_id, 

391 bot_user_id=bot_user_id, 

392 role_ids=role_ids,

394 )

407def _effective_channel_permissions( 

408 *, 

409 base: int, 

410 owner: bool, 

411 guild_id: str, 

412 bot_user_id: str, 

413 role_ids: frozenset[str], 

414 overwrites: Any, 

415) -> int: 

416 """Apply Discord's channel-overwrite algorithm to a base permission integer. 

417 

418 Order (per the Discord permissions reference): owner/Administrator → ALL; otherwise apply the 

419 @everyone overwrite (keyed by guild id), then the union of the bot's role overwrites, then the 

420 bot's member-specific overwrite — each as "clear deny bits, then set allow bits", with later 

421 stages winning. Implemented here (rather than pulled from discord.py) to keep the wizard path 

422 import-light and the rule unit-testable in isolation. 

423 """ 

424 if owner or base & _ADMINISTRATOR: 

425 return _ALL_PERMISSIONS 

426 

427 by_target = _index_overwrites(overwrites) 

428 permissions = base 

429 

430 everyone = by_target.get((_OVERWRITE_ROLE, guild_id)) 

431 if everyone: 

432 permissions = (permissions & ~everyone[1]) | everyone[0] 

433 

434 role_allow = 0 

435 role_deny = 0 

436 for role_id in role_ids: 

437 ov = by_target.get((_OVERWRITE_ROLE, role_id)) 

438 if ov: 438 ↛ 436line 438 didn't jump to line 436 because the condition on line 438 was always true

439 role_allow |= ov[0] 

440 role_deny |= ov[1] 

441 permissions = (permissions & ~role_deny) | role_allow 

442 

443 member = by_target.get((_OVERWRITE_MEMBER, bot_user_id)) 

444 if member:
