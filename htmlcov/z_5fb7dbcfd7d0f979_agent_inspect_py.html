1"""``disco agent list`` / ``disco agent show`` — read-only agent inspection. 

2 

3These two commands let an operator survey the install ("which agents exist, on 

4what model, with how many tools?") and drill into one ("show me every field this 

5agent declares") without opening the ``.md`` by hand. They are deliberately 

6**non-interactive** (no :class:`~calfcord.cli._prompts.Prompter`): both are pure 

7functions of ``agents_dir`` + a name, so they compose into scripts and pipes and 

8stay trivially testable. 

9 

10Both render through the shared :data:`calfcord.cli._fields.FIELDS` registry and 

11:func:`calfcord.cli._fields.render_value`, the same source the ``edit`` menu and 

12``set`` command consume. Routing the *display* of a field through the same 

13registry that *edits* it is what stops ``show`` from listing a field ``set`` 

14can't write (or omitting one it can) — the two surfaces can't drift because 

15there is only one list. 

16 

17The skip rules match :func:`calfcord.cli._agents.detect_agents` (dotfiles and 

18``*.template.md`` templates are not live agents), so ``list`` reports exactly the 

19set ``calfkit-agent`` would run. An individual ``.md`` that fails to parse is 

20*noted* rather than fatal: one malformed file must not blank the whole listing. 

21 

22This module imports only the lightweight ``calfcord.agents`` / ``calfcord.cli`` 

23seams (no provider SDK), so it stays importable from the argparse entry point. 

24"""

40# The DESCRIPTION column in the ``list`` table is truncated to this many 

41# characters so a near-100-char description (Discord's slash limit) can't wrap 

42# the row on a typical terminal. The full text is always available via 

43# ``show``/``--json``. 

44_DESCRIPTION_TRUNCATE = 48 

45 

46# How many leading characters of the system-prompt body ``show`` previews in its 

47# human form. Long enough to recognize which prompt is loaded; the full body is 

48# in ``--json``. 

49_PROMPT_PREVIEW_LEN = 200

52def _tools_summary(tools: tuple[str, ...] | None) -> str: 

53 """Summarize an agent's ``tools`` for the ``list`` table's TOOLS column. 

54 

55 Mirrors the loader's omitted/empty/explicit semantics 

56 (:attr:`AgentDefinition.tools`): ``None`` (frontmatter ``tools:`` omitted) 

57 means "all builtins" → ``"all"``; an explicit empty tuple is the deliberate 

58 no-tools opt-out → ``"0"``; otherwise the count. A scalar the operator can 

59 scan beats a wrapping comma list in a table cell. 

60 """ 

61 if tools is None: 

62 return "all" 

63 return str(len(tools))

81def _list_row(defn: AgentDefinition) -> dict[str, Any]: 

82 """Build the per-agent JSON object for ``list --json``. 

83 

84 ``tools`` is serialized as the concrete list (or ``null`` for the implicit 

85 "all builtins" default) rather than the table's summary string, so the JSON 

86 consumer gets the structured value while the human table gets the scannable 

87 count. This is the machine-readable counterpart of one table row. 

88 """ 

89 return { 

90 "name": defn.agent_id, 

91 "provider": defn.provider, 

92 "model": defn.model, 

93 "tools": list(defn.tools) if defn.tools is not None else None, 

94 "description": defn.description, 

95 }

98def _parse_all(agents_dir: Path) -> tuple[list[AgentDefinition], list[str]]: 

99 """Parse every detected agent, returning ``(parsed, failed_names)``. 

100 

101 A single malformed ``.md`` must not blank the whole listing, so a parse 

102 failure is collected into ``failed_names`` (for a trailing note) rather than 

103 raised. The parsed list preserves :func:`detect_agents`' sorted order for 

104 deterministic output. 

105 """ 

106 parsed: list[AgentDefinition] = [] 

107 failed: list[str] = [] 

108 for name in detect_agents(agents_dir): 

109 try: 

110 parsed.append(parse_agent_md(agents_dir / f"{name}.md")) 

111 except (ValueError, OSError): 

112 failed.append(name) 

113 return parsed, failed

116def run_list(agents_dir: Path, *, as_json: bool = False) -> int: 

117 """``disco agent list``: survey every agent in ``agents_dir``. 

118 

119 Parses each detected agent (skipping dotfiles / ``*.template.md`` templates, 

120 matching the loader) and prints either an aligned NAME / PROVIDER·MODEL / 

121 TOOLS / DESCRIPTION table (human) or a JSON array of agent objects 

122 (``--json``). A ``.md`` that fails to parse is noted on stderr-style trailing 

123 line in the human form (and simply omitted from the JSON array) so one broken 

124 file doesn't hide the rest. An empty directory prints a friendly line (human) 

125 or ``[]`` (JSON). Always returns 0 — listing is read-only and "no agents" is 

126 a valid state, not an error. 

127 """ 

128 parsed, failed = _parse_all(agents_dir) 

129 

130 if as_json: 

131 print(json.dumps([_list_row(d) for d in parsed], indent=2)) 

132 return 0 

133 

134 if not parsed: 

135 print(f"no agents in {agents_dir}") 

136 if failed: 136 ↛ 137line 136 didn't jump to line 137 because the condition on line 136 was never true

137 print(f"(skipped {len(failed)} unparseable file(s): {', '.join(failed)})") 

138 return 0

140 _print_table(parsed) 

141 if failed: 

142 print(f"(skipped {len(failed)} unparseable file(s): {', '.join(failed)})") 

143 return 0

146def _print_table(agents: list[AgentDefinition]) -> None: 

147 """Print the aligned ``list`` table for ``agents``. 

148 

149 Column widths are computed from the data (and the headers) so the table 

150 aligns regardless of name/model lengths, then each row is left-justified into 

151 those widths. The DESCRIPTION column is truncated last (its value can be the 

152 longest) so it never forces a wide, wrapping table. 

153 """ 

154 headers = ("NAME", "PROVIDER·MODEL", "TOOLS", "DESCRIPTION") 

155 rows = [ 

156 ( 

157 d.agent_id, 

158 _provider_model(d), 

159 _tools_summary(d.tools), 

160 truncate(d.description, _DESCRIPTION_TRUNCATE), 

161 ) 

162 for d in agents 

163 ] 

164 

165 widths = [ 

166 max(len(headers[col]), *(len(row[col]) for row in rows)) 

167 for col in range(len(headers)) 

168 ] 

169 fmt = " ".join(f"{{:<{w}}}" for w in widths) 

170 print(fmt.format(*headers)) 

171 for row in rows: 

172 print(fmt.format(*row))

175def run_show(agents_dir: Path, name: str, *, as_json: bool = False) -> int: 

176 """``disco agent show <name>``: print one agent's full configuration. 

177 

178 Parses ``agents_dir/<name>.md``; a missing or unparseable file prints an 

179 ``error: ...`` line and returns 1 (operator-recoverable, per the CLI 

180 convention — no traceback escapes). In human form it prints the name + file 

181 path, then every :data:`FIELDS` row rendered through 

182 :func:`render_value` (the same renderer the edit menu uses, so ``show`` and 

183 ``edit`` can't disagree), plus a short system-prompt preview. ``--json`` 

184 emits the full config object — name, every editable field, and the complete 

185 ``system_prompt`` body — so it round-trips into tooling without the human 

186 previews' truncation. Returns 0 on success. 

187 """ 

188 md_path = agents_dir / f"{name}.md" 

189 try: 

190 defn = parse_agent_md(md_path) 

191 except (ValueError, OSError) as e: 

192 print(f"error: cannot read agent {name!r}: {e}") 

193 return 1 

194 

195 if as_json: 

196 print(json.dumps(_show_object(defn), indent=2)) 

197 return 0 

198 

199 print(f"{defn.agent_id} ({md_path})") 

200 label_width = max(len(field.label) for field in FIELDS) 

201 for field in FIELDS: 

202 print(f" {field.label:<{label_width}} {render_value(defn, field)}") 

203 print() 

204 print("System prompt:") 

205 print(f" {truncate(defn.system_prompt, _PROMPT_PREVIEW_LEN)}") 

206 return 0

210 """Build the full-config JSON object for ``show --json``. 

211 

212 Includes ``name`` plus every editable field and the complete 

213 ``system_prompt`` body (not the human preview) so the output is a faithful, 

214 machine-readable image of the agent's declaration. ``tools`` is the concrete 

215 list (or ``null`` for the implicit all-builtins default), preserving the 

216 omitted/empty distinction the table's summary string would lose. 

217 """ 

218 return { 

219 "name": defn.agent_id, 

220 "description": defn.description, 

221 "provider": defn.provider, 

222 "model": defn.model, 

223 "tools": list(defn.tools) if defn.tools is not None else None, 

224 "thinking_effort": defn.thinking_effort, 

225 "memory": defn.memory, 

226 "system_prompt": defn.system_prompt,
