1"""Render slim per-agent / per-tool Dockerfiles. 

2 

3The shape mirrors the project's canonical ``Dockerfile`` exactly, 

4varying these five things: 

5 

61. The runtime stage's ``apt-get install`` list — narrowed by the 

7 per-tool OS-dep mapping (e.g. no ``tmux`` if ``shell`` isn't 

8 included, no ``ripgrep`` if neither ``grep`` nor ``glob`` is). 

92. The builder's ``COPY agents`` line — for per-agent images, COPYs 

10 only the selected ``agents/<name>.md`` files rather than the whole 

11 directory. For per-tool images, the line is OMITTED entirely (a 

12 tools-only worker has no use for agent definitions). 

133. The runtime ``ENV`` block — bakes ``CALFCORD_TOOLS_INCLUDE`` for 

14 per-tool images so the auto-discovery loader filters to just those 

15 tools at boot. Optionally also bakes ``CALFCORD_TOOLS_ALIAS`` (per 

16 ``--rename`` on ``calfcord-package-tools``) so the same Python tool 

17 body registers under a different schema name in this image — used 

18 for multi-host deployments of the same tool. Always bakes 

19 ``OPENHANDS_SUPPRESS_BANNER=1`` so boot logs aren't drowned in the 

20 openhands SDK ASCII banner. 

214. The default ``CMD`` — ``calfkit-tools`` for tools images and 

22 ``calfkit-agent`` for agents images. The canonical Dockerfile 

23 defaults to ``calfkit-bridge`` because it's the only entry point 

24 that makes sense for an all-in-one image; the slim images each 

25 default to the runner that matches their payload. 

265. The generated banner comment at the top — names the inputs that 

27 produced the image so a copy on disk is self-describing without 

28 having to re-derive them from the ``ENV`` block. 

29 

30If the canonical ``Dockerfile`` is ever restructured, the constants in 

31this module must be updated to match. Tests in 

32``tests/packaging/test_dockerfile.py`` assert structural properties 

33of the output so drift fails CI. 

34 

35Pure Python, no external deps — the CLIs in this package can invoke 

36the templater without spinning up the full ``TOOL_REGISTRY``. 

37"""

43# Always-on OS packages for TOOLS images. ``ca-certificates`` is needed 

44# by any tool that talks HTTPS (web_fetch, web_search, and any 

45# third-party tool an operator might add later). ``git`` is kept because 

46# the ``shell`` tool's most common ergonomic use — agents asking for 

47# ``git status`` / ``git log`` — only works if the binary is present. 

48# Removing either to save image size would trade a few tens of MB for 

49# "works in dev, fails in prod" surprises. 

50_ALWAYS_ON_TOOL_OS_DEPS: tuple[str, ...] = ("ca-certificates", "git") 

51 

52# Always-on OS packages for AGENTS images. Narrower than the tools set: 

53# agent containers don't host tool bodies, so ``git`` (plus its 

54# transitive Debian deps — perl, libs, ~30MB installed) is dead weight. 

55# Keep ``ca-certificates`` because the LLM-provider HTTP clients 

56# (anthropic, openai SDKs) need a trust store. 

57_ALWAYS_ON_AGENT_OS_DEPS: tuple[str, ...] = ("ca-certificates",) 

58 

59# Per-tool OS-dep mapping. Tools not listed here imply no extra OS 

60# packages — they're pure Python. When adding a new tool that needs an 

61# OS binary, add it here AND to the canonical Dockerfile so the 

62# all-in-one image keeps working. 

63_TOOL_OS_DEPS: dict[str, tuple[str, ...]] = { 

64 "shell": ("tmux",), 

65 "grep": ("ripgrep",), 

66 "glob": ("ripgrep",), 

67 # web_fetch / web_search only need ca-certificates, which is 

68 # always-on. Listed explicitly so the table is exhaustive. 

69 "web_fetch": (), 

70 "web_search": (), 

71 # FS tools and the in-memory tools have no extra OS deps. 

72 "read_file": (), 

73 "write_file": (), 

74 "edit_file": (), 

75 "todo_view": (), 

76 "todo_write": (), 

77 "private_chat": (), 

78}

81def os_deps_for_tools(include_tools: Iterable[str]) -> list[str]: 

82 """Return the sorted union of OS package names needed for the given tools. 

83 

84 Includes :data:`_ALWAYS_ON_TOOL_OS_DEPS` regardless of the tool 

85 list. Unknown tool names are silently skipped — validation happens 

86 at the CLI layer, not here, so the templater stays loose-coupled 

87 to the live ``TOOL_REGISTRY``. 

88 """ 

89 deps: set[str] = set(_ALWAYS_ON_TOOL_OS_DEPS) 

90 for tool in include_tools: 

91 deps.update(_TOOL_OS_DEPS.get(tool, ())) 

92 return sorted(deps)

95def _render_apt_install(packages: Iterable[str]) -> str: 

96 """Render the ``RUN apt-get install`` block with one package per line. 

97 

98 Indentation matches the canonical Dockerfile (8 spaces) so the 

99 output diffs cleanly against the source-of-truth file. 

100 """ 

101 lines = ["RUN apt-get update \\", " && apt-get install -y --no-install-recommends \\"] 

102 for pkg in packages: 

103 lines.append(f" {pkg} \\") 

104 lines.append(" && rm -rf /var/lib/apt/lists/*") 

105 return "\n".join(lines)

108def _generated_header( 

109 *, 

110 kind: str, 

111 names: Iterable[str], 

112 aliases: dict[str, str] | None = None, 

113) -> str: 

114 """Banner at the top of every generated Dockerfile. 

115 

116 Tells the reader this file is build-tool output, names the inputs 

117 that produced it (so a copy on disk is self-describing), and 

118 points at the canonical source for any non-templated changes. 

119 

120 When ``aliases`` is non-empty, the header surfaces the alias map 

121 so a reader inspecting the image (or a copy of the file pulled 

122 out for forensics) can see the rename without having to decode 

123 the ENV block. 

124 """ 

125 name_list = ", ".join(sorted(names)) 

126 alias_line = "" 

127 if aliases: 

128 alias_pairs = ", ".join(f"{src} → {dst}" for src, dst in sorted(aliases.items())) 

129 alias_line = f"# Renames: {alias_pairs}\n" 

130 return ( 

131 f"# Generated by calfcord-package-{kind} for: {name_list}\n" 

132 f"{alias_line}" 

133 f"# DO NOT EDIT by hand — regenerate via the CLI. The canonical\n" 

134 f"# source is the project's top-level ``Dockerfile``; this file\n" 

135 f"# is templated from ``src/calfcord/packaging/dockerfile.py``.\n" 

136 )

139def render_tools_dockerfile( 

140 *, 

141 include_tools: list[str], 

142 aliases: dict[str, str] | None = None, 

143) -> str: 

144 """Render a Dockerfile for an image hosting only ``include_tools``. 

145 

146 Args: 

147 include_tools: Tool schema names to include (post-rename, if 

148 any aliases are passed). Validation against ``TOOL_REGISTRY`` 

149 is the CLI's job; this function trusts its input. 

150 aliases: Optional ``{src: dst}`` map baked into the image as 

151 ``CALFCORD_TOOLS_ALIAS=src1=dst1,src2=dst2``. The discovery 

152 loader at runtime clones each ``src`` tool's ``ToolNodeDef`` 

153 under the ``dst`` name with all four name-bound fields 

154 rewritten. Pairs naturally with ``include_tools`` containing 

155 the ``dst`` names to give true rename behavior — the 

156 original ``src`` drops out of the filter, only the clone 

157 survives. Empty/``None`` produces no ``CALFCORD_TOOLS_ALIAS`` 

158 line in the output. 

159 

160 Returns: 

161 The complete Dockerfile content as a string, ready to write 

162 to a tempdir and feed to ``docker buildx build``. 

163 """ 

164 apt_block = _render_apt_install(os_deps_for_tools(include_tools)) 

165 include_csv = ",".join(sorted(include_tools)) 

166 header = _generated_header( 

167 kind="tools", names=include_tools, aliases=aliases or None 

168 ) 

169 # Sort aliases by src for deterministic output — same input dict 

170 # contents in different insertion orders must produce byte-identical 

171 # Dockerfiles so build caches hit reliably. 

172 alias_csv = ( 

173 ",".join(f"{src}={dst}" for src, dst in sorted((aliases or {}).items())) 

174 if aliases 

175 else None 

176 ) 

177 # The ALIAS env line is conditionally appended below the always-on 

178 # ENV block. Wrapping in a leading backslash-newline keeps the 

179 # final ``\\`` on the previous line valid when present, and 

180 # produces zero output when absent — no dangling backslash to 

181 # confuse the Dockerfile parser. 

182 alias_env_line = ( 

183 f" \\\n CALFCORD_TOOLS_ALIAS={alias_csv}" if alias_csv else "" 

184 ) 

185 

186 # ``# syntax=`` MUST be on line 1 of the Dockerfile for Docker's 

187 # frontend selector to honor it. The generated banner comment goes 

188 # below the directive — Docker still treats the rest of the file 

189 # correctly, but the syntax directive's "first non-blank, 

190 # non-directive line" rule means the banner can't be above it. 

191 return f"""# syntax=docker/dockerfile:1.7 

192{header}# 

193# Per-tool calfcord image. Hosts ONLY: {include_csv} 

194# 

195# Tools subscribe to ``tool.<name>.input`` topics; this image's 

196# ``calfkit-tools`` worker subscribes only to the listed names because 

197# the ``CALFCORD_TOOLS_INCLUDE`` env var (baked below) filters the 

198# auto-discovery walk in ``tools/discovery.py``. 

199 

200 

201# ── builder ────────────────────────────────────────────────────────────────── 

202FROM python:3.12-slim AS builder 

203 

204COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /usr/local/bin/ 

205 

206WORKDIR /app 

207 

208ENV PYTHONDONTWRITEBYTECODE=1 

209 

210COPY pyproject.toml uv.lock README.md ./ 

211RUN --mount=type=cache,target=/root/.cache/uv \\ 

212 uv sync --frozen --no-dev --no-install-project 

213 

214COPY src ./src 

215RUN --mount=type=cache,target=/root/.cache/uv \\ 

216 uv sync --frozen --no-dev 

217 

218 

219# ── runtime ────────────────────────────────────────────────────────────────── 

220FROM python:3.12-slim AS runtime 

221 

222# OS packages narrowed to those the included tools actually use. 

223{apt_block} 

224 

225ARG UID=1000 

226ARG GID=1000 

227RUN groupadd --gid ${{GID}} calfcord \\ 

228 && useradd --uid ${{UID}} --gid ${{GID}} --create-home --shell /bin/bash calfcord 

229 

230WORKDIR /app 

231 

232# OPENHANDS_SUPPRESS_BANNER silences the openhands SDK's ASCII boot 

233# banner. The banner is operationally noise (printed on every import 

234# of the openhands package, before any calfcord work begins) and 

235# pollutes both compose-logs output and ad-hoc ``docker run`` sessions 

236# operators use to inspect the registry. Operators who want the banner 

237# can restore it with ``-e OPENHANDS_SUPPRESS_BANNER=0`` at run time — 

238# Docker ``-e`` overrides image ``ENV``. 

239ENV PATH=/app/.venv/bin:$PATH \\ 

240 PYTHONUNBUFFERED=1 \\ 

241 PYTHONDONTWRITEBYTECODE=1 \\ 

242 OPENHANDS_SUPPRESS_BANNER=1 \\ 

243 CALFCORD_TOOLS_INCLUDE={include_csv}{alias_env_line} 

244 

245COPY --from=builder --chown=calfcord:calfcord /app /app 

246 

247USER calfcord 

248 

249# Per-tool images default to the tools entry point. The bridge / agent 

250# / router entry points are still on PATH (same image, same venv) but 

251# don't make sense here — running ``calfkit-bridge`` in a tools-only 

252# image would try to load ``agents/*.md`` which isn't in this image. 

253CMD ["calfkit-tools"] 

254"""

257def render_agents_dockerfile( 

258 *, 

259 include_agents: list[str], 

260) -> str: 

261 """Render a Dockerfile for an image hosting only ``include_agents``. 

262 

263 Args: 

264 include_agents: Agent names (filename stems, e.g. ``["scribe", 

265 "conan"]``). Validation against on-disk ``agents/<name>.md`` 

266 files is the CLI's job. 

267 

268 Returns: 

269 Complete Dockerfile content. Differs from 

270 :func:`render_tools_dockerfile`: 

271 

272 * No ``CALFCORD_TOOLS_INCLUDE`` (agent images don't host tools). 

273 * No ``tmux`` / ``ripgrep`` / ``git`` (agent containers don't 

274 execute tool bodies). 

275 * The ``COPY agents`` line names individual files instead of 

276 the whole directory, so the image's ``/app/agents/`` only 

277 contains the selected agents. 

278 """ 

279 # Agent images don't host tools, so no tool-driven OS deps. 

280 apt_block = _render_apt_install(_ALWAYS_ON_AGENT_OS_DEPS) 

281 header = _generated_header(kind="agents", names=include_agents) 

282 # COPY each agent's .md file individually. Sorted for deterministic 

283 # layer hashes — same inputs produce byte-identical Dockerfiles. 

284 # No explicit ``mkdir -p ./agents`` needed: Docker auto-creates 

285 # COPY destination directories. 

286 agent_copy_lines = "\n".join( 

287 f"COPY agents/{name}.md ./agents/{name}.md" for name in sorted(include_agents) 

288 ) 

289 

290 # ``# syntax=`` must be on line 1 — same constraint as the tools 

291 # render. The banner header goes below. 

292 return f"""# syntax=docker/dockerfile:1.7 

293{header}# 

294# Per-agent calfcord image. Hosts ONLY: {", ".join(sorted(include_agents))} 

295# 

296# The image's ``/app/agents/`` contains only the listed agent .md 

297# files. ``calfkit-agent`` loads whatever's present, so no runtime 

298# filter is needed — the filesystem IS the filter. 

299 

300 

301# ── builder ────────────────────────────────────────────────────────────────── 

302FROM python:3.12-slim AS builder 

303 

304COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /usr/local/bin/ 

305 

306WORKDIR /app 

307 

308ENV PYTHONDONTWRITEBYTECODE=1 

309 

310COPY pyproject.toml uv.lock README.md ./ 

311RUN --mount=type=cache,target=/root/.cache/uv \\ 

312 uv sync --frozen --no-dev --no-install-project 

313 

314COPY src ./src 

315RUN --mount=type=cache,target=/root/.cache/uv \\ 

316 uv sync --frozen --no-dev 

317 

318# Per-agent COPY: only the .md files explicitly named in the build 

319# command end up in the image. Agents not in the list literally 

320# don't exist inside this container. 

321{agent_copy_lines} 

322 

323 

324# ── runtime ────────────────────────────────────────────────────────────────── 

325FROM python:3.12-slim AS runtime 

326 

327# OS packages: agent images don't run tool bodies, so no tmux/ripgrep/git. 

328{apt_block} 

329 

330ARG UID=1000 

331ARG GID=1000 

332RUN groupadd --gid ${{GID}} calfcord \\ 

333 && useradd --uid ${{UID}} --gid ${{GID}} --create-home --shell /bin/bash calfcord 

334 

335WORKDIR /app 

336 

337# OPENHANDS_SUPPRESS_BANNER silences the openhands SDK's ASCII boot 

338# banner. Same rationale as the tools image — boot-log signal/noise. 

339ENV PATH=/app/.venv/bin:$PATH \\ 

340 PYTHONUNBUFFERED=1 \\ 

341 PYTHONDONTWRITEBYTECODE=1 \\ 

342 OPENHANDS_SUPPRESS_BANNER=1 

343 

344COPY --from=builder --chown=calfcord:calfcord /app /app 

345 

346USER calfcord 

347 

348CMD ["calfkit-agent"]
