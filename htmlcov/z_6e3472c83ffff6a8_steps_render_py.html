1"""Pure step-render helpers — no Discord, no Kafka, no state (spec §5.1). 

2 

3Two render surfaces share this module: 

4 

5* **Live progress** — :func:`render_step_line` turns ONE normalized 

6 :class:`~calfcord.bridge.step_events.StepEvent` into one compact line (model 

7 text as prose, a ``tool_name(args)`` inline-code line, a short fenced ``⎿`` 

8 result block). :func:`_tail_window` / :func:`_progress_content` shape an 

9 accumulated list of those lines into the transient progress message, bounded 

10 to Discord's per-message cap. The stateful lifecycle (post / debounced edit / 

11 delete) lives in :mod:`calfcord.bridge.progress`. 

12* **Full transcript** — :func:`_render_tree_blocks` projects a turn's 

13 ``message_history`` slice into the Claude-Code-style ``● tool(args)`` / 

14 ``⎿ result`` blocks behind the reply's ``⤵ steps`` button 

15 (:mod:`calfcord.bridge.steps_toggle`); its block COUNT is reused as the 

16 ``⤵ N steps`` label. This surface still operates on 

17 ``Sequence[ModelMessage]`` because it renders persisted deltas, and its 

18 output is byte-for-byte stable so stored transcripts keep rendering the same. 

19 

20Everything here is pure: no I/O, no time, no mutable module state. That keeps 

21both surfaces trivially unit-testable and lets the live source change (it now 

22arrives as :class:`StepEvent` from the run ``stream()``, not a Kafka envelope) 

23without touching the renderers. 

24"""

49# --- Live progress body rendering ------------------------------------------- 

50# The transient progress message shows the agent's ACTUAL work as it streams 

51# (model text, tool calls, tool returns) — not just a counter. It is edited in 

52# place, and Discord hard-caps message content at 2000 chars, so the body is a 

53# bounded *tail window*: the most recent lines that fit, with a marker when 

54# older steps are elided. The full, untruncated transcript remains on the 

55# reply's ``⤵ steps`` button after completion. These caps apply ONLY to this 

56# live view; the durable transcript / toggle renderer 

57# (:func:`_render_tree_blocks`) is untouched. 

58 

59_DISCORD_MESSAGE_LIMIT: Final[int] = 2000 

60"""Discord's hard per-message content cap. The rendered progress body is 

61clamped under this so an ``edit_message`` never fails with a 400-too-long.""" 

62 

63_PROGRESS_BODY_MAX_CHARS: Final[int] = 1900 

64"""Tail-window budget for the progress body (the whole message — there is no 

65header line). Kept under :data:`_DISCORD_MESSAGE_LIMIT` to leave room for the 

66elision marker prepended when older lines are dropped; a final slice in 

67:func:`_progress_content` enforces the hard cap regardless.""" 

68 

69_LIVE_TEXT_MAX_CHARS: Final[int] = 1000 

70"""Per-part cap for a single model text part in the live body. Generous (it's 

71prose) but bounded so one long preamble can't dominate the window.""" 

72 

73_LIVE_TOOL_MAX_CHARS: Final[int] = 200 

74"""Per-part cap for a single ``tool_name(args)`` call line in the live body. 

75Tight on purpose — a tool call must never exceed one line, and short lines let 

76more recent steps stay visible in the tail window. The result line has its own 

77(multi-line) caps below.""" 

78 

79_LIVE_RETURN_MAX_LINES: Final[int] = 3 

80"""Max real lines kept from a tool result in the live body. The result is the 

81noisiest part; a handful of lines gives useful context while staying compact. 

82Beyond this the last kept line carries a ``… (truncated)`` marker — the full 

83result lives on the ⤵ transcript.""" 

84 

85_LIVE_RETURN_LINE_MAX_CHARS: Final[int] = 120 

86"""Per-line cap for a kept live-result line, so one very long line can't blow 

87the compact window (and also marks the result truncated).""" 

88 

89_HIDDEN_STEPS_MARKER: Final[str] = "… *(earlier steps hidden — full trace on the ⤵ button)*" 

90"""Prepended to the body when the tail window dropped older lines.""" 

91 

92_PROGRESS_DEBOUNCE_SECONDS: Final[float] = 1.0 

93"""Trailing-debounce window for progress-message edits. Coalesces a 

94burst of steps into one Discord ``edit_message`` call so a fast tool loop 

95doesn't hammer the per-channel webhook rate bucket (5 req / 2 s, shared 

96by co-tenant agents). The first renderable step posts immediately; only 

97subsequent edits are debounced. Consumed by :mod:`calfcord.bridge.progress`."""

99# --- Full transcript tree renderer (the ⤵ steps expand view) ---------------- 

100# Renders the turn's steps as a Claude-Code-style trace: model text as prose, 

101# each tool call as ``● tool(args)`` with its result nested under ``⎿``. Unlike 

102# the live view this is the FULL trace — NO per-part truncation; the only bound 

103# is Discord's message cap, beyond which steps_toggle attaches the whole render 

104# as a steps.md file. One string per visual block (a prose block, or a 

105# call+return pair); the block COUNT is reused by the reply poster as the 

106# ``⤵ N steps`` label, so a tool call and its result count as ONE step. 

107 

108_TREE_CALL_MARKER: Final[str] = "●" 

109_TREE_RETURN_MARKER: Final[str] = "⎿" 

110 

111_TRIPLE_BACKTICK_RUN: Final[re.Pattern[str]] = re.compile(r"`{3,}")

144def _inline_code(inner: str) -> str: 

145 """Wrap ``inner`` in Discord inline code, neutralizing backticks. 

146 

147 Inline code cannot contain a backtick (it would close the span early), so 

148 any backtick in tool args/results is swapped for an apostrophe. This is a 

149 live preview; the exact bytes are preserved in the durable transcript. 

150 """ 

151 return f"`{inner.replace('`', chr(39))}`"

154def _fence_safe(content: str) -> str: 

155 """Neutralize runs of 3+ backticks so embedded fences can't break out. 

156 

157 Discord closes a ``` code block at the next run of three-or-more 

158 backticks regardless of the opening fence's length, so a triple-backtick 

159 inside tool output would otherwise terminate the block early and spill the 

160 remainder as raw markdown. Weaving a zero-width space between the backticks 

161 of any 3+ run leaves the text visually identical while ensuring no raw 

162 ``` survives to close the fence. Single/double backticks are left untouched 

163 — they render literally inside a block. (The only cost: a steps.md copied 

164 from output that itself contained ``` carries invisible zero-width spaces.) 

165 """ 

166 return _TRIPLE_BACKTICK_RUN.sub(lambda m: "\u200b".join("`" * len(m.group())), content)

169def _fenced(content: str) -> str: 

170 """Wrap ``content`` in a code fence, neutralizing any inner ``` first.""" 

171 return f"```\n{_fence_safe(content)}\n```"

174def _format_call_args(args: dict[str, Any], *, collapse: bool = True) -> str: 

175 """Render tool-call args as keyword form: ``k=<json-value>, …``. 

176 

177 ``args`` is the already-coerced argument dict — the live path gets it off 

178 :attr:`StepEvent.args` (the step seam normalized it) and the tree path gets 

179 it from :func:`_tool_call_args`, so both guarantee a dict and the old 

180 non-dict / unserializable fallback ladder is gone (an unparseable arg is 

181 coerced to ``{}`` upstream and renders as ``name()``). A flat object renders 

182 as ``city="Tokyo", n=5`` — each value JSON-encoded, so strings keep their 

183 quotes and booleans read as ``true``/``false``; nested values use compact 

184 separators to keep the line tight. 

185 

186 ``collapse`` (default ``True``) folds runs of real whitespace to single 

187 spaces for the compact live preview. The full transcript tree passes 

188 ``collapse=False`` to keep byte fidelity — JSON already escapes real 

189 newlines, so the signature stays one line either way, but inner spacing in 

190 string values is preserved. 

191 """ 

192 if not args: 

193 return "" 

194 pairs: list[str] = [] 

195 for key, value in args.items(): 

196 try: 

197 rendered_value = json.dumps(value, separators=(",", ":"), default=str) 

198 except (TypeError, ValueError): 

199 rendered_value = str(value) 

200 pairs.append(f"{key}={rendered_value}") 

201 joined = ", ".join(pairs) 

202 return _collapse_ws(joined) if collapse else joined

205def _tool_call_args(call: ToolCallPart) -> dict[str, Any]: 

206 """Best-effort argument dict for a transcript ``ToolCallPart`` (never raises). 

207 

208 The tree renderer projects persisted ``message_history`` (no step seam in 

209 front of it), so it coerces a tool call's raw args here the same way the 

210 seam does for the live path: a non-object or unparseable arg becomes ``{}`` 

211 (rendered as ``name()``) rather than raising and losing the WHOLE stored 

212 transcript over one malformed call. ``args_as_dict`` is pydantic-ai's 

213 canonical accessor — it returns ``{}`` for empty args and asserts on a 

214 bare list/scalar — so well-formed object args render byte-for-byte as before. 

215 """ 

216 try: 

217 args = call.args_as_dict() 

218 except Exception: 

219 # A bare list/scalar makes args_as_dict assert; malformed JSON makes it 

220 # raise. Either way the args aren't an object → render as ``name()``. 

221 return {} 

222 return args if isinstance(args, dict) else {}

245def _tool_tree_block(call: ToolCallPart, ret: ToolReturnPart | None) -> str: 

246 """Render a tool call and its (optional) result as one fenced tree block. 

247 

248 ``● tool(args)`` on the first line; when a matching return is present, its 

249 result is nested under ``⎿`` with continuation lines aligned beneath the 

250 first result character. Args use the keyword form WITHOUT whitespace 

251 collapsing — this is the full view, so byte fidelity is preserved (real 

252 newlines are already JSON-escaped, so the signature stays one line). No 

253 truncation: the only bound is the overall message cap enforced upstream. 

254 """ 

255 sig = f"{call.tool_name}({_format_call_args(_tool_call_args(call), collapse=False)})" 

256 lines = [f"{_TREE_CALL_MARKER} {sig}"] 

257 if ret is not None: 

258 first, *rest = ret.model_response_str().split("\n") 

259 lines.append(f" {_TREE_RETURN_MARKER} {first}") 

260 lines.extend(f" {line}" for line in rest) 

261 return _fenced("\n".join(lines))

264def _return_tree_block(ret: ToolReturnPart) -> str: 

265 """Render an orphan tool return (no call with its id in the slice) standalone. 

266 

267 Practically unreachable — a tool call and its return live in the same 

268 agent run, after the history cursor, so they're sliced together. Rendered 

269 defensively so an orphan return is never silently dropped, which would also 

270 skew the step count that gates the ⤵ button.

273 lines = [f"{_TREE_RETURN_MARKER} {first}"] 

274 lines.extend(f" {line}" for line in rest)

278def _render_tree_blocks(messages: Sequence[ModelMessage]) -> list[str]: 

279 """Project the turn's ``message_history`` slice into full tree blocks. 

280 

281 The VERBOSE/full renderer behind the reply's ``⤵ steps`` expand view 

282 (:mod:`calfcord.bridge.steps_toggle`) and the source of the 

283 reply poster's step COUNT (``len(...)`` → the ``⤵ N steps`` label). It is NOT 

284 used by the live progress message — that path renders one 

285 :class:`StepEvent` at a time via :func:`render_step_line`. Walks the delta 

286 in order, emitting one string per visual block: 

287 

288 * a model ``TextPart`` → a prose block (whitespace-only skipped); 

289 * a ``ToolCallPart`` → ``● tool(args)`` with its matched ``ToolReturnPart`` 

290 (looked up by ``tool_call_id``) nested under ``⎿`` — a call and its 

291 result are ONE block, so the step count credits a tool use once. 

292 

293 Skips the same parts as the live renderer (``ThinkingPart``, ``FilePart``, 

294 ``BuiltinTool*Part``, ``UserPromptPart`` / ``SystemPromptPart``, 

295 ``RetryPromptPart`` — see the original v1 rationale). 

296 

297 Pairing is purely by id and independent of message order: a return is 

298 folded into its call iff a call with that id exists anywhere in the slice; 

299 a return whose call is absent renders standalone (so nothing is dropped, 

300 and the orphan path can't double-render a return that arrives before its 

301 call). Output order follows message order. Duplicate ``tool_call_id``s 

302 don't occur in well-formed pydantic-ai history; on a collision the last 

303 return for an id wins. 

304 

305 Caller wraps this in a try/except — ``model_response_str`` can raise on 

306 malformed payloads. 

307 """ 

308 # Two index passes (order-independent): which ids have a call in the 

309 # slice, and the return for each id. A return is then an orphan iff its id 

310 # has no call here — decided without relying on walk order. 

311 call_ids: set[str] = set() 

312 returns_by_id: dict[str, ToolReturnPart] = {} 

313 for msg in messages: 

314 if isinstance(msg, ModelResponse): 

315 for part in msg.parts: 

316 if isinstance(part, ToolCallPart): 

317 call_ids.add(part.tool_call_id) 

318 elif isinstance(msg, ModelRequest): 318 ↛ 313line 318 didn't jump to line 313 because the condition on line 318 was always true

319 for part in msg.parts: 

320 if isinstance(part, ToolReturnPart): 

321 returns_by_id[part.tool_call_id] = part 

322 

323 out: list[str] = [] 

324 for msg in messages: 

325 if isinstance(msg, ModelResponse): 

326 for part in msg.parts: 

327 if isinstance(part, TextPart): 

328 rendered = _render_text_part(part) 

329 if rendered is not None: 329 ↛ 326line 329 didn't jump to line 326 because the condition on line 329 was always true

330 out.append(rendered) 

331 elif isinstance(part, ToolCallPart): 331 ↛ 326line 331 didn't jump to line 326 because the condition on line 331 was always true

332 out.append(_tool_tree_block(part, returns_by_id.get(part.tool_call_id))) 

333 elif isinstance(msg, ModelRequest): 333 ↛ 324line 333 didn't jump to line 324 because the condition on line 333 was always true

334 for part in msg.parts: 

335 if isinstance(part, ToolReturnPart) and part.tool_call_id not in call_ids: 

336 out.append(_return_tree_block(part)) 

337 return out

343def _render_tool_result_line(text: str) -> str: 

344 """Compact fenced result: ``⎿`` first line + up to a few aligned more. 

345 

346 Keeps the first :data:`_LIVE_RETURN_MAX_LINES` real lines of ``text`` (the 

347 seam already concatenated the result's text parts) so multi-line output 

348 stays readable in the live stream, but bounded. The two kinds of truncation 

349 are marked where they happen: a line longer than 

350 :data:`_LIVE_RETURN_LINE_MAX_CHARS` is cut with a trailing ``…`` ON THAT 

351 line, and when whole lines are dropped beyond the kept few a 

352 ``… (truncated)`` marker is appended to the last kept line. Wrapped in a 

353 fence (embedded ``` is neutralized) so a stray triple-backtick can't break 

354 the progress message; the full, untruncated result lives on the ⤵ transcript. 

355 """ 

356 lines = text.split("\n") 

357 kept: list[str] = [] 

358 for line in lines[:_LIVE_RETURN_MAX_LINES]: 

359 line = line.rstrip() 

360 if len(line) > _LIVE_RETURN_LINE_MAX_CHARS: 

361 # Mark the cut on the line that was actually cut, not elsewhere. 

362 line = line[: _LIVE_RETURN_LINE_MAX_CHARS - 1].rstrip() + "…" 

363 kept.append(line) 

364 if not kept: 364 ↛ 365line 364 didn't jump to line 365 because the condition on line 364 was never true

365 kept = [""] 

366 if len(lines) > _LIVE_RETURN_MAX_LINES: 

367 # Whole lines were dropped — flag it at the end of the block. 

368 kept[-1] = f"{kept[-1]} … (truncated)".strip() 

369 first, *rest = kept 

370 tree = [f"{_TREE_RETURN_MARKER} {first}"] 

371 tree.extend(f" {line}" for line in rest) 

372 return _fenced("\n".join(tree))

375def render_step_line(step: StepEvent) -> str | None: 

376 """Render ONE :class:`StepEvent` into a single compact live line. 

377 

378 Returns ``None`` when the step has nothing to show (a whitespace-only 

379 ``agent_message`` preamble, or a ``handoff`` — which never reaches the live 

380 renderer anyway, since the A2A dispatcher claims it upstream). Otherwise: 

381 

382 * ``agent_message`` → the stripped text as plain markdown prose, capped at 

383 :data:`_LIVE_TEXT_MAX_CHARS` (this is only ever a NON-terminal preamble: 

384 the terminal answer never arrives as a step — it rides ``handle.result()``, 

385 posted by the reply poster — so there is no flash/terminal special case); 

386 * ``tool_call`` → one inline-code ``tool_name(args)`` line capped at 

387 :data:`_LIVE_TOOL_MAX_CHARS`; 

388 * ``tool_result`` → a short fenced ``⎿`` block (see 

389 :func:`_render_tool_result_line`). 

390 """ 

391 if step.kind == "agent_message": 

392 text = step.text.strip() 

393 if not text: 

394 return None 

395 return _truncate(text, _LIVE_TEXT_MAX_CHARS) 

396 if step.kind == "tool_call": 

397 call = f"{step.name}({_format_call_args(step.args or {})})" 

398 return _inline_code(_truncate_inline(call, _LIVE_TOOL_MAX_CHARS)) 

399 if step.kind == "tool_result": 

400 return _render_tool_result_line(step.text) 

401 # handoff (or any future kind): claimed by the A2A dispatcher upstream, so 

402 # it never reaches on_step — nothing to render live. 

403 return None

406def _tail_window(lines: Sequence[str], max_chars: int) -> str: 

407 """Join the most recent ``lines`` that fit in ``max_chars`` (newline-joined). 

408 

409 Walks from the end, keeping lines until the next one would overflow the 

410 budget (the most recent line is always kept, even if oversized — the final 

411 clamp in :func:`_progress_content` guards the hard cap). When older lines 

412 were dropped, prepends :data:`_HIDDEN_STEPS_MARKER` so the elision is 

413 visible. Returns ``""`` for no lines. 

414 """ 

415 if not lines: 415 ↛ 416line 415 didn't jump to line 416 because the condition on line 415 was never true

416 return "" 

417 kept: list[str] = [] 

418 total = 0 

419 for line in reversed(lines): 

420 cost = len(line) + 1 # the newline that will join it 

421 if kept and total + cost > max_chars: 

422 break 

423 kept.append(line) 

424 total += cost 

425 kept.reverse() 

426 if len(kept) < len(lines): 

427 return _HIDDEN_STEPS_MARKER + "\n" + "\n".join(kept) 

428 return "\n".join(kept)

431def _progress_content(lines: Sequence[str]) -> str: 

432 """Render the transient progress message body: the live trace, nothing else. 

433 

434 A tail-windowed compact render of the accumulated ``lines`` — the agent's 

435 actual work as it streams (model text, ``tool_name(args)``, ``⎿ result``), 

436 most-recent-fits-first. No header/counter line: the message IS the trace. 

437 Hard-clamped to :data:`_DISCORD_MESSAGE_LIMIT` so an in-place 

438 ``edit_message`` can never fail for length (the tail window keeps it well 

439 under; this slice is a final safety net). 

440 

441 The message only ever posts/edits on a renderable step, so ``lines`` is 

442 non-empty whenever this is called; the empty-body fallback (a lone ``…``) 

443 is purely defensive — Discord rejects an empty-content message. 

444 """ 

445 body = _tail_window(lines, _PROGRESS_BODY_MAX_CHARS) 

446 if not body: 446 ↛ 451line 446 didn't jump to line 451 because the condition on line 446 was never true

447 # Unreachable in practice — the message only posts/edits on a 

448 # renderable step, so ``lines`` is non-empty here. The lone "…" 

449 # keeps a future regression from sending empty content (Discord 

450 # rejects it with a 400) while surfacing the anomaly in the log. 

451 logger.warning("steps: progress content rendered empty (unexpected); sending a placeholder") 

452 return "…" 

453 return body[:_DISCORD_MESSAGE_LIMIT]
