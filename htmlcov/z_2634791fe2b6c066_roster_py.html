1"""Per-invocation agent roster injected as the router's ``temp_instructions``. 

2 

3The router needs to know which assistants exist before it can sensibly 

4emit a :class:`~calfcord.agents.routing.RoutingDecision`. We 

5surface that list as ``temp_instructions`` on each invocation — same 

6pattern as :mod:`calfcord.agents.peer_roster`, but the 

7constraints differ: 

8 

9* The router has no ``private_chat`` tool, so we do NOT gate visibility 

10 on a tool name. Every assistant is a candidate respondent regardless 

11 of which tools (if any) it declares. 

12* The router itself is excluded from the roster — it can never be a 

13 respondent. In production this filter is belt-and-suspenders: 

14 :func:`~calfcord.agents.phonebook.phonebook_from_registry` 

15 already drops the router upstream, so the input phonebook normally 

16 doesn't contain it. The filter here defends against direct callers 

17 that construct a phonebook some other way (tests, future 

18 integrations) and pass the registry verbatim. 

19 

20Operates on a :class:`PhonebookEntry` sequence (the same wire shape used 

21elsewhere in the project) so the same helper is reusable in any 

22deployment that received the phonebook via ``deps``. 

23""" 

24 

25from __future__ import annotations 

26 

27import logging 

28from collections.abc import Sequence 

29 

30from calfcord.agents.phonebook import PhonebookEntry, format_roster_lines 

31from calfcord.router.definition import ROUTER_AGENT_ID 

32 

33logger = logging.getLogger(__name__) 

34 

35 

36def build_router_temp_instructions( 

37 phonebook: Sequence[PhonebookEntry], 

38) -> str | None: 

39 """Return the ``temp_instructions`` to inject for a router invocation. 

40 

41 Returns ``None`` when there is nothing useful to advertise — the 

42 phonebook is empty, or after filtering out the router itself the 

43 list is empty. Callers can pass the result straight through to 

44 :meth:`Client.send` 

45 (``temp_instructions=None`` is treated as no-op by calfkit). 

46 

47 Args: 

48 phonebook: The current set of known agents. The bridge builds 

49 this on every invocation so any registry hot-add takes 

50 effect on the next call. The router is excluded from the 

51 output even if present in the input. 

52 

53 Returns: 

54 A short multi-line block listing each non-router agent's id 

55 and description, or ``None`` if no eligible respondents exist. 

56 

57 Operator signal: when ``None`` is returned (the 

58 no-eligible-respondents case), a WARN is logged on this side 

59 identifying the registry shape. The bridge ingress logs an 

60 ERROR at the same point (with the event/channel that triggered 

61 it), so a single failure window produces one WARN + one ERROR 

62 that an operator can correlate. The WARN names the cause (no 

63 assistants registered); the ERROR names the symptom (a 

64 specific ambient message that won't get a response). 

65 """ 

66 candidates = [e for e in phonebook if e.agent_id != ROUTER_AGENT_ID] 

67 if not candidates: 

68 logger.warning( 

69 "build_router_temp_instructions: no eligible respondents " 

70 "(phonebook size=%d excluding router). Router LLM will " 

71 "run with no agents to choose from; ambient messages " 

72 "will likely produce no responses. Check that the " 

73 "registry includes at least one non-router agent.", 

74 len(phonebook), 

75 ) 

76 return None 

77 return f"Available agents you can route to:\n{format_roster_lines(candidates)}"
