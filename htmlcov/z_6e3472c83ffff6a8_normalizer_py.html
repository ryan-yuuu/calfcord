1"""Discord event → :class:`WireMessage` translation. 

2 

3Two normalizers — one for incoming Discord messages (``MESSAGE_CREATE``), one 

4for slash command interactions. Both are pure functions of their inputs and 

5the agent registry; they hold no state across calls. 

6 

7Identity resolution rules (shared): 

8 - ``is_bot`` mirrors Discord's ``author.bot`` flag. 

9 - ``is_webhook`` is true when ``message.webhook_id`` is set. 

10 - ``agent_id`` is set when the author is a webhook whose ``display_name`` 

11 matches a registered :class:`AgentDefinition`. Display-name match is 

12 the bridge↔agent self-recognition primitive. 

13 - ``is_human_owner`` is set when ``message.author.id`` equals 

14 ``settings.owner_user_id`` AND the author is not a bot. 

15 

16Thread handling: messages posted inside a Discord thread carry the parent 

17channel's ID as ``channel_id`` (via the thread's ``parent_id``). Thread IDs 

18are not part of the wire schema; thread messages and parent-channel messages 

19share one topic per locked decision #1. 

20"""

38# Matches "@<agent_id>" where @ starts a whitespace-delimited token — i.e. 

39# preceded by start-of-string or whitespace. This excludes embedded @s in 

40# things like email addresses ("foo@bar.com"), URLs, or markdown. 

41# Character class is derived from ``AGENT_ID_CHARSET`` (the canonical 

42# agent_id character set) — the surrounding pattern shape differs from 

43# the agent_id validator (no length cap, capture group, no anchors) but 

44# the character set must stay in lockstep with the validator. 

45# Case-insensitive so "@Echo" and "@ECHO" also match. 

46_MENTION_RE = re.compile(rf"(?:^|\s)@([{AGENT_ID_CHARSET}]+)", re.IGNORECASE)

100 def normalize(self, message: Any) -> WireMessage: 

101 """Build a :class:`WireMessage` from a ``discord.Message``. 

102 

103 If the message starts with ``@<agent_id>`` matching a registered agent, 

104 the wire event is tagged ``kind="slash"`` with ``slash_target`` set to 

105 that agent's id — this lets agents' existing slash-aware gates handle 

106 @-mention invocations identically to native slash commands. The 

107 ``@<agent_id>`` prefix is **not** stripped from ``content``; the agent 

108 receives the full original text. 

109 

110 Raises: 

111 ValueError: If the message has no guild (DM). Callers should filter 

112 DMs out before calling this, but the contract is enforced here 

113 as a defensive measure. 

114 """ 

115 if message.guild is None: 

116 raise ValueError("MessageNormalizer received a DM (message.guild is None)") 

117 

118 channel_id = _resolve_channel_id(message.channel) 

119 # ``source_channel_id`` is the un-flattened channel id (= thread id 

120 # for thread messages, == channel_id otherwise). The bridge's 

121 # history fetcher uses this so it fetches the thread's own 

122 # history, not the parent channel's. 

123 source_channel_id = message.channel.id 

124 author = self._build_author(message) 

125 kind, slash_target = self._classify(message.content) 

126 

127 return WireMessage( 

128 event_id=uuid_utils.uuid7().hex, 

129 kind=kind, 

130 slash_target=slash_target, 

131 message_id=message.id, 

132 channel_id=channel_id, 

133 source_channel_id=source_channel_id, 

134 guild_id=message.guild.id,

136 author=author, 

137 created_at=message.created_at,

140 def normalize_task(self, message: Any, *, thread_id: int) -> WireMessage: 

141 """Build an ambient :class:`WireMessage` for a plaintext ``/task`` command. 

142 

143 Mirrors :meth:`normalize` but is purpose-built for the ``/task`` 

144 command (a ``/task <text>`` message the gateway has just opened a 

145 thread off of). Differences from :meth:`normalize`: 

146 

147 - **Always ambient** (``kind="message"``, ``slash_target=None``): a 

148 ``/task`` is routed through the router regardless of any ``@<name>`` 

149 in the text, so it does NOT scan for mentions via :meth:`_classify`. 

150 - ``source_channel_id`` is the freshly-created task ``thread_id``, so 

151 agent replies and live-step progress post *into* the thread (see 

152 :attr:`WireMessage.thread_id`), while ``channel_id`` stays the 

153 flattened parent text channel that hosts the persona webhook and is 

154 the Kafka topic key -- so the task inherits the parent channel's 

155 reachable agents. 

156 

157 ``message_id`` is the user's own message -- the genuine, user-authored 

158 thread anchor (the thread shares this id) used as the inline-reply 

159 target. ``content`` is the full message text, prefix included. 

160 

161 Raises: 

162 ValueError: If the message has no guild (DM). The gateway filters 

163 DMs before this is reached, but the contract is enforced here 

164 defensively. 

165 """ 

166 if message.guild is None: 

167 raise ValueError("MessageNormalizer.normalize_task received a DM (message.guild is None)") 

168 

169 # ``message.channel`` is the parent text channel (the gateway rejects 

170 # ``/task`` inside threads/forums), so ``_resolve_channel_id`` returns 

171 # its own id. ``source_channel_id`` is the new thread, which is what 

172 # makes ``WireMessage.thread_id`` resolve and replies/steps land 

173 # in-thread. 

174 channel_id = _resolve_channel_id(message.channel) 

175 author = self._build_author(message) 

176 

177 return WireMessage( 

178 event_id=uuid_utils.uuid7().hex, 

179 kind="message", 

180 slash_target=None, 

181 message_id=message.id, 

182 channel_id=channel_id, 

183 source_channel_id=thread_id, 

184 guild_id=message.guild.id, 

185 content=message.content, 

186 author=author,

190 def _classify(self, content: str) -> tuple[Literal["message", "slash"], str | None]: 

191 """Return ``(kind, slash_target)`` based on @<agent_id> mention scanning. 

192 

193 Scans **every** ``@<name>`` token in the message (where ``@`` starts a 

194 whitespace-delimited token, so emails are excluded). If any mentioned 

195 name does not resolve to a registered agent, raises 

196 :class:`UnknownAgentMentionError` so the gateway can fail fast and 

197 report back to the user. If all mentions resolve, the first 

198 mention's ``agent_id`` becomes ``slash_target``. If no mentions are 

199 present, returns ``("message", None)``. 

200 

201 The built-in router agent (``role="router"``) is treated as 

202 unknown here by design: it's not user-invocable via @-mention, 

203 only via the ambient ingress topic. Surfacing a router 

204 @-mention as ``UnknownAgentMentionError`` produces the standard 

205 operator-actionable error reply rather than a silently 

206 misrouted message (the router has no Discord persona and 

207 wouldn't reply). 

208 """ 

209 raw_names = _MENTION_RE.findall(content) 

210 if not raw_names: 

211 return "message", None 

212 

213 slash_target: str | None = None 

214 unknown: list[str] = [] 

215 for raw in raw_names: 

216 spec = self._registry.by_id(raw.lower()) 

217 if spec is None or spec.role == "router": 

218 unknown.append(raw) 

219 elif slash_target is None: 

220 slash_target = spec.agent_id 

221 

222 if unknown: 

223 raise UnknownAgentMentionError(unknown) 

224 

225 assert slash_target is not None # at least one match, all known 

226 return "slash", slash_target
