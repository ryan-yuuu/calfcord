1"""Minimal, position-preserving dotenv reader/writer for ``calfcord init``. 

2 

3The install's ``config/.env`` is the *seeded* ``.env.example``: it is 

4heavily commented and its ordering is meaningful documentation for the 

5operator. A general-purpose dotenv library would happily rewrite that file 

6(reordering keys, dropping comments, normalizing quoting), so we hand-roll 

7the trivial ``KEY=VALUE`` format here and guarantee an in-place upsert that 

8leaves every comment, blank line, and unrelated key exactly where it was. 

9 

10Writes are atomic (the shared :func:`calfcord._atomic.atomic_write_text` — 

11temp file + :func:`os.replace`) and ``chmod 0600`` because the file holds API 

12keys and the Discord bot token — a partial write or a world-readable secrets 

13file is a real hazard, so both are handled by the one writer the upsert routes 

14through. 

15 

16Scope: values are single-line ``KEY=VALUE`` pairs (tokens, keys, ids, urls). A 

17value containing a newline is rejected (it would split into a second, malformed 

18line and silently corrupt the file), and :func:`read_env` matches the runtime 

19dotenv loaders (python-dotenv / ``uv run --env-file``) on the parsing details 

20that matter — surrounding whitespace, an ``export`` prefix, quotes, and inline 

21comments — so the wizard's "current value" never disagrees with what the 

22processes actually load. 

23""" 

24 

25from __future__ import annotations 

26 

27from collections.abc import Mapping 

28from pathlib import Path

35def _parse_value(value: str) -> str: 

36 """Decode one already-whitespace-stripped dotenv value the way the loaders do. 

37 

38 Mirrors python-dotenv / ``uv run --env-file`` so ``init``'s "current value" 

39 display agrees with what the processes load: a value wrapped in one matching 

40 pair of quotes is taken verbatim (a ``#`` inside quotes is literal); an 

41 unquoted value ends at the first `` #`` (whitespace + hash), which begins an 

42 inline comment. Operators (and tooling) quote values and append comments, so 

43 the reader must honour both or a re-run would mis-read an already-set key. 

44 """ 

45 if len(value) >= 2 and value[0] == value[-1] and value[0] in ("'", '"'): 

46 return value[1:-1] 

47 comment = value.find(" #") 

48 if comment != -1: 

49 value = value[:comment].rstrip() 

50 return value

53def read_env(path: Path) -> dict[str, str]: 

54 """Parse ``path`` into a ``{KEY: VALUE}`` dict; missing file yields ``{}``. 

55 

56 Blank lines and comment lines (first non-space char ``#``) are ignored, as 

57 are lines without ``=``. A leading ``export `` is part of the dotenv/shell 

58 syntax (the key is what follows it), values are stripped of surrounding 

59 whitespace and decoded by :func:`_parse_value` (quotes / inline comments), and 

60 a later assignment of the same key wins (dotenv last-wins) — all so a re-run of 

61 ``init`` reads a key exactly as the process would. 

62 """ 

63 if not path.exists(): 

64 return {} 

65 result: dict[str, str] = {} 

66 for raw in path.read_text(encoding="utf-8").splitlines(): 

67 line = raw.strip() 

68 if not line or line.startswith("#") or "=" not in line: 

69 continue 

70 if line.startswith("export "): 

71 line = line[len("export ") :].lstrip() 

72 key, _, value = line.partition("=") 

73 key = key.strip() 

74 if not key: 74 ↛ 75line 74 didn't jump to line 75 because the condition on line 74 was never true

75 continue 

76 result[key] = _parse_value(value.strip()) 

77 return result

80def upsert(path: Path, updates: Mapping[str, str]) -> None: 

81 """Idempotently set each key in ``updates``, preserving the rest of the file. 

82 

83 For every key already present, its ``KEY=...`` line is replaced **in 

84 place** so all comments, ordering, and unrelated lines survive untouched; 

85 keys not yet present are appended (one per line) after the existing 

86 content. The file (and its parent directory) is created if absent. The 

87 write is atomic (temp file in the same directory + :func:`os.replace`) and 

88 the result is ``chmod 0600`` because it holds secrets. An empty ``updates`` 

89 is a no-op, so callers can upsert unconditionally without a guard. 

90 

91 Values must be single-line: a value containing a newline raises 

92 :class:`ValueError` rather than splitting into a second, malformed line that 

93 would silently corrupt the file (and, on a re-run, append it again). For 

94 single-line values, running this twice with the same ``updates`` produces 

95 byte-identical output: the first run sets the keys, the second finds them 

96 already on their lines and rewrites the same bytes. 

97 

98 Raises: 

99 ValueError: a value contains a newline (``\\n`` / ``\\r``). 

100 """ 

101 if not updates: 

102 return 

103 

104 for key, value in updates.items(): 

105 if "\n" in value or "\r" in value: 

106 raise ValueError( 

107 f"value for {key!r} contains a newline; .env values must be single-line " 

108 "(check for a stray newline in a pasted secret)" 

109 ) 

110 

111 path.parent.mkdir(parents=True, exist_ok=True) 

112 

113 original = path.read_text(encoding="utf-8") if path.exists() else "" 

114 lines = original.splitlines() 

115 

116 remaining = dict(updates) 

117 new_lines: list[str] = [] 

118 for line in lines: 

119 stripped = line.lstrip() 

120 replaced = False 

121 if stripped and not stripped.startswith("#") and "=" in line: 

122 key = line.partition("=")[0].strip() 

123 if key in remaining: 

124 new_lines.append(f"{key}={remaining.pop(key)}") 

125 replaced = True 

126 if not replaced: 

127 new_lines.append(line) 

128 

129 # Append keys that were never present, in the caller's iteration order. 

130 for key, value in remaining.items(): 

131 new_lines.append(f"{key}={value}") 

132 

133 # Re-join. Always terminate with a newline: secrets files are line-oriented 

134 # and a missing final newline trips naive `grep '^KEY='` style readers (the 

135 # shim's set-broker uses exactly that). With no content lines, preserve the 

136 # original's blank/empty shape rather than inventing one. (Kept as an 

137 # if/else rather than a nested ternary for readability — see SIM108.) 

138 if new_lines: # noqa: SIM108 138 ↛ 141line 138 didn't jump to line 141 because the condition on line 138 was always true

139 body = "\n".join(new_lines) + "\n" 

140 else: 

141 body = "\n" if original.endswith("\n") else "" 

142 

143 # Atomic same-dir tmp + os.replace, chmod 0600 (the shared 

144 # calfcord._atomic.atomic_write_text): a process loading ``.env`` never reads 

145 # a half-written secrets file, and a crashed write leaves no ``.tmp`` orphan. 

146 atomic_write_text(path, body, mode=_SECRET_FILE_MODE)
