1"""Filesystem tools: ``read_file``, ``write_file``, ``edit_file``. 

2 

3Thin :func:`agent_tool`-decorated wrappers around :class:`FileEditorExecutor` 

4from ``openhands-tools``. The executor is a module-global singleton whose 

5``workspace_root`` is the calfcord shared workspace 

6(:func:`~calfcord.tools.builtin.workspace.get_workspace_root`). 

7Relative paths in ``path`` arguments are resolved against that root before 

8being passed to upstream so the LLM can use ``"main.py"`` style addressing 

9without knowing the container/host filesystem layout. 

10 

11The three tools mirror Claude Code's surface 1:1: 

12 

13* ``read_file`` → ``FileEditorAction(command="view", ...)`` — supports 

14 ``view_range=[start, end]`` for partial views (1-indexed; ``-1`` for "to end"). 

15* ``write_file`` — creates or overwrites a file with the provided content. 

16 Uses :meth:`pathlib.Path.write_text` directly because the upstream 

17 ``create`` command refuses to overwrite (by design — they expect 

18 ``str_replace`` for in-place edits). Parent directories are created 

19 if missing, matching Claude Code's ``Write`` semantics. 

20* ``edit_file`` → ``FileEditorAction(command="str_replace", ...)`` — 

21 exact-string replacement. The upstream tool requires ``old_string`` to 

22 occur exactly once (otherwise it errors with "Multiple occurrences…"). 

23 We add a ``replace_all`` knob: when true, our wrapper drives upstream in 

24 a loop replacing one occurrence at a time until ``old_string`` no longer 

25 appears. This matches Claude Code's ``Edit`` semantics without patching 

26 upstream. 

27 

28The upstream tool's ``insert`` and ``undo_edit`` commands are intentionally 

29not exposed in v1 — they widen the LLM's tool surface beyond Claude Code's 

30and aren't worth the extra description tokens for our use case. 

31"""

51def _get_executor() -> FileEditorExecutor: 

52 """Return the module-global executor, constructing it on first use. 

53 

54 Lazy so importing this module doesn't read ``CALFCORD_WORKSPACE_DIR`` 

55 or create the workspace directory — that work happens on the first 

56 actual tool call. Tests reset the singleton by patching ``_executor`` 

57 directly. 

58 """ 

59 global _executor 

60 if _executor is None: 60 ↛ 63line 60 didn't jump to line 63 because the condition on line 60 was always true

61 root = get_workspace_root() 

62 _executor = FileEditorExecutor(workspace_root=str(root)) 

63 return _executor

66def _resolve_path(path: str) -> str: 

67 """Resolve a user-supplied ``path`` against the workspace root. 

68 

69 Absolute paths pass through unchanged (the trusted-workspace model 

70 means the LLM is allowed to address absolute paths on the host). 

71 Relative paths are joined to the workspace root so ``read_file("foo.py")`` 

72 works without the LLM knowing the host's absolute layout. 

73 """ 

74 p = Path(path) 

75 if p.is_absolute(): 

76 return str(p) 

77 return str((get_workspace_root() / p).resolve())

80async def read_file( 

81 ctx: ToolContext, 

82 path: str, 

83 view_range: list[int] | None = None, 

84) -> str: 

85 """Read a file's contents (or list a directory's children). 

86 

87 Use this to inspect a file before editing it, to spot-check a config, 

88 or to enumerate what's in a directory. The output includes ``cat -n`` 

89 style line numbers, which you can reference when writing follow-up 

90 ``edit_file`` calls. 

91 

92 Args: 

93 path: Absolute path on the host, or a path relative to the 

94 calfcord workspace root. Both files and directories are 

95 supported (directories return a recursive listing up to 2 

96 levels deep). 

97 view_range: Optional ``[start, end]`` line numbers (1-indexed) to 

98 return only part of the file. ``end=-1`` means "to end of 

99 file". Ignored for directory views. 

100 

101 Returns: 

102 The file contents (or directory listing) as a string. On error 

103 (missing file, no read permission, binary file too large), an 

104 ``"error: ..."`` style message describing what went wrong. 

105 """ 

106 _ = ctx # unused 

107 resolved = _resolve_path(path) 

108 action = FileEditorAction(command="view", path=resolved, view_range=view_range) 

109 obs = _get_executor()(action) 

110 return flatten_observation_text(obs)

113async def write_file(ctx: ToolContext, path: str, content: str) -> str: 

114 """Create a new file or overwrite an existing one with ``content``. 

115 

116 Use this for new files, or when you want to rewrite a file end-to-end. 

117 For targeted single-region edits, prefer ``edit_file``. 

118 

119 Args: 

120 path: Absolute path on the host, or a path relative to the 

121 calfcord workspace root. Missing parent directories are 

122 created. 

123 content: The complete new file content. Existing content (if 

124 any) is replaced. 

125 

126 Returns: 

127 A short confirmation message including the path, or an 

128 ``"error: ..."`` message on failure. 

129 """ 

130 _ = ctx 

131 resolved = _resolve_path(path) 

132 # Use ``Path.write_text`` directly rather than the upstream 

133 # ``FileEditorAction(command="create")``: the upstream command 

134 # explicitly refuses to overwrite existing files (it's positioned as 

135 # "create new" only). Claude Code's ``Write`` overwrites, so we 

136 # match that semantic here. Trade-off: we skip openhands' file- 

137 # history tracking — acceptable because git is the system of record 

138 # for change history in the calfcord workspace anyway. 

139 p = Path(resolved) 

140 try: 

141 p.parent.mkdir(parents=True, exist_ok=True) 

142 # Pin UTF-8 explicitly. ``Path.write_text`` would otherwise pick 

143 # ``locale.getpreferredencoding(False)`` which is non-UTF-8 on 

144 # some Windows hosts and inside ``LANG=C`` containers — silent 

145 # corruption of non-ASCII content. The shipped docker image 

146 # sets ``C.UTF-8`` so this matters mostly for native dev hosts. 

147 p.write_text(content, encoding="utf-8") 

148 except OSError as e: 

149 return f"error: cannot write {resolved}: {e}" 

150 return f"Wrote {len(content)} characters to {resolved}"

153async def edit_file( 

154 ctx: ToolContext, 

155 path: str, 

156 old_string: str, 

157 new_string: str, 

158 replace_all: bool = False, 

159) -> str: 

160 """Replace exact text in a file. 

161 

162 By default, ``old_string`` must occur exactly once in the file — if 

163 it appears multiple times the tool errors so you can supply more 

164 surrounding context to disambiguate. Set ``replace_all=True`` to 

165 replace every occurrence in one call. 

166 

167 Args: 

168 path: Absolute path on the host, or a path relative to the 

169 calfcord workspace root. 

170 old_string: The exact substring to find. Whitespace and indentation 

171 must match — copy from a recent ``read_file`` if unsure. 

172 new_string: The replacement text. May be empty (deletion). 

173 replace_all: When ``True``, replace every occurrence of 

174 ``old_string``. When ``False`` (default), require exactly one 

175 occurrence and error otherwise. 

176 

177 Returns: 

178 A short confirmation message with a diff snippet, or an 

179 ``"error: ..."`` message on failure (file missing, no match, 

180 ambiguous match when ``replace_all=False``). 

181 """ 

182 _ = ctx 

183 resolved = _resolve_path(path) 

184 if not replace_all: 

185 action = FileEditorAction( 

186 command="str_replace", path=resolved, old_str=old_string, new_str=new_string 

187 ) 

188 obs = _get_executor()(action) 

189 return flatten_observation_text(obs) 

190 

191 # replace_all path: openhands' ``str_replace`` rejects multi-match 

192 # inputs as ambiguous (by design — that's the safety net for the 

193 # single-shot edit case). To honor ``replace_all=True`` we bypass it 

194 # and do the substitution in pure Python. 

195 # 

196 # Trade-off: skipping the upstream executor means no file-history 

197 # tracking and no charset auto-detection (we pin UTF-8 below). Git 

198 # is the system of record for change history in the calfcord 

199 # workspace, so the missing history is acceptable. The 

200 # ``exists()``/``is_dir()`` checks are TOCTOU race-y vs. the write, 

201 # which is also acceptable under the v1 trusted-workspace model 

202 # (no concurrent operator-on-host editing during agent runs). 

203 p = Path(resolved) 

204 if not p.exists(): 

205 return f"error: file not found at {resolved}" 

206 if p.is_dir(): 

207 return f"error: edit_file does not support directories: {resolved}" 

208 try: 

209 original = p.read_text(encoding="utf-8") 

210 except (OSError, UnicodeDecodeError) as e: 

211 return f"error: cannot read {resolved}: {e}" 

212 if old_string not in original: 

213 return ( 

214 f"error: old_string did not appear verbatim in {resolved}; " 

215 "check whitespace/indentation and re-read the file if needed" 

216 ) 

217 count = original.count(old_string) 

218 new_content = original.replace(old_string, new_string) 

219 try: 

220 p.write_text(new_content, encoding="utf-8") 

221 except OSError as e: 

222 return f"error: cannot write {resolved}: {e}" 

223 logger.info("edit_file replace_all path=%s replacements=%d", resolved, count) 

224 return f"Edited {resolved}: replaced {count} occurrence(s) of old_string."
