1"""Atomic mutation of frontmatter fields in an ``agents/<name>.md`` file. 

2 

3The Discord ``/thinking-effort`` slash command and the ``disco agent 

4tools`` editor both persist their values into the agent's declarative 

5``.md`` file rather than a parallel state file, so ``agents/<name>.md`` is 

6the single source of truth for an agent's declared defaults. This module 

7owns that write — load the file with ``python-frontmatter``, mutate the 

8metadata, validate the mutated state **in memory**, dump to disk 

9atomically. 

10 

11Every public mutator routes through the one :func:`_update_fields` path so 

12there is a single validate-before-write implementation: a per-field mutator 

13would be a place for the atomicity/validation invariant below to drift. 

14:func:`update_thinking_effort` and :func:`update_tools` differ only in the 

15``updates`` dict they pass (and the token pre-validation :func:`update_tools` 

16layers on top before delegating). 

17 

18Validate-before-write 

19--------------------- 

20Validation runs on a synthetic :class:`AgentDefinition` built from the 

21mutated metadata before any disk write. If validation fails the existing 

22file is untouched. This is what keeps the on-disk file and any validated 

23in-memory :class:`AgentDefinition` a CLI caller has cached from diverging 

24when an operator has hand-edited the ``.md`` between boot and the slash 

25invocation: either both succeed or neither does. The function returns the 

26validated in-memory definition rather than re-parsing from disk, so a 

27post-write read error (transient OS, concurrent edit) can't break the 

28invariant either. 

29 

30Atomicity on disk uses a tmp-file + fsync + ``os.replace`` + parent-dir 

31fsync sequence. This is now the only such atomic-write call site; it is 

32kept inline rather than abstracted — extract a shared helper if a second 

33atomic-write call site appears. 

34 

35Frontmatter round-trip caveat: ``python-frontmatter`` ultimately dumps 

36through PyYAML's ``safe_dump``, which alphabetizes keys (PyYAML defaults 

37to ``sort_keys=True`` and the library does not override it) and does not 

38preserve comments. Operators should avoid putting load-bearing comments 

39in agent frontmatter. 

40"""

63def _validate_and_write(md_path: Path, post: frontmatter.Post) -> AgentDefinition: 

64 """Validate ``post``'s mutated state in memory, then atomically rewrite ``md_path``. 

65 

66 The shared tail every mutator converges on once it has produced the final 

67 :class:`frontmatter.Post` (metadata overlaid, or body replaced): build a 

68 synthetic :class:`AgentDefinition` from the post's metadata + stripped body 

69 **before** any disk write so a bad value raises with the file untouched, dump 

70 the post, normalize the trailing newline, and atomically replace. Factored out 

71 so :func:`_update_fields` and :func:`update_system_prompt` can't drift on the 

72 validate-before-write invariant — their only difference is how they mutate the 

73 post (metadata overlay vs. body replacement) before handing it here. 

74 

75 Returns the validated in-memory definition rather than re-parsing from disk: 

76 the bytes just written are what produced it, and a re-parse exception here 

77 would leave a caller's cached in-memory definition stale relative to disk — 

78 the very desync the validate-before-write design prevents. 

79 """ 

80 candidate_metadata = dict(post.metadata) 

81 candidate_metadata["system_prompt"] = post.content.strip() 

82 candidate_metadata["source_path"] = md_path 

83 validated = AgentDefinition(**candidate_metadata) 

84 

85 payload = frontmatter.dumps(post) 

86 # Ensure a trailing newline — frontmatter.dumps may omit it depending 

87 # on the body's own trailing whitespace, and POSIX text files 

88 # conventionally end with one. 

89 if not payload.endswith("\n"): 89 ↛ 92line 89 didn't jump to line 92 because the condition on line 89 was always true

90 payload += "\n" 

91 

92 _atomic_write_text(md_path, payload) 

93 return validated

96def _update_fields(md_path: Path, updates: dict[str, object]) -> AgentDefinition: 

97 """Apply ``updates`` to ``md_path``'s frontmatter, validating before write. 

98 

99 The single validate-before-write path every mutator delegates to: load 

100 the file, overlay ``updates`` onto its metadata, build and validate a 

101 synthetic :class:`AgentDefinition` from the mutated metadata **in 

102 memory**, and only then atomically rewrite the file. A bad value raises 

103 before any disk write, so the on-disk file (and any in-memory definition a 

104 caller has cached) is left untouched — the desync-prevention invariant the 

105 module docstring describes. The returned definition is the validated 

106 in-memory object, not a re-parse, so a transient post-write read error 

107 can't break that invariant either. 

108 

109 ``updates`` carries already-coerced field values (e.g. an explicit 

110 ``list`` for ``tools``); semantic pre-validation that needs richer error 

111 messages than pydantic gives (the bad-token report in 

112 :func:`update_tools`) belongs in the caller, before delegating here. 

113 

114 Raises: 

115 FileNotFoundError: ``md_path`` does not exist. 

116 ValueError: the existing ``.md`` is unparseable YAML or the mutated 

117 metadata fails :class:`AgentDefinition` validation. The on-disk 

118 file is unchanged. 

119 OSError: a filesystem error during the atomic write (e.g. 

120 permission denied, no space). The on-disk file is unchanged. 

121 """ 

122 try: 

123 post = frontmatter.load(md_path) 

124 except yaml.YAMLError as e: 

125 raise ValueError(f"{md_path}: existing frontmatter is malformed YAML: {e}") from e 

126 

127 post.metadata.update(updates) 

128 

129 validated = _validate_and_write(md_path, post) 

130 logger.info("rewrote %s fields=%s in %s", "/".join(updates), list(updates), md_path) 

131 return validated

134def update_system_prompt(md_path: Path, body: str) -> AgentDefinition: 

135 """Rewrite the system-prompt **body** of ``md_path``, validating before write. 

136 

137 The system prompt is the Markdown body (``post.content``), not a frontmatter 

138 metadata field, so it cannot ride :func:`_update_fields`'s metadata-overlay 

139 path — it sets ``post.content`` directly instead. Everything else mirrors 

140 :func:`_update_fields` exactly: build and validate a synthetic 

141 :class:`AgentDefinition` from the mutated state **in memory** (the 

142 ``system_prompt`` validator rejects an empty/whitespace-only body, so a bad 

143 value raises before any disk write), then atomically rewrite. The on-disk 

144 file is left untouched on any failure, keeping a caller's cached in-memory 

145 definition from diverging from disk — the same desync-prevention invariant 

146 the rest of the module upholds. The returned definition is the validated in-memory object, 

147 not a re-parse, so a transient post-write read error can't break it either. 

148 

149 Raises: 

150 FileNotFoundError: ``md_path`` does not exist. 

151 ValueError: the existing ``.md`` is unparseable YAML, the new ``body`` is 

152 empty/whitespace-only, or the mutated metadata otherwise fails 

153 :class:`AgentDefinition` validation. The on-disk file is unchanged. 

154 OSError: a filesystem error during the atomic write (e.g. permission 

155 denied, no space). The on-disk file is unchanged. 

156 """ 

157 try: 

158 post = frontmatter.load(md_path) 

159 except yaml.YAMLError as e: 

160 raise ValueError(f"{md_path}: existing frontmatter is malformed YAML: {e}") from e 

161 

162 post.content = body 

163 

164 validated = _validate_and_write(md_path, post) 

165 logger.info("rewrote system_prompt in %s", md_path) 

166 return validated

170 """Rewrite the ``thinking_effort`` frontmatter field in ``md_path``.

175

184 OSError: a filesystem error during the atomic write (e.g. 

185 permission denied, no space). The on-disk file is unchanged.

187 return _update_fields(md_path, {"thinking_effort": value})

190def update_tools(md_path: Path, tools: Sequence[str]) -> AgentDefinition: 

191 """Rewrite the ``tools`` frontmatter list in ``md_path`` to ``tools``. 

192 

193 Every token is validated *before* the shared write path runs, so an 

194 unknown builtin or a malformed ``mcp/`` selector raises a precise 

195 :class:`ValueError` (naming the offending token) with the on-disk file 

196 untouched, rather than surfacing as a generic pydantic error: 

197 

198 * an ``mcp/...`` token must be a well-formed selector 

199 (:func:`~calfcord.mcp.selector.parse_mcp_selector`) — syntax only, 

200 mirroring the frontmatter validator; whether the named server is 

201 configured/running is a runtime concern the writer deliberately does 

202 not check; 

203 * a *builtin* token must be a key of 

204 :data:`calfcord.tools.TOOL_REGISTRY`. 

205 

206 The writer always persists an *explicit* list (``tools: []`` when 

207 ``tools`` is empty), which is what lets the editor convert the implicit 

208 "tools omitted → all builtins" default into an unambiguous on-disk state 

209 on first save. 

210 

211 The ``TOOL_REGISTRY`` import is deferred to here (rather than module 

212 scope) so :func:`update_thinking_effort`'s path stays light: importing 

213 ``TOOL_REGISTRY`` eagerly composes the tool registry (importing the 

214 vendored ``calfkit-tools`` nodes), which the thinking-effort slash 

215 command has no reason to pay for. 

216 

217 Raises: 

218 FileNotFoundError: ``md_path`` does not exist. 

219 ValueError: an unknown builtin token, a malformed ``mcp/`` selector, 

220 or a post-mutation :class:`AgentDefinition` validation failure. 

221 The on-disk file is unchanged. 

222 OSError: a filesystem error during the atomic write. The on-disk 

223 file is unchanged. 

224 """ 

225 from calfcord.mcp.selector import is_mcp_selector, parse_mcp_selector 

226 from calfcord.tools import TOOL_REGISTRY 

227 

228 for token in tools: 

229 # ``is_mcp_selector`` reaches for ``token.startswith`` — a non-str token 

230 # would surface as an ``AttributeError`` instead of the ``ValueError`` this 

231 # seam documents, so reject it up front with the contract-honoring error. 

232 if not isinstance(token, str): 

233 raise ValueError(f"invalid tool {token!r}: expected a string") 

234 if is_mcp_selector(token): 

235 parse_mcp_selector(token) # syntactic check only; raises naming the token 

236 continue 

237 if token not in TOOL_REGISTRY: 

238 valid = ", ".join(sorted(TOOL_REGISTRY)) or "(none registered)" 

239 raise ValueError( 

240 f"unknown tool {token!r}; expected a builtin ({valid}) or an MCP " 

241 f"selector (mcp/<server> or mcp/<server>/<tool>)" 

242 ) 

243 

244 return _update_fields(md_path, {"tools": list(tools)})

247def _atomic_write_text(path: Path, payload: str) -> None: 

248 """Write ``payload`` to ``path`` via tmp-file + fsync + atomic rename. 

249 

250 Caller is expected to have verified ``path`` exists (and therefore 

251 its parent does) — no defensive ``mkdir`` here. 

252 """ 

253 # mkstemp creates the tmp file 0o600, and os.replace adopts the tmp file's 

254 # mode — so without this an existing 0o644 agent .md would silently become 

255 # 0o600 on every rewrite. Capture the target's mode now to restore it after. 

256 original_mode = stat.S_IMODE(path.stat().st_mode) if path.exists() else None 

257 

258 fd, tmp_name = tempfile.mkstemp( 

259 dir=path.parent, 

260 prefix=f".{path.name}.", 

261 suffix=".tmp", 

262 ) 

263 tmp_path = Path(tmp_name) 

264 try: 

265 with os.fdopen(fd, "w", encoding="utf-8") as tmp: 

266 tmp.write(payload) 

267 tmp.flush() 

268 os.fsync(tmp.fileno()) 

269 os.replace(tmp_path, path) 

270 if original_mode is not None: 270 ↛ 276line 270 didn't jump to line 276 because the condition on line 270 was always true

271 os.chmod(path, original_mode) 

272 except Exception: 

273 tmp_path.unlink(missing_ok=True) 

274 raise 

275 

276 if os.name == "posix": 276 ↛ exitline 276 didn't return from function '_atomic_write_text' because the condition on line 276 was always true

277 try: 

278 dir_fd = os.open(path.parent, os.O_RDONLY) 

279 try: 

280 os.fsync(dir_fd) 

281 finally: 

282 os.close(dir_fd) 

283 except OSError: 

284 # The rename is already durable on most filesystems even if 

285 # we can't fsync the parent. Don't fail the caller's commit; 

286 # we just lose the strong-durability guarantee on power loss. 

287 logger.warning( 

288 "parent-dir fsync failed for %s; rename is committed but durability " 

289 "may be weaker on power loss", 

290 path, 

291 exc_info=True, 

292 )
