1"""Agent-path MCP tool selection — schema-free, ``mcp.json``-free. 

2 

3calfkit resolves an agent's MCP tools per turn by calling each declared 

4``ToolSelector`` against the capability view (the ``KafkaTable`` projection 

5of the ``mcp.capabilities`` topic the ``Worker`` auto-registers whenever a 

6hosted agent declares selectors). The selector type is calfkit's public 

7:class:`~calfkit.mcp.MCPToolboxRef` — an identity-only handle constructible 

8with just the server name, so on a distributed deploy the agent host needs 

9neither ``mcp.json`` nor the secrets inside it. This module owns the 

10calfcord side only: collapsing an agent's ``mcp/...`` frontmatter entries 

11into one ref per server. 

12 

13Policy: refs are **non-strict** (the upstream default, never overridden 

14here). An agent whose MCP server is down (or not yet started) boots and 

15answers normally; the affected tools drop out of that turn with calfkit 

16logging the degradation. This matches the roster's "nothing runs that you 

17didn't start" property — declaring ``mcp/github`` in an agent's frontmatter 

18must not hold the agent hostage to the github server's uptime. 

19""" 

20 

21from __future__ import annotations 

22 

23from collections.abc import Iterable 

24 

25from calfkit.mcp import MCPToolboxRef 

26 

27from calfcord.mcp.selector import is_mcp_selector, parse_mcp_selector 

28 

29 

30def selectors_from_entries(entries: Iterable[str]) -> list[MCPToolboxRef]: 

31 """Collapse an agent's ``mcp/...`` frontmatter entries into per-server refs. 

32 

33 Merge semantics match the old schema-build resolution: a bare 

34 ``mcp/<server>`` subsumes that server's explicit ``mcp/<server>/<tool>`` 

35 entries; explicit-only selections dedupe into a sorted ``include`` 

36 tuple; servers come back sorted so the agent's tool surface is 

37 deterministic regardless of frontmatter order. 

38 

39 Args: 

40 entries: ``mcp/...`` selector strings only — the factory partitions 

41 builtin names out first. A non-MCP entry here is a programming 

42 error and raises. 

43 

44 Raises: 

45 ValueError: For a non-MCP or malformed entry (message names the 

46 entry verbatim, via :func:`parse_mcp_selector`). 

47 """ 

48 wildcard: set[str] = set() 

49 explicit: dict[str, set[str]] = {} 

50 for entry in entries: 

51 if not is_mcp_selector(entry): 

52 raise ValueError( 

53 f"expected an mcp/... selector, got {entry!r} (builtin names are resolved separately)" 

54 ) 

55 server, tool = parse_mcp_selector(entry) 

56 if tool is None: 

57 wildcard.add(server) 

58 else: 

59 explicit.setdefault(server, set()).add(tool) 

60 return [ 

61 # A bare mcp/<server> wildcard subsumes that server's explicit picks. 

62 MCPToolboxRef( 

63 server, 

64 include=None if server in wildcard else tuple(sorted(explicit[server])), 

65 ) 

66 for server in sorted(wildcard | set(explicit)) 

67 ]
