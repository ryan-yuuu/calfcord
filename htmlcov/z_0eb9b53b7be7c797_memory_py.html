1"""Memory-prompt loading and the per-agent runtime instructions hook. 

2 

3calfcord agents opt into a persistent notepad with ``memory: true`` in their 

4frontmatter. The "how memory works" explanation is **not** bundled into agent 

5prompts. Instead: 

6 

7* the **bridge** reads the template (:func:`load_memory_prompt`) and ships it 

8 in ``deps`` under :data:`MEMORY_PROMPT_DEPS_KEY` on every invocation (only 

9 when the deployment has at least one memory-enabled agent); 

10* ``private_chat`` projects ``deps`` forward, so the template survives the 

11 A2A hop without per-key plumbing; 

12* each memory-enabled agent carries a dynamic-instructions hook 

13 (:func:`memory_instructions`, registered by the factory) that, at runtime, 

14 reads the template from ``deps``, localizes it to that agent's 

15 ``memory/<agent_id>/`` directory, and appends it to the agent's 

16 instructions. 

17 

18Keeping the template out of agent images preserves calfcord's 

19distributed-in-space property: the **bridge is the single reader** of the 

20prompt file; agents and the tools process never read it — they receive or 

21forward it over Kafka. The text lives in an editable Markdown file 

22(:file:`memory_prompt.md` beside this module, overridable via 

23``CALFCORD_MEMORY_PROMPT_PATH`` on the bridge host). Substitution is a plain 

24:meth:`str.replace` so the literal braces in the template's YAML-frontmatter 

25example survive untouched. 

26""" 

27 

28from __future__ import annotations 

29 

30import os 

31from collections.abc import Callable, Iterable 

32from pathlib import Path 

33 

34from calfkit._vendor.pydantic_ai import RunContext 

35 

36from calfcord.agents.definition import AgentDefinition 

37 

38_PROMPT_PATH_ENV = "CALFCORD_MEMORY_PROMPT_PATH" 

39_DEFAULT_PROMPT_PATH = Path(__file__).with_name("memory_prompt.md") 

40_MEMORY_DIR_PLACEHOLDER = "{{MEMORY_DIR}}" 

41 

42MEMORY_PROMPT_DEPS_KEY = "memory_prompt" 

43"""``deps`` key under which the bridge ships the raw (un-localized) template 

44and the agent hook reads it. ``private_chat`` forwards ``deps`` wholesale, so 

45this key propagates through A2A chains with no per-key handling.""" 

46 

47_cached_prompt: str | None = None 

48 

49 

50def load_memory_prompt() -> str: 

51 """Return the memory-explanation template, cached after the first read. 

52 

53 Called by the **bridge** (the single reader). Reads 

54 ``CALFCORD_MEMORY_PROMPT_PATH`` when set, otherwise the bundled 

55 :file:`memory_prompt.md` beside this module. Cached on a module global; 

56 an operator editing the file takes effect on the next bridge restart. 

57 

58 Raises: 

59 ValueError: if the resolved file cannot be read (e.g. a configured 

60 override path that doesn't exist) or is empty — an empty template 

61 would make ``memory: true`` a silent no-op. Callers in the bridge 

62 degrade gracefully (log + skip injection) rather than failing the 

63 whole invocation. 

64 """ 

65 global _cached_prompt 

66 if _cached_prompt is not None: 

67 return _cached_prompt 

68 override = os.getenv(_PROMPT_PATH_ENV) 

69 path = Path(override).expanduser() if override else _DEFAULT_PROMPT_PATH 

70 try: 

71 text = path.read_text(encoding="utf-8") 

72 except (OSError, UnicodeError) as e: 

73 # ``UnicodeError`` (covers ``UnicodeDecodeError``) is a ``ValueError`` 

74 # subclass, not an ``OSError`` — a readable-but-non-UTF-8 override file 

75 # would otherwise escape as a bare, context-less ``ValueError``. 

76 raise ValueError( 

77 f"cannot read memory prompt at {path} ({_PROMPT_PATH_ENV}={override!r}): {e}" 

78 ) from e 

79 if not text.strip(): 

80 raise ValueError(f"memory prompt at {path} is empty") 

81 _cached_prompt = text 

82 return _cached_prompt

85def render_memory_block(template: str, agent_id: str) -> str: 

86 """Localize ``template`` for one agent: substitute its memory dir, then strip. 

87 

88 The agent dir is workspace-relative (``memory/<agent_id>/``) on purpose: 

89 the LLM passes it straight to ``read_file`` / ``write_file``, which resolve 

90 relative paths against the workspace root in the tools process. 

91 """ 

92 return template.replace(_MEMORY_DIR_PLACEHOLDER, f"memory/{agent_id}/").strip() 

93

95def memory_instructions(agent_id: str) -> Callable[[RunContext[dict]], str | None]: 

96 """Build the dynamic-instructions hook for ``agent_id``. 

97 

98 The factory registers the returned callable on memory-enabled agent nodes 

99 via ``Agent.instructions``. calfkit invokes it per-run with a 

100 :class:`RunContext` whose ``.deps`` is the deps dict the invoker passed 

101 (the bridge injects the template; ``private_chat`` forwards it). It returns 

102 the localized memory block, or ``None`` when no template is present (a 

103 non-memory deployment, or an invocation path that didn't carry it), so the 

104 agent degrades gracefully instead of erroring. calfkit appends the result 

105 to the agent's instructions alongside its system prompt and the per-call 

106 peer roster. 

107 """ 

108 

109 def _hook(ctx: RunContext[dict]) -> str | None: 

110 deps = ctx.deps 

111 template = deps.get(MEMORY_PROMPT_DEPS_KEY) if isinstance(deps, dict) else None 

112 if not template: 

113 return None 

114 return render_memory_block(template, agent_id) 

115 

116 return _hook

119def memory_prompt_deps_for_registry(specs: Iterable[AgentDefinition]) -> dict[str, str]: 

120 """Build the ``deps`` entry that ships the memory-prompt template, or ``{}``. 

121 

122 The bridge is the single reader of the template (:func:`load_memory_prompt`). 

123 It ships the raw (un-localized) text under :data:`MEMORY_PROMPT_DEPS_KEY` on 

124 every agent invocation it originates **whenever the deployment has at least 

125 one memory-enabled agent** — so the template reaches every agent and, because 

126 ``private_chat`` forwards ``deps`` wholesale, propagates through A2A chains; 

127 each memory-enabled agent's instructions hook then localizes it. 

128 

129 Returns ``{}`` when no agent in ``specs`` opted into memory (existing 

130 deployments stay byte-identical — no template read, no wire cost). Raises 

131 :class:`ValueError` (propagated from :func:`load_memory_prompt`) when a memory 

132 agent exists but the template can't be loaded; the **caller** decides how to 

133 log and degrade — the high-frequency bridge-ingress path dedups the error to a 

134 single log, while rarer call sites (e.g. the outbox retry) log per occurrence. 

135 

136 Shared by every bridge path that originates an agent invocation 

137 (:meth:`~calfcord.bridge.ingress.BridgeIngress._memory_prompt_deps` 

138 and the outbox retry-with-feedback publish) so the "is memory enabled + 

139 load the template" decision lives in exactly one place. 

140 """ 

141 if not any(spec.memory for spec in specs): 

142 return {} 

143 return {MEMORY_PROMPT_DEPS_KEY: load_memory_prompt()}
