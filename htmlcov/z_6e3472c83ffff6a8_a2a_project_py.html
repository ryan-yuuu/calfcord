1"""Render A2A activity (consults + handoffs) into the unified Discord audit channel. 

2 

3The stateful :class:`~calfcord.bridge.a2a_dispatch.A2ADispatcher` pulls native 

4``message_agent`` consults and ``HandoffEvent``s off a run's stream and emits the 

5:class:`~calfcord.bridge.a2a_dispatch.A2AProjection` dataclasses; this projector 

6turns each into Discord posts in the unified A2A channel (re-homed from the old 

7``private_chat`` tool, spec §6.2). It is the ``A2AProjectorLike`` collaborator the 

8bridge's :class:`~calfcord.bridge.mention_handler.MentionHandler` drives. 

9 

10Anchoring (round-3 M3): one thread per **``correlation_id``** (one human turn's A2A 

11activity), created lazily on the first projection for that turn — its message is the 

12thread's starter. Every later request/reply/reject/handoff/fault for the same 

13``correlation_id`` posts into that thread. Peer identity comes from the projection 

14dataclasses (already resolved by the dispatcher from the request's ``args["name"]``, 

15the one source stable across success and rejection); personas come from the pure 

16:func:`~calfcord.bridge.persona_resolve.persona_for` (no roster). 

17 

18Best-effort audit: the bridge is no longer the A2A *transport* (the consult already 

19happened inside the agent runtime and its reply is in-hand on the stream), so a 

20failed Discord render is logged and swallowed — it never faults the human turn. 

21"""

43_EMPTY_PLACEHOLDER = "(empty response)" 

44"""Discord rejects an empty webhook message; substitute this for empty content."""

46_SYSTEM_PERSONA = Persona(name="a2a") 

47"""Persona for *meta* notes (rejections, handoffs, faults) that are not a peer's 

48own words — rendered as a system annotation, not attributed to an agent (D-2)."""

50# Thread-name shaping (re-homed from the old private_chat tool). 

51_THREAD_NAME_MAX_TOTAL = 100 

52"""Discord's hard cap on thread names; exceeding it 400s the create.""" 

53_THREAD_NAME_CONTENT_MAX = 40 

54"""Soft cap on the topic-tail portion (after ``caller→peer: ``).""" 

55_THREAD_NAME_EMPTY_PLACEHOLDER = "<empty>" 

56"""Substituted when the seed content is empty, avoiding a bare-trailing-space name."""

59def _build_thread_name(caller: str, peer: str, content: str) -> str: 

60 """Produce a thread name like ``'conan→scribe: please summarize the doc'``. 

61 

62 Control characters are normalized to spaces and runs collapsed; the topic tail 

63 is truncated to :data:`_THREAD_NAME_CONTENT_MAX` and the whole name hard-capped 

64 at :data:`_THREAD_NAME_MAX_TOTAL` (Discord's limit). The ``→`` (U+2192) 

65 separator is char-counted (not byte-counted) by Discord, so it is cheap. 

66 """ 

67 cleaned = " ".join("".join(c if c.isprintable() else " " for c in content).split()) 

68 if not cleaned: 

69 cleaned = _THREAD_NAME_EMPTY_PLACEHOLDER 

70 name = f"{caller}→{peer}: {cleaned[:_THREAD_NAME_CONTENT_MAX]}" 

71 return name[:_THREAD_NAME_MAX_TOTAL]

74class A2AProjector: 

75 """Renders :class:`A2AProjection`s into the unified A2A audit channel. 

76 

77 One instance per bridge; its ``correlation_id → thread_id`` map is the only 

78 state, mirroring one thread per human turn's A2A activity. 

79 """ 

80 

81 def __init__(self, resolver: A2AChannelResolver, personas: DiscordPersonaSender) -> None: 

82 self._resolver = resolver 

83 self._personas = personas 

84 self._threads: dict[str, int] = {} 

85 self._channel_id: int | None = None 

86 

87 async def project(self, projection: A2AProjection) -> None: 

88 """Render one projection (best-effort — a Discord failure is swallowed).""" 

89 try: 

90 await self._dispatch(projection) 

91 except Exception: 

92 # Best-effort audit: a failed render must never fault the human turn 

93 # (this runs inside the mention handler's stream-drain loop). 

94 logger.warning( 

95 "A2A projection failed (audit gap); continuing kind=%s", 

96 type(projection).__name__, 

97 exc_info=True, 

98 ) 

99 

100 async def project_fault(self, call: A2ACall) -> None: 

101 """Note a consult that never got a reply because the peer faulted (D-2).""" 

102 try: 

103 await self._emit( 

104 call.correlation_id, 

105 _SYSTEM_PERSONA, 

106 f"⚠️ {call.peer} did not reply — the consult faulted before a response.", 

107 thread_name=_build_thread_name(call.caller, call.peer, call.message), 

108 ) 

109 except Exception: 

110 logger.warning("A2A fault note failed (audit gap); continuing", exc_info=True) 

111 

112 async def _dispatch(self, projection: A2AProjection) -> None: 

113 if isinstance(projection, A2ARequest): 

114 await self._emit( 

115 projection.correlation_id, 

116 persona_for(projection.caller), 

117 projection.message, 

118 thread_name=_build_thread_name(projection.caller, projection.peer, projection.message), 

119 ) 

120 elif isinstance(projection, A2AReply): 

121 await self._emit( 

122 projection.correlation_id, 

123 persona_for(projection.peer), 

124 projection.text, 

125 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

126 ) 

127 elif isinstance(projection, A2AReject): 

128 # A rejected consult (peer offline / cycle / self) is a system note, 

129 # not a peer post — the peer never spoke (D-2). 

130 await self._emit( 

131 projection.correlation_id, 

132 _SYSTEM_PERSONA, 

133 f"⚠️ consult to {projection.peer} was rejected: {projection.text}", 

134 thread_name=_build_thread_name(projection.caller, projection.peer, projection.text), 

135 ) 

136 elif isinstance(projection, A2AHandoff): 136 ↛ 147line 136 didn't jump to line 147 because the condition on line 136 was always true

137 reason = f": {projection.reason}" if projection.reason else "" 

138 await self._emit( 

139 projection.correlation_id, 

140 _SYSTEM_PERSONA, 

141 f"↪ {projection.emitter} handed off to {projection.target}{reason}", 

142 thread_name=_build_thread_name(projection.emitter, projection.target, projection.reason), 

143 ) 

144 else: 

145 # Exhaustiveness guard: a 5th A2AProjection variant added without a 

146 # branch here is a mypy error, not a silent no-op render. 

147 assert_never(projection) 

148 

149 async def _channel(self) -> int: 

150 if self._channel_id is None: 

151 self._channel_id = await self._resolver.resolve_unified_channel() 

152 return self._channel_id 

153 

154 async def _emit(self, correlation_id: str, persona: Persona, content: str, *, thread_name: str) -> None: 

155 """Post ``content`` under ``persona`` into ``correlation_id``'s thread. 

156 

157 Creates the thread lazily on the first projection for the turn — the first 

158 content chunk becomes the thread's anchor/starter message — then posts any 

159 remaining chunks into the thread. Content over Discord's 2000-char limit is 

160 split (:func:`chunk_split`); empty content uses :data:`_EMPTY_PLACEHOLDER`. 

161 """ 

162 channel_id = await self._channel() 

163 chunks = chunk_split(content) or [_EMPTY_PLACEHOLDER] 

164 thread_id = self._threads.get(correlation_id) 

165 if thread_id is None: 

166 sent = await self._personas.send(persona, channel_id=channel_id, content=chunks[0]) 

167 thread_id = await self._resolver.create_anchored_thread(channel_id, sent.id, name=thread_name) 

168 self._threads[correlation_id] = thread_id 

169 rest = chunks[1:] 

170 else: 

171 rest = chunks 

172 for chunk in rest: 

173 await self._personas.send(persona, channel_id=channel_id, content=chunk, thread_id=thread_id)
