1"""The editable-field registry: one source of truth for ``calfcord agent`` edits. 

2 

3Three surfaces onto an agent's ``.md`` — the interactive ``edit`` 

4menu (which rows exist, in what order, showing each field's current value), the 

5non-interactive ``set`` command (which ``--flag`` writes which field), and 

6``show`` (how each field renders) — must agree on the *same* set of editable 

7fields or they drift: a flag with no menu row, a menu row ``show`` can't render, 

8a field one surface validates differently from another. This module is that 

9single source: :data:`FIELDS` is the ordered list every surface consumes, and 

10:func:`render_value` is the one renderer the menu and ``show`` share. 

11 

12Two kinds of field live here: 

13 

14* **Simple** (``text`` / ``select`` / ``int`` / ``bool``) — a scalar frontmatter 

15 value written through the one validated-atomic seam, 

16 :func:`calfcord.agents.md_writer._update_fields`, via :func:`write_simple_field`. 

17 Validation is delegated to :class:`~calfcord.agents.definition.AgentDefinition` 

18 (a bad ``thinking_effort`` or an out-of-range ``history_turns`` raises there), 

19 so this module never re-encodes a constraint pydantic already owns — keeping 

20 the two from drifting. 

21* **Compound** (``provider_model`` / ``tools`` / ``prompt``) — fields with 

22 dedicated editors (the provider+model pair shares the validated 

23 provider flow; ``tools`` reuses the checkbox editor; ``prompt`` rewrites the 

24 body via :func:`calfcord.agents.md_writer.update_system_prompt`). These are 

25 *not* written through :func:`write_simple_field`; they appear in :data:`FIELDS` 

26 only so the menu and ``show`` can list and render them. 

27 

28The module imports only :mod:`calfcord.agents` seams — no provider SDK — so it 

29stays importable from the lightweight CLI entry point. 

30""" 

31

45# The closed set of field kinds. A ``Literal`` (not a bare ``str``) so a typo in a 

46# ``Field(...)`` literal — e.g. ``"boool"`` — is a type error a checker can flag, 

47# rather than a field with no matching dispatch branch that fails only when an 

48# operator happens to select it. 

49FieldKind = Literal["text", "select", "int", "bool", "provider_model", "tools", "prompt"] 

50 

51THINKING_EFFORTS = ("none", "minimal", "low", "medium", "high", "xhigh", "max") 

52"""The :data:`~calfcord.agents.definition.ThinkingEffort` tiers, as a tuple for 

53the ``thinking_effort`` field's ``choices``. Mirrors the ``Literal`` in 

54:mod:`calfcord.agents.definition`; kept here as the menu/flag choice list so a 

55``select`` row and the pydantic ``Literal`` enumerate the same values.""" 

56 

57# This tuple is hand-authored but must enumerate exactly the ``ThinkingEffort`` 

58# Literal it mirrors; assert it at import so a drift (a tier added to one and not 

59# the other) becomes a hard failure the test suite catches, not a silently wrong 

60# menu/validator. 

61assert get_args(ThinkingEffort) == THINKING_EFFORTS, "THINKING_EFFORTS drifted from ThinkingEffort" 

62 

63# The longest single-line preview shown for the system-prompt body in the edit 

64# menu / ``show``. Long enough to be recognizable, short enough not to wrap a 

65# typical terminal row. 

66_PROMPT_PREVIEW_LEN = 60 

67 

68# How many tool tokens to spell out before collapsing the rest into a "(+N)" 

69# count, so an agent with the full builtin set renders as a short, scannable 

70# label rather than a wrapping comma list. 

71_TOOLS_PREVIEW_COUNT = 2

74@dataclass(frozen=True) 

75class Field: 

76 """One editable field, shared by the ``edit`` menu, ``set`` flags, and ``show``. 

77 

78 ``key`` is the :class:`~calfcord.agents.definition.AgentDefinition` attribute 

79 (and frontmatter key) the field maps to — except the special value 

80 ``"system_prompt"``, which is the Markdown *body*, not a metadata key. 

81 ``kind`` drives both how the value is edited and how :func:`render_value` 

82 formats it. ``flag`` is the ``agent set`` long option that writes this field 

83 non-interactively. ``choices`` enumerates the allowed values for a 

84 ``select`` field; ``int_min`` / ``int_max`` bound an ``int`` field (mirroring 

85 the pydantic constraint so the menu can show the range without importing the 

86 model's internals). 

87 """ 

88 

89 key: str 

90 label: str 

91 kind: FieldKind 

92 flag: str 

93 choices: tuple[str, ...] | None = None 

94 int_min: int | None = None 

95 int_max: int | None = None 

96 

97 def __post_init__(self) -> None: 

98 """Assert the per-kind shape so a malformed registry fails loudly at import. 

99 

100 ``FIELDS`` is hand-authored; a ``select`` row with no ``choices`` would 

101 otherwise degrade to a silently empty menu (``field.choices or ()`` in the 

102 editor), and stray ``int_min``/``int_max`` on a non-``int`` field would be 

103 dead state. Catch both here — at module import — rather than at edit time. 

104 """ 

105 if self.kind == "select": 

106 assert self.choices, f"{self.key!r}: a select field needs choices" 

107 else: 

108 assert self.choices is None, f"{self.key!r}: only a select field takes choices" 

109 if self.kind == "int": 

110 assert self.int_min is not None and self.int_max is not None, ( 

111 f"{self.key!r}: an int field needs int_min and int_max" 

112 ) 

113 else: 

114 assert self.int_min is None and self.int_max is None, ( 

115 f"{self.key!r}: only an int field takes int_min/int_max" 

116 ) 

117 

118 

119# Menu order. ``provider_model`` is ONE row spanning provider+model (the 

120# provider editor writes both together through the validated provider flow); 

121# ``tools`` and ``system_prompt`` have dedicated editors. The ordering puts the 

122# fields an operator most often tweaks (identity, then provider, then tools and 

123# prompt) before the runtime-tuning knobs (effort, history, memory, avatar). 

124FIELDS: list[Field] = [ 

125 Field("description", "Description", "text", "--description"), 

126 Field("display_name", "Display name", "text", "--display-name"), 

127 Field("provider_model", "Provider / model", "provider_model", "--model"), 

128 Field("tools", "Tools", "tools", "--tools"), 

129 Field("system_prompt", "System prompt", "prompt", "--system-prompt"), 

130 Field( 

131 "thinking_effort", 

132 "Thinking effort", 

133 "select", 

134 "--thinking-effort", 

135 choices=THINKING_EFFORTS, 

136 ), 

137 Field("history_turns", "History turns", "int", "--history-turns", int_min=0, int_max=100), 

138 Field("memory", "Memory", "bool", "--memory"), 

139 Field("avatar_url", "Avatar URL", "text", "--avatar-url"), 

140] 

141 

142# Fast lookup by key for the ``set`` command (resolve a ``--flag`` to its field) 

143# and any caller that needs one field by name. Built from FIELDS so it can't 

144# drift from the ordered list. 

145FIELDS_BY_KEY: dict[str, Field] = {f.key: f for f in FIELDS}

148def render_value(defn: AgentDefinition, field: Field) -> str: 

149 """Human-readable CURRENT value of ``field`` on ``defn``, for the menu and ``show``. 

150 

151 One renderer so the edit menu's "current value" column and the ``show`` 

152 command can never format the same field differently. Compound fields get a 

153 compact preview (``tools`` collapses the tail into ``(+N)``; 

154 ``provider_model`` joins the pair; ``system_prompt`` previews the first line); 

155 an unset optional field renders as ``(default)`` or ``(none)`` rather than a 

156 bare ``None`` so the menu reads cleanly. 

157 """ 

158 if field.kind == "provider_model": 

159 provider = defn.provider or "(default)" 

160 model = defn.model or "(default)" 

161 return f"{provider} · {model}" 

162 

163 if field.kind == "tools": 

164 return _render_tools(defn.tools) 

165 

166 if field.kind == "prompt": 

167 return _render_prompt(defn.system_prompt) 

168 

169 value = getattr(defn, field.key) 

170 

171 if field.kind == "bool": 

172 return "on" if value else "off" 

173 

174 if value is None or value == "": 

175 # ``thinking_effort`` unset means the provider default applies; an unset 

176 # avatar_url means the webhook default. Both read better as "(default)" 

177 # than an empty cell. 

178 return "(default)" 

179 

180 return str(value)

183def _render_tools(tools: tuple[str, ...] | None) -> str: 

184 """Render a tool tuple as a short label: first few names, then ``(+N)``. 

185 

186 ``None`` (frontmatter ``tools:`` omitted) means "all builtins" — shown as 

187 ``(all builtins)`` rather than an empty cell, matching the loader's 

188 omitted→all semantics. An explicit empty tuple is the deliberate "no tools" 

189 opt-out, shown as ``(none)``. 

190 """ 

191 if tools is None: 

192 return "(all builtins)" 

193 if not tools: 

194 return "(none)" 

195 head = ", ".join(tools[:_TOOLS_PREVIEW_COUNT]) 

196 extra = len(tools) - _TOOLS_PREVIEW_COUNT 

197 return f"{head} (+{extra})" if extra > 0 else head

200def _render_prompt(system_prompt: str) -> str: 

201 """Render the body as a single-line preview, truncated with an ellipsis.""" 

202 return truncate(system_prompt, _PROMPT_PREVIEW_LEN)

205def truncate(text: str, limit: int) -> str: 

206 """Flatten internal whitespace and clip ``text`` to ``limit`` chars with an ellipsis. 

207 

208 Collapsing newlines/runs of whitespace to single spaces keeps a multi-line 

209 value on one row (a menu line, or a ``list`` table cell); the trailing ``…`` 

210 signals the value was clipped so the reader knows to use ``show`` for the rest. 

211 Shared by the edit-menu/``show`` prompt preview here and the ``list`` table in 

212 :mod:`calfcord.cli.agent_inspect` so the two can't clip differently. 

213 """ 

214 flat = " ".join(text.split()) 

215 if len(flat) <= limit: 

216 return flat 

217 return flat[: limit - 1].rstrip() + "…"

220def write_simple_field(md_path: Path, field: Field, raw: str) -> AgentDefinition: 

221 """Validate and write a SIMPLE field's new value through the one shared seam. 

222 

223 Both the ``edit`` menu (after prompting) and the ``set`` command (from a 

224 ``--flag`` argument) call this for ``text`` / ``select`` / ``int`` / ``bool`` 

225 fields, so the coercion and the validate-before-write path are defined once. 

226 The value is coerced to the frontmatter type the field expects and handed to 

227 :func:`calfcord.agents.md_writer._update_fields`, which builds and validates a 

228 synthetic :class:`AgentDefinition` before any disk write — so an out-of-range 

229 ``history_turns`` or a bad ``thinking_effort`` raises there, with the on-disk 

230 file untouched. Coercion is intentionally thin: only the int parse lives here 

231 (pydantic can't turn ``"abc"`` into the ``ValueError`` the caller wants); 

232 every domain constraint (range, allowed choices) stays owned by 

233 :class:`AgentDefinition` so this helper can't drift from it. 

234 

235 Compound fields (``provider_model`` / ``tools`` / ``prompt``) have dedicated 

236 editors and must NOT be routed here. 

237 

238 Raises: 

239 ValueError: ``field`` is not a simple field, an ``int`` value isn't 

240 numeric, or :func:`~calfcord.agents.md_writer._update_fields` rejects 

241 the coerced value (bad choice, out-of-range int, …). The on-disk file 

242 is unchanged. 

243 OSError: a filesystem error during the atomic write. The on-disk file is 

244 unchanged. 

245 """ 

246 value = _coerce_simple(field, raw) 

247 return md_writer._update_fields(md_path, {field.key: value})

250def _coerce_simple(field: Field, raw: str) -> object: 

251 """Coerce a ``set``-flag / prompt string to the field's frontmatter type. 

252 

253 ``bool`` accepts the usual truthy/falsey spellings so ``--memory on`` and 

254 ``--memory false`` both work; ``int`` parses to ``int`` (a non-numeric value 

255 raises a precise :class:`ValueError` here rather than a confusing pydantic 

256 one); ``text`` / ``select`` pass through as the raw string and let 

257 :class:`~calfcord.agents.definition.AgentDefinition` enforce length / choice. 

258 """ 

259 if field.kind == "bool": 

260 return _coerce_bool(field, raw) 

261 if field.kind == "int": 

262 try: 

263 return int(raw) 

264 except ValueError as e: 

265 raise ValueError(f"{field.flag} expects an integer, got {raw!r}") from e 

266 if field.kind in ("text", "select"): 

267 return raw 

268 raise ValueError(f"{field.key!r} (kind={field.kind!r}) is not a simple field; it has a dedicated editor") 

269

271def _coerce_bool(field: Field, raw: str) -> bool: 

272 """Parse a boolean spelling, raising a clear error on anything unrecognized.""" 

273 normalized = raw.strip().lower() 

274 if normalized in ("1", "true", "yes", "on"): 

275 return True 

276 if normalized in ("0", "false", "no", "off"): 

277 return False 

278 raise ValueError(f"{field.flag} expects a boolean (on/off, true/false, yes/no), got {raw!r}")
