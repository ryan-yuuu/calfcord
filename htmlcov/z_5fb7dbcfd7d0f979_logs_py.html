1"""``calfcord logs [component] [-f]`` — tail the supervisor's per-process logs. 

2 

3Process Compose writes each supervised process's stdout/stderr to 

4``$CALFCORD_HOME/state/logs/<name>.log`` (the §13.2 ``log_location`` contract), 

5plus its own ``process-compose.log``. This command reads those files straight 

6off disk — a path that works even when the supervisor REST daemon is down (the 

7file is the durable record), so "what did the broker say before it died?" is 

8always answerable. 

9 

10Why read files rather than the REST log endpoint: the on-disk path is 

11supervisor-independent and host-local with no port to derive or daemon to be up; 

12it is the lowest-coupling way to surface "what happened here." A live process's 

13logs are *also* available over REST (``GET /process/logs/...``), but that path is 

14only useful while the daemon answers, so it is left to a future need. 

15 

16The set of names that *may* have a log is not hardcoded here — that would drift 

17from what the generator actually declares. It is reconstructed from the same two 

18seams the generator uses: the reserved substrate/component names 

19(:data:`calfcord.supervisor.compose._RESERVED_PROCESS_NAMES`) and the host's 

20agent ids (:func:`calfcord.cli._agents.detect_agents`, the seam ``start`` and 

21``agent list`` consume), plus the supervisor's own ``process-compose`` log. 

22 

23This module imports only ``_agents.detect_agents`` and the ``compose`` 

24log-path / name-set seams, both of which are import-light. 

25 

26The native-install guard (a dev run has no ``$CALFCORD_HOME`` and therefore no 

27state/logs dir) lives in the ``main.py`` veneer, alongside every other 

28supervisor-scoped verb's identical guard — this module is handed a concrete 

29``home`` and concerns itself only with the files under it. 

30"""

47# The supervisor's own log (``process-compose up -L ...``) sits beside the 

48# per-process logs and is a legitimate, useful tail target (it captures 

49# supervisor-level start/restart events), so it is part of the known set even 

50# though it is not a process the generator declares. Its name is the shared stem 

51# ``compose`` owns so it can never drift from the filename ``lifecycle`` writes. 

52_SUPERVISOR_LOG_NAME = SUPERVISOR_LOG_STEM 

53 

54# How long the follow loop waits between polls for appended bytes. Small enough 

55# to feel live, large enough that the loop never busy-spins a CPU. Injectable via 

56# the ``sleep`` parameter so tests drive (and bound) the loop with no real wait. 

57_FOLLOW_POLL_INTERVAL_SECONDS = 0.5

60def _known_names(agents_dir: Path) -> list[str]: 

61 """The component names that may have a log, substrate/components first. 

62 

63 Built from the same seams the generator uses so the set can never drift from 

64 what ``calfcord start`` actually declares: the reserved substrate + fixed 

65 component slots, the host's agent ids, and the supervisor's own log. Order is 

66 deterministic (substrate, fixed components, sorted agents, supervisor) so the 

67 merged "all logs" view and the unknown-name hint read predictably. 

68 """ 

69 # ``_RESERVED_PROCESS_NAMES`` is a frozenset; pin a stable, readable order 

70 # (substrate then the fixed components) rather than its hash order. 

71 ordered_reserved = [ 

72 name for name in ("broker", "bridge", "tools", "router") 

73 if name in _RESERVED_PROCESS_NAMES 

74 ] 

75 # MCP slots come from the same no-secrets mcp.json seam the compose 

76 # generator uses. Tolerant on purpose: a broken mcp.json must not take the 

77 # logs command down with it ("always show what the broker said before it 

78 # died") — the strict readers (start, mcp start) surface the config error. 

79 try: 

80 mcp_slots = [mcp_slot_name(s) for s in list_server_names(resolve_config_path())] 

81 except McpConfigError: 

82 mcp_slots = [] 

83 return [ 

84 *ordered_reserved, 

85 *detect_agents(agents_dir), 

86 *mcp_slots, 

87 _SUPERVISOR_LOG_NAME, 

88 ]

91def _emit_file(path: Path, *, label: str | None, out: Callable[..., None]) -> None: 

92 """Print a log file's current contents, optionally line-prefixed. 

93 

94 ``label`` is the component name in the merged view (each line becomes 

95 ``<name> | <line>`` so a reader can tell who said what); ``None`` for a single 

96 explicit component streams the raw lines. Trailing-newline-only lines are 

97 dropped so an empty tail does not print blank rows. This is the one-shot dump 

98 only; the follow path tracks its own per-file byte offsets, so there is no 

99 offset to hand back here. 

100 

101 Decoding is tolerant (``errors="replace"``) to match the follow path: a log 

102 line with non-UTF-8 bytes (a partial multibyte write, a binary splat, a 

103 mis-encoded child) must never raise an uncaught ``UnicodeDecodeError`` and 

104 crash the command — "always show what the broker said before it died" holds 

105 even when what it said was not clean UTF-8. 

106 """ 

107 data = path.read_text(encoding="utf-8", errors="replace") 

108 for line in data.splitlines(): 

109 out(f"{label} | {line}" if label is not None else line)

112def _follow( 

113 targets: list[tuple[str, Path]], 

114 *, 

115 labeled: bool, 

116 out: Callable[..., None], 

117 sleep: Callable[[float], None], 

118 poll_interval: float, 

119) -> int: 

120 """Stream existing content then poll each target for *appended* bytes. 

121 

122 ``targets`` are ``(name, path)`` pairs. A path that does not exist yet is 

123 tolerated (a slot may clock in moments later): it is simply skipped this pass 

124 and its offset stays at 0, so when the file appears its full content is emitted 

125 on the next poll. Each pass reads only the bytes past the per-file offset, so a 

126 long-lived process never re-prints history. 

127 

128 The loop runs until ``KeyboardInterrupt`` (a real Ctrl-C, or the injected 

129 sleep in tests), then returns 0 — a clean, expected exit, not a failure. The 

130 bounded ``sleep`` between passes keeps it from busy-spinning. 

131 """ 

132 label_for = (lambda name: name) if labeled else (lambda _name: None) 

133 offsets: dict[str, int] = dict.fromkeys((name for name, _ in targets), 0) 

134 

135 try: 

136 while True: 

137 for name, path in targets: 

138 if not path.is_file(): 

139 continue 

140 data = path.read_bytes() 

141 if len(data) <= offsets[name]: 

142 # Truncated (rotation) or unchanged: reset to the new end so a 

143 # rotated-smaller file is not re-streamed from a stale offset. 

144 offsets[name] = len(data) 

145 continue 

146 fresh = data[offsets[name]:].decode("utf-8", errors="replace") 

147 offsets[name] = len(data) 

148 label = label_for(name) 

149 for line in fresh.splitlines(): 

150 out(f"{label} | {line}" if label is not None else line) 

151 sleep(poll_interval) 

152 except KeyboardInterrupt: 

153 return 0

156def tail( 

157 home: Path, 

158 *, 

159 agents_dir: Path, 

160 component: str | None = None, 

161 follow: bool = False, 

162 out: Callable[..., None] = print, 

163 sleep: Callable[[float], None] = time.sleep, 

164 poll_interval: float = _FOLLOW_POLL_INTERVAL_SECONDS, 

165) -> int: 

166 """Tail unified or per-component supervisor logs under ``home``. 

167 

168 ``component`` ``None`` streams every component that has a log (each line 

169 labeled with its name); a name streams just that file unlabeled. ``follow`` 

170 keeps streaming appended bytes until Ctrl-C. ``out`` and ``sleep`` are 

171 injected so the follow loop is testable without real stdout or real time. 

172 

173 Returns a POSIX exit code: 

174 

175 * **1** — the ``state/logs`` dir does not exist (the workspace was never 

176 started), or ``component`` is not a known name (a typo). Both print a clean, 

177 actionable ``error:`` rather than raising — a missing workspace or a typo is 

178 operator input, not an infrastructure bug. 

179 * **0** — otherwise, including the benign "this slot has no log yet" case (a 

180 declared component that never clocked in), which is informational, not an 

181 error. 

182 """ 

183 log_dir = home / "state" / "logs" 

184 if not log_dir.is_dir(): 

185 out( 

186 f"error: no logs under {log_dir} — the workspace may not be running " 

187 "(start it with: calfcord start)." 

188 ) 

189 return 1 

190 

191 names = _known_names(agents_dir) 

192 

193 if component is not None and component not in names: 

194 out( 

195 f"error: unknown component {component!r}; choose one of: " 

196 f"{', '.join(names)} (or omit it to tail all)." 

197 ) 

198 return 1 

199 

200 selected = [component] if component is not None else names 

201 labeled = component is None # the merged view labels; a single component does not 

202 

203 if follow: 

204 targets = [(name, Path(_log_location(str(home), name))) for name in selected] 

205 return _follow( 

206 targets, labeled=labeled, out=out, sleep=sleep, poll_interval=poll_interval 

207 ) 

208 

209 # One-shot: dump each selected file's current contents. An explicitly named 

210 # component with no file yet is informational (the slot never ran); in the 

211 # merged view an absent file is simply skipped so the output shows only what 

212 # exists. 

213 emitted_any = False 

214 for name in selected: 

215 path = Path(_log_location(str(home), name)) 

216 if not path.is_file(): 

217 if component is not None: 

218 out(f"no logs yet for {name} (it may not have started).") 

219 continue 

220 _emit_file(path, label=name if labeled else None, out=out) 

221 emitted_any = True 

222 

223 if labeled and not emitted_any: 

224 # The dir exists but nothing has produced a log yet — tell the operator 

225 # plainly rather than printing nothing and looking broken. 

226 out("no logs yet (no component has produced output).") 

227 

228 return 0
