1"""The single seam that knows Process Compose's REST routes (design §13.2). 

2 

3Everything calfcord does to a running supervisor — list/state/logs, start/stop/ 

4restart a roster member, push an updated project — goes through this thin async 

5wrapper. Confining the wire contract to one module is deliberate (design §12.4, 

6Risk #2): Process Compose's API is version-fragile and the published docs are 

7partly wrong, so a binary upgrade is a one-file change here, and the CLI veneer 

8above stays the stable surface. 

9 

10The routes are pinned to Process Compose ``v1.110.0`` from the Phase-0 spike, 

11and several correct an upstream doc error: 

12 

13* single-process state is ``GET /process/{name}`` — ``GET /process/{name}/state`` 

14 is a 404, so it is intentionally absent; 

15* **stop is a ``PATCH``** (``PATCH /process/stop/{name}``), not a ``POST`` — 

16 start and restart are ``POST``; 

17* logs carry the window in the path: ``GET /process/logs/{name}/{end}/{limit}``. 

18 

19An optional ``PC_API_TOKEN`` rides only in the ``X-PC-Token-Key`` header. Every 

20call is an infrastructure call against the local supervisor, so a non-2xx is a 

21genuine bug (a missing process / a wedged server), not something an LLM can 

22adapt to — per the CLAUDE.md convention these ``raise RuntimeError`` with the 

23caller, route, and status rather than returning an ``"error: ..."`` string. 

24""" 

25

36# The supervisor is detached but local; calls should be near-instant. The 

37# timeout exists only so a wedged server fails loudly instead of hanging the CLI. 

38_HTTP_TIMEOUT = 10.0 

39 

40# Process Compose's optional API auth (design §13.2): a >=20ch shared key sent 

41# in this header. Confined here so no other module reconstructs the scheme. 

42_TOKEN_HEADER = "X-PC-Token-Key" 

43 

44# Cap on how much of PC's response body we append to an error: enough to carry its 

45# informative reason (e.g. "process X is not defined") without an adversarially 

46# large body flooding the message. The token rides a HEADER, never the body, so 

47# echoing the body leaks no secret. 

48_MAX_BODY_CHARS = 500 

49

51class ProcessComposeError(RuntimeError): 

52 """A failed Process Compose REST call, carrying the HTTP status structurally. 

53 

54 Stays a ``RuntimeError`` (the package-wide infra-failure contract) so existing 

55 ``except RuntimeError`` callers and the readiness poll are unaffected, while 

56 adding a ``status_code`` so callers can branch on the response class WITHOUT 

57 string-parsing the message. ``status_code`` is ``None`` for a transport failure 

58 (no HTTP response was received), so a 4xx/5xx branch reads "no status" as "not 

59 a client error". 

60 """ 

61 

62 def __init__(self, message: str, *, status_code: int | None = None) -> None: 

63 super().__init__(message) 

64 self.status_code = status_code

67class ProcessComposeClient: 

68 """Async wrapper over the Process Compose REST API on ``http://localhost:{port}``. 

69 

70 ``port`` defaults to ``8080`` (the supervisor default); calfcord derives a 

71 per-home port and passes it both to ``process-compose up -p`` and here so a 

72 second ``$CALFCORD_HOME`` does not collide. ``base_url`` overrides ``port`` 

73 outright for non-default hosts/ports. ``token`` is the optional 

74 ``PC_API_TOKEN`` and ``client_factory`` injects an ``httpx.AsyncClient`` for 

75 tests — mirroring the ``client_factory`` seam in ``cli/doctor.py`` and the 

76 Codex prompt resolver, so unit tests stub the transport and never touch a 

77 real supervisor. 

78 """ 

79 

80 def __init__( 

81 self, 

82 *, 

83 port: int = 8080, 

84 base_url: str | None = None, 

85 token: str | None = None, 

86 client_factory: Callable[[], httpx.AsyncClient] | None = None, 

87 ) -> None: 

88 self._base_url = (base_url or f"http://localhost:{port}").rstrip("/") 

89 self._token = token 

90 self._client_factory = client_factory or self._default_client 

91 

92 @staticmethod 

93 def _default_client() -> httpx.AsyncClient: 

94 return httpx.AsyncClient(timeout=_HTTP_TIMEOUT) 

95 

96 def _headers(self) -> dict[str, str] | None: 

97 # Only send the auth header when a token is configured, so an unauthed 

98 # supervisor (the common dev case) sees a clean request. 

99 return {_TOKEN_HEADER: self._token} if self._token else None 

100 

101 async def _request( 

102 self, 

103 *, 

104 caller: str, 

105 method: str, 

106 route: str, 

107 json: Any | None = None, 

108 ) -> httpx.Response: 

109 """Issue one request and raise a context-rich ``RuntimeError`` on non-2xx. 

110 

111 Each call opens its own client via the factory (the resolver pattern) so 

112 the wrapper holds no long-lived connection — the supervisor is local and 

113 calls are sparse, so a per-call client keeps lifetime management trivial. 

114 ``json`` (the project-update path) is serialized with ``Content-Type: 

115 application/json``, which the v1.110.0 server requires for ``POST /project``. 

116 """ 

117 async with self._client_factory() as client: 

118 try: 

119 response = await client.request( 

120 method, 

121 f"{self._base_url}{route}", 

122 headers=self._headers(), 

123 json=json, 

124 ) 

125 except httpx.RequestError as exc: 

126 # Transport failure (server not up yet / wedged / wrong port) — an 

127 # infra failure, so normalize to the same RuntimeError contract as 

128 # a non-2xx, letting callers (e.g. the start readiness poll) handle 

129 # one error type instead of leaking httpx. No HTTP status exists, so 

130 # status_code stays None. 

131 raise ProcessComposeError( 

132 f"{caller}: process-compose {method} {route} failed to connect: {exc}" 

133 ) from exc 

134 if not response.is_success: 

135 # Carry the status structurally (so callers branch on it without parsing 

136 # the message) AND append PC's informative body, truncated — its reasons 

137 # (e.g. "process X is not defined") are otherwise dropped. The body is 

138 # safe to echo: the token rides a header, never the body. 

139 body = (response.text or "").strip() 

140 detail = f": {body[:_MAX_BODY_CHARS]}" if body else "" 

141 raise ProcessComposeError( 

142 f"{caller}: process-compose {method} {route} " 

143 f"failed with HTTP {response.status_code}{detail}", 

144 status_code=response.status_code, 

145 ) 

146 return response

219 async def update_project(self, yaml_text: str) -> Any: 

220 """Apply an updated project to the running supervisor (``POST /project``). 

221 

222 Powers the dynamic-add path (an agent authored after ``start``) and the 

223 priming reconcile. Callers pass the *rendered project YAML* (the one body 

224 type the renderer produces); the v1.110.0 server's ``POST /project`` 

225 decodes **JSON**, not YAML — a raw-YAML body is rejected with HTTP 400 — 

226 so we parse here and ship JSON, exactly as the ``process-compose project 

227 update -f <yaml>`` CLI does. Confining that YAML→JSON detail to this seam 

228 keeps the wire contract in one place (design §12.4, Risk #2). 

229 

230 A no-op reconcile answers **207 Multi-Status** with a per-process result 

231 map (e.g. ``{"broker": "error"}`` for an unchanged, still-running 

232 process); 207 is a 2xx, so it is returned, not raised — the supervisor 

233 reconciles without bouncing unchanged processes in steady state (the 

234 once-only first-update bounce, upstream #494, is absorbed by the caller's 

235 priming reconcile). 

236 """ 

237 response = await self._request( 

238 caller="update_project", 

239 method="POST", 

240 route="/project", 

241 json=yaml.safe_load(yaml_text), 

242 ) 

243 return response.json()
