1"""CLI entry point for the ``calfkit-tools`` deployment. 

2 

3Hosts every :class:`ToolNodeDef` registered in :data:`TOOL_REGISTRY` on a 

4single calfkit :class:`Worker`. Standalone process — separate from the 

5bridge and the agent runner — so the tool lifecycle is decoupled from 

6both, matching calfkit's tool-as-deployment model. 

7 

8This deployment intentionally has no read access to ``agents/*.md``. 

9Agent identities (display name, avatar, description, tools) arrive at 

10the tool body via the phonebook the bridge places in ``deps`` on every 

11invocation. 

12 

13The runner is **resource-light by design**: it connects the process-wide 

14:class:`calfkit.client.Client` (with a private reply topic distinct from the 

15bridge's ``discord.outbox``, so target-agent replies route here and are NOT 

16re-projected by the bridge's outbox poster) and exposes it as the worker-scoped 

17``a2a_client`` resource. Any *tool-specific* live resource is owned by the tool 

18itself via a node-scoped ``@resource`` bracket that calfkit builds only when 

19that tool is hosted — notably ``private_chat`` opens its own Discord connection 

20and enforces ``DISCORD_GUILD_ID`` there. So a host serving only non-A2A tools 

21(e.g. ``terminal``/``read_file``) needs no Discord credentials. 

22 

23Run:: 

24 

25 uv run calfkit-tools 

26"""

58def _configure_tool_workspace() -> Path: 

59 """Point the vendored hermes terminal backend at the shared workspace. 

60 

61 The hermes tools start each agent session's shell in ``TERMINAL_CWD`` 

62 (falling back to the process cwd). Setting it to the calfcord workspace 

63 root gives every agent a consistent, writable base directory while the 

64 per-session ``task_id`` keying keeps one agent's shell state out of 

65 another's. The workspace-relative memory layout (``memory/<agent_id>/``, 

66 see :mod:`calfcord.agents.memory`) resolves against this root. 

67 

68 An operator-set ``TERMINAL_CWD`` wins and is left untouched. Otherwise 

69 the root comes from ``CALFCORD_WORKSPACE_DIR`` (default 

70 ``<cwd>/state/workspace``) and is created on demand so a fresh checkout 

71 doesn't error before any tool has written to it. 

72 

73 Returns: 

74 The resolved workspace root (the value ``TERMINAL_CWD`` now carries). 

75 """ 

76 explicit = os.environ.get(_TERMINAL_CWD_ENV, "").strip() 

77 if explicit: 

78 return Path(explicit) 

79 

80 raw = os.environ.get(_WORKSPACE_ENV) 

81 root = ( 

82 Path(raw).expanduser().resolve() 

83 if raw 

84 else (Path.cwd() / _DEFAULT_WORKSPACE).resolve() 

85 ) 

86 root.mkdir(parents=True, exist_ok=True) 

87 os.environ[_TERMINAL_CWD_ENV] = str(root) 

88 logger.info( 

89 "tools workspace root=%s (TERMINAL_CWD) from_env=%s", 

90 root, 

91 _WORKSPACE_ENV in os.environ, 

92 ) 

93 return root

111def _resolve_tool_nodes(registry: dict[str, Any]) -> list[Any]: 

112 """Validate the tool registry has at least one tool and return its values. 

113 

114 Extracted from ``_amain`` so the empty-registry guard can be tested 

115 without standing up Discord/Kafka. The guard prevents the worker from 

116 starting in an inert state where it subscribes to no topics — a 

117 failure mode that would be very confusing in production logs. 

118 

119 Empty-registry is most commonly caused by a typo in 

120 ``CALFCORD_TOOLS_INCLUDE`` (per-host tool narrowing), so the SystemExit 

121 message includes the env var value to short-circuit the operator's 

122 "why is my registry empty" hunt. A complementary WARNING fires at 

123 composition time naming the specific typo'd entries (see 

124 :func:`calfcord.tools.deploy_filters.apply_deploy_filters`). 

125 """ 

126 nodes = list(registry.values()) 

127 if not nodes: 

128 include_filter = os.environ.get("CALFCORD_TOOLS_INCLUDE") or "<unset>" 

129 raise SystemExit( 

130 "TOOL_REGISTRY is empty; nothing to host " 

131 f"(CALFCORD_TOOLS_INCLUDE={include_filter})" 

132 ) 

133 return nodes

136async def _run_worker(worker: Worker) -> None: 

137 """Run ``worker`` until SIGINT/SIGTERM, then drain cleanly. 

138 

139 Delegates to the shared :func:`calfcord._worker_runtime.run_worker_until_signal` 

140 so the shutdown contract (signal-driven drain plus the 

141 "clean return without a signal is a crash" supervisor invariant) is 

142 defined in exactly one place across runners. Kept as a thin local 

143 wrapper because existing tests reference ``_run_worker`` by name. 

144 """ 

145 await run_worker_until_signal(worker, drain_label="tools worker")

148async def _amain() -> None: 

149 server_urls = os.getenv("CALF_HOST_URL") or "localhost" 

150 

151 async with Client.connect( 

152 server_urls, reply_topic=_REPLY_TOPIC, provisioning=PROVISIONING 

153 ) as client: 

154 # No manual provisioning: this runner uses the managed Worker (started 

155 # via _run_worker below), whose _on_startup hook + the connect-time 

156 # pre-start hook auto-provision the worker's tool-node topics AND the 

157 # client reply topic at broker start. Tools only ``execute`` while 

158 # consuming a message — which can only happen after ``Worker.start()`` has 

159 # started the broker — so no eager start is needed for the dispatcher. 

160 tool_nodes = _resolve_tool_nodes(TOOL_REGISTRY) 

161 

162 worker = Worker(client, tool_nodes) 

163 # Expose the process-wide client as a worker-scoped resource so A2A tool 

164 # bodies (private_chat) reach it via ``ctx.resources`` — calfkit merges 

165 # worker resources under each node's own. The client's lifecycle is owned 

166 # by the ``async with`` above; this only publishes the live reference. 

167 # Any Discord connection a tool needs is built by that tool's own 

168 # node-scoped ``@resource`` bracket, so nothing Discord-related is 

169 # constructed here. 

170 worker.resources[_A2A_CLIENT_RESOURCE] = client 

171 logger.info( 

172 "starting calfkit-tools worker tools=%s broker=%s reply_topic=%s include_filter=%s", 

173 sorted(TOOL_REGISTRY), 

174 server_urls, 

175 _REPLY_TOPIC, 

176 os.environ.get("CALFCORD_TOOLS_INCLUDE") or "<unset>", 

177 ) 

178 await _run_worker(worker)
