1"""Readiness logic for the ``calfcord _healthcheck <component>`` exec probe. 

2 

3Process Compose runs this probe ON THE AGENT/TOOLS HOSTS to gate readiness 

4(design §12.1 / §13.2), so this module must stay **light**: a readiness probe 

5carries no secrets and no heavy broker deps at import time. The 

6broker-reachability default lazy-imports its admin client *inside* the function 

7that needs it, keeping ``import calfcord.health.check`` pure-filesystem. 

8 

9Only **two** components carry a readiness signal, so only two are probeable 

10(§12.1): 

11 

12* ``"broker"`` → reachability is **metadata/admin** reachability, not bare TCP: 

13 Tansu is no-auto-create, so a bound port does not mean the broker can serve. The 

14 probe is an injected async callable (a stub in unit tests, a real metadata fetch 

15 in production) returning ``True`` when the broker answers. 

16* ``"bridge"`` → a **fresh heartbeat** must exist. The bridge refreshes its beat 

17 every few seconds and writes the first one only once Discord is connected (see 

18 ``bridge/gateway.py``'s ``_on_ready``), so a fresh bridge beat means 

19 "Discord-connected"; a stale or missing beat means "not ready". 

20 

21The roster runners (agents, ``tools``) deliberately have NO 

22readiness signal: they run via ``run_worker_until_signal`` and never beat, and 

23Process Compose attaches a readiness probe only to the broker and the bridge (see 

24``supervisor/compose.py``). Probing any other component is therefore a 

25programming/config bug, so it RAISES rather than fabricating a "not ready" verdict 

26for a signal that does not exist (a fabricated verdict would lie about readiness). 

27 

28The two probeable paths return a POSIX exit code — ``0`` healthy, ``1`` not — so 

29the caller can ``sys.exit`` it directly. 

30""" 

31 

32from __future__ import annotations 

33 

34import contextlib 

35import os 

36from collections.abc import Awaitable, Callable 

37from datetime import datetime 

38from typing import Any 

39 

40from calfcord.health.heartbeat import is_fresh, read_beat 

41 

42_DEFAULT_TTL_SECONDS = 10 

43 

44# A metadata round-trip on a healthy local broker returns in well under a second; 

45# a generous-but-bounded cap turns a hung/dead broker into a "not ready" verdict 

46# instead of a probe that never exits (Process Compose would treat that as a 

47# failure anyway, but a clean bool keeps the exit code well-defined). 

48_METADATA_TIMEOUT_MS = 5000 

49 

50BrokerProbe = Callable[[], Awaitable[bool]] 

51AdminFactory = Callable[..., Any] 

52 

53 

54async def healthcheck( 

55 home: str | os.PathLike[str], 

56 component: str, 

57 *, 

58 now: datetime, 

59 ttl_seconds: int = _DEFAULT_TTL_SECONDS, 

60 broker_probe: BrokerProbe, 

61) -> int: 

62 """Return ``0`` if ``component`` is healthy, ``1`` if not (a POSIX exit code). 

63 

64 Only ``"broker"`` and ``"bridge"`` are probeable — they are the sole components 

65 that carry a readiness signal (see the module docstring). For ``"broker"`` the 

66 verdict is metadata reachability via the injected ``broker_probe`` (why not bare 

67 TCP: module docstring). For ``"bridge"`` the verdict is a fresh heartbeat under 

68 ``<home>/state/health/`` — ``now`` and ``ttl_seconds`` are injected so freshness 

69 is deterministic and pinned to the bridge's refresh interval (§12.1). 

70 

71 Any other component raises :class:`RuntimeError`: the roster runners never beat, 

72 so there is nothing to read, and a readiness probe must not invent a verdict for 

73 a signal that does not exist (it must not lie). The raise names the offending 

74 component so the misconfiguration is obvious in the exec probe's output. 

75 """ 

76 if component == "broker": 

77 return 0 if await broker_probe() else 1 

78 

79 if component == "bridge": 

80 beat = read_beat(home, component) 

81 if beat is not None and is_fresh(beat, now=now, ttl_seconds=ttl_seconds): 

82 return 0 

83 return 1 

84 

85 raise RuntimeError( 

86 f"no readiness signal for component {component!r}: only 'broker' and 'bridge' " 

87 "are probeable (the roster runners run via run_worker_until_signal and never " 

88 "write a heartbeat)" 

89 ) 

90

92def default_broker_probe( 

93 server_urls: str, *, admin_factory: AdminFactory | None = None 

94) -> BrokerProbe: 

95 """Build the production broker-reachability probe for ``server_urls``. 

96 

97 The returned coroutine connects an admin client, fetches cluster metadata 

98 (``list_topics``), and reports whether the broker actually *served* it — this 

99 is the §12.1 contract: on Tansu (no-auto-create) a bound port does not mean 

100 the broker can answer, so we probe metadata, not bare TCP. Any failure 

101 (connect refused, metadata timeout, auth error) returns ``False`` rather than 

102 raising, because a readiness probe must report "not ready", never crash. 

103 

104 ``aiokafka``'s admin client is **lazy-imported inside the coroutine** (via the 

105 default factory) so importing this module stays pure-filesystem and free of 

106 heavy broker deps — it runs as a Process Compose exec probe on the agent/tools 

107 hosts (see the module docstring). ``admin_factory`` is an injection seam for 

108 tests; production passes ``None`` and gets the real client. 

109 """ 

110 

111 def _make_admin(**kwargs: Any) -> Any: 

112 # Lazy import: the admin client (and aiokafka) must not load at module 

113 # import time — only when a broker probe actually runs. 

114 from aiokafka.admin import AIOKafkaAdminClient 

115 

116 return AIOKafkaAdminClient(**kwargs) 

117 

118 factory = admin_factory if admin_factory is not None else _make_admin 

119 

120 async def _probe() -> bool: 

121 admin = None 

122 try: 

123 # Construct INSIDE the try so even a constructor failure (e.g. a 

124 # malformed bootstrap string) degrades to "not ready" per the 

125 # never-raises contract, rather than escaping the exec probe. 

126 admin = factory( 

127 bootstrap_servers=server_urls, 

128 request_timeout_ms=_METADATA_TIMEOUT_MS, 

129 ) 

130 await admin.start() 

131 await admin.list_topics() 

132 return True 

133 except Exception: 

134 # Any connection / metadata failure means "not ready"; the probe never 

135 # raises so the exec probe gets a clean bool (→ exit code). 

136 return False 

137 finally: 

138 # Always release the admin connection (when one was built), even when 

139 # start/metadata raised; a leaked socket would outlive the probe. 

140 if admin is not None: 

141 with contextlib.suppress(Exception): 

142 await admin.close() 

143 

144 return _probe
