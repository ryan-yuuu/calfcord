1"""Pure projection helpers between AgentDefinition and AgentStateEvent. 

2 

3Used by the agent process when announcing (definition -> event) and by the bridge 

4when projecting received events into its registry (event -> definition). 

5 

6The event form excludes agent-internal fields (system_prompt, tools, model, 

7publish_topic, source_path). The bridge stubs them with sentinel values when 

8rebuilding an AgentDefinition -- those fields are never read on the bridge side 

9but the AgentDefinition validator still requires legal values. 

10""" 

11from __future__ import annotations 

12 

13from datetime import UTC, datetime 

14 

15from calfcord.agents.definition import AgentDefinition 

16from calfcord.control_plane.schema import AgentStateEvent, StateEventCause 

17 

18 

19def build_state_event( 

20 definition: AgentDefinition, cause: StateEventCause 

21) -> AgentStateEvent: 

22 """Project an AgentDefinition into a wire-friendly AgentStateEvent. 

23 

24 Agent-internal fields (system_prompt, tools, model, publish_topic, source_path) 

25 are intentionally excluded -- the bridge never reads them. 

26 """ 

27 return AgentStateEvent( 

28 agent_id=definition.agent_id, 

29 display_name=definition.display_name, 

30 description=definition.description, 

31 avatar_url=definition.avatar_url, 

32 role=definition.role, 

33 history_turns=definition.history_turns, 

34 thinking_effort=definition.thinking_effort, 

35 provider=definition.provider, 

36 memory=definition.memory, 

37 emitted_at=datetime.now(UTC), 

38 cause=cause, 

39 ) 

40 

41 

42# Stubs used when the bridge rebuilds an AgentDefinition from a state event. 

43# These fields are never read on the bridge side; their values exist only to 

44# satisfy AgentDefinition's pydantic validators. 

45_BRIDGE_STUB_SYSTEM_PROMPT = "(agent-internal; bridge projection from state event)" 

46 

47 

48def state_event_to_definition(event: AgentStateEvent) -> AgentDefinition: 

49 """Reconstitute an AgentDefinition from a state event for bridge-side use. 

50 

51 Stubs agent-internal fields. The reconstituted definition is valid (passes 

52 pydantic validation) and answers all the bridge-side accessors the registry 

53 and downstream code use. Do not call this in agent-side code paths -- agents 

54 have their own real AgentDefinition from disk. 

55 """ 

56 return AgentDefinition( 

57 # AgentDefinition uses ``alias="name"`` for agent_id; populate_by_name=True 

58 # in its model_config means we can pass either; use the field name here. 

59 agent_id=event.agent_id, 

60 display_name=event.display_name, 

61 description=event.description, 

62 avatar_url=event.avatar_url, 

63 provider=event.provider, 

64 model=None, 

65 tools=(), 

66 thinking_effort=event.thinking_effort, 

67 role=event.role, 

68 publish_topic=None, # state events come from assistants only; bridge 

69 # doesn't receive router announcements 

70 history_turns=event.history_turns, 

71 memory=event.memory, # the bridge reads this to decide whether to ship the 

72 # memory-prompt template in deps. Safe to set even though 

73 # tools is stubbed to (): the factory's memory-needs-fs-tools 

74 # guard runs only in build_node (agent-runner side), and the 

75 # bridge never builds nodes from these reconstituted defs. 

76 system_prompt=_BRIDGE_STUB_SYSTEM_PROMPT, 

77 source_path=None, 

78 )
