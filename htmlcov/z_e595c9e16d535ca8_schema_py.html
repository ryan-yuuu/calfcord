1"""Pydantic protocol types for the bridge<->agent control plane. 

2 

3Two flows: 

4- Bridge -> agent: AgentControlEnvelope wraps an AgentControlCommand (discriminated 

5 union of operations). Targeted commands ride agent.<id>.control.in topics; the 

6 DiscoveryPingOp rides the broadcast bridge.discovery topic. 

7- Agent -> bridge: AgentStateMessage (discriminated union of AgentStateEvent and 

8 AgentDepartureEvent) rides the shared agent.state topic. 

9 

10All payloads carry a schema_version field. The bridge and agent each log + skip 

11messages whose schema_version doesn't match CONTROL_PLANE_SCHEMA_VERSION. 

12""" 

13from __future__ import annotations 

14 

15from datetime import datetime 

16from typing import Annotated, Literal 

17 

18from pydantic import BaseModel, ConfigDict, Field 

19 

20from calfcord.agents.definition import AgentRole, Provider, ThinkingEffort 

21 

22CONTROL_PLANE_SCHEMA_VERSION = 2 

23 

24 

25# --- Bridge -> agent commands --- 

26 

27class SetThinkingEffortOp(BaseModel): 

28 op: Literal["set_thinking_effort"] = "set_thinking_effort" 

29 agent_id: str # defensive cross-check vs publish topic 

30 value: ThinkingEffort 

31 request_id: str # UUIDv4, logs only -- no ack channel in v1 

32 issued_by: str # discord user id, audit-only 

33 

34 

35class DiscoveryPingOp(BaseModel): 

36 op: Literal["discovery_ping"] = "discovery_ping" 

37 issued_at: datetime 

38 request_id: str 

39 

40 

41AgentControlCommand = Annotated[ 

42 SetThinkingEffortOp | DiscoveryPingOp, 

43 Field(discriminator="op"), 

44] 

45 

46 

47class AgentControlEnvelope(BaseModel): 

48 schema_version: int = CONTROL_PLANE_SCHEMA_VERSION 

49 command: AgentControlCommand 

50 

51 

52# --- Agent -> bridge state messages --- 

53 

54StateEventCause = Literal["startup", "command_applied", "discovery_response"] 

55 

56 

57class AgentStateEvent(BaseModel): 

58 # Explicit ``extra="ignore"`` (pydantic's default, made load-bearing on 

59 # purpose): a NEW agent may add wire fields (as ``memory`` was added) without 

60 # bumping CONTROL_PLANE_SCHEMA_VERSION, and an OLD bridge that predates the 

61 # field must silently ignore it rather than reject the event. Do NOT change 

62 # this to ``extra="forbid"`` for "consistency" with AgentDefinition — that 

63 # would break backward compatibility for mixed-version rolling deploys. 

64 model_config = ConfigDict(extra="ignore") 

65 

66 kind: Literal["state"] = "state" 

67 schema_version: int = CONTROL_PLANE_SCHEMA_VERSION 

68 agent_id: str 

69 display_name: str 

70 description: str 

71 avatar_url: str | None = None 

72 role: AgentRole 

73 history_turns: int 

74 thinking_effort: ThinkingEffort | None = None 

75 provider: Provider | None = None # nullable: required at the AgentDefinition 

76 # level only for non-bridge use; bridge needs 

77 # it for tier-3 model_settings resolution 

78 memory: bool = False # whether the agent opted into persistent 

79 # memory; the bridge gates shipping the 

80 # memory-prompt template in deps on this 

81 # (ingress._memory_prompt_deps). Defaulted 

82 # so events from an older agent that predates 

83 # this field deserialize as memory-off rather 

84 # than failing the schema_version gate. 

85 emitted_at: datetime 

86 cause: StateEventCause 

87 

88 

89class AgentDepartureEvent(BaseModel): 

90 kind: Literal["departure"] = "departure" 

91 schema_version: int = CONTROL_PLANE_SCHEMA_VERSION 

92 agent_id: str 

93 departed_at: datetime 

94 reason: Literal["shutdown"] = "shutdown" # leave room for "heartbeat_loss" later 

95 

96 

97AgentStateMessage = Annotated[ 

98 AgentStateEvent | AgentDepartureEvent, 

99 Field(discriminator="kind"), 

100]
