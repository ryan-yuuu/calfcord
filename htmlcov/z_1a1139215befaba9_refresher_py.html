1"""Health-gated heartbeat refresher for the long-lived runners (design §12.1). 

2 

3A runner must refresh its heartbeat every few seconds, but **only while it is 

4healthy**. The point of §12.1 is that a beat stamped on a bare timer lies: it 

5stays green while a revoked token or a dropped Discord gateway leaves the bot 

6silent. The fix is to gate every refresh on a liveness predicate and *skip the 

7write* when it is false — so ``last_beat`` stops advancing, the beat goes stale 

8within the TTL, and the readiness probe correctly reads "not ready". The 

9refresher never *demotes* the beat (writes a "down" status); it simply stops 

10feeding it, which is what freshness is built to detect. 

11 

12Both the predicate (``is_healthy``) and the display ``identity`` are injected as 

13callables so the bridge can pass a live connection-state check and a bot-identity 

14getter that only resolves once the gateway is ready. The clock and ``sleep`` are 

15injected too, so the loop is fully unit-testable with no wall-clock waits. 

16 

17Kept import-light on purpose (heartbeat + stdlib + asyncio only): this runs inside 

18every runner, including the agent/tools hosts (see the package docstring). 

19""" 

20 

21from __future__ import annotations 

22 

23import asyncio 

24import logging 

25import os 

26from collections.abc import Awaitable, Callable 

27from datetime import UTC, datetime 

28 

29from calfcord.health.heartbeat import write_beat 

30 

31logger = logging.getLogger(__name__) 

32 

33# Pinned relative to the TTL in heartbeat.py (10s): ~2s gives a few beats of slack 

34# before the TTL trips, so a single slow tick does not flap the readiness probe. 

35_DEFAULT_INTERVAL_SECONDS = 2.0 

36 

37IsHealthy = Callable[[], bool] 

38Identity = Callable[[], str | None] 

39Clock = Callable[[], datetime] 

40Sleep = Callable[[float], Awaitable[None]] 

41 

42 

43def refresh_once( 

44 home: str | os.PathLike[str], 

45 component: str, 

46 *, 

47 is_healthy: IsHealthy, 

48 identity: Identity, 

49 status: str = "healthy", 

50 now: datetime, 

51) -> bool: 

52 """Write one heartbeat IFF ``is_healthy()`` is true; return whether it wrote. 

53 

54 When healthy, persist a beat via :func:`calfcord.health.heartbeat.write_beat` 

55 (which advances ``last_beat`` to ``now`` while preserving ``started_at``). 

56 When unhealthy, **skip the write entirely** so ``last_beat`` stays pinned to 

57 the last healthy tick and the beat ages past the TTL — the §12.1 contract that 

58 turns a silent failure into a "not ready" verdict at the probe. 

59 

60 ``is_healthy`` and ``identity`` are evaluated *here, per tick*, so the beat 

61 always reflects the connection state and bot identity at the moment it is 

62 written. ``now`` must be timezone-aware UTC (enforced by ``write_beat``). 

63 """ 

64 if not is_healthy(): 

65 return False 

66 

67 write_beat(home, component, status=status, identity=identity(), now=now) 

68 return True 

69

71async def run_refresher( 

72 home: str | os.PathLike[str], 

73 component: str, 

74 *, 

75 is_healthy: IsHealthy, 

76 identity: Identity, 

77 status: str = "healthy", 

78 interval_seconds: float = _DEFAULT_INTERVAL_SECONDS, 

79 clock: Clock = lambda: datetime.now(UTC), 

80 sleep: Sleep = asyncio.sleep, 

81) -> None: 

82 """Refresh ``component``'s heartbeat on a timer until cancelled. 

83 

84 Each iteration ticks (:func:`refresh_once`, gated on ``is_healthy``) then 

85 awaits ``sleep(interval_seconds)``, repeating forever. The runner cancels the 

86 task on shutdown: :class:`asyncio.CancelledError` returns cleanly here (rather 

87 than propagating a spurious task error) because a cancelled refresher is an 

88 expected, orderly stop, not a fault. 

89 

90 ``clock`` defaults to an aware-UTC ``datetime.now(UTC)`` (matching the probe's 

91 clock so freshness never trips a naive/aware mismatch); ``sleep`` defaults to 

92 :func:`asyncio.sleep`. Both are injected so tests drive the loop deterministically 

93 with zero wall-clock waits. 

94 """ 

95 try: 

96 while True: 

97 try: 

98 refresh_once( 

99 home, 

100 component, 

101 is_healthy=is_healthy, 

102 identity=identity, 

103 status=status, 

104 now=clock(), 

105 ) 

106 except Exception: 

107 # A transient write failure (disk full / read-only volume) must not 

108 # kill the refresher task — that would freeze the beat forever and 

109 # leave the runner permanently "not ready" even after the fault 

110 # clears. Log and keep ticking; the missed beat ages toward the TTL, 

111 # which is the correct "not ready" signal until a write succeeds. 

112 # (CancelledError is a BaseException, so it is NOT caught here.) 

113 logger.warning( 

114 "heartbeat refresh failed for %s; retrying next tick", 

115 component, 

116 exc_info=True, 

117 ) 

118 await sleep(interval_seconds) 

119 except asyncio.CancelledError: 

120 # Orderly shutdown: a cancelled refresher is the normal stop path, so 

121 # swallow the cancellation and return instead of surfacing a task error. 

122 return
