1"""Wire schema for Discord events flowing through Kafka. 

2 

3These pydantic models are the contract between the bridge (producer) and any 

4calfkit agent (consumer). The bridge serializes a :class:`WireMessage` to JSON 

5and tucks it into ``envelope.context.deps["discord"]`` on each 

6publish. Agents read it back via the same path inside their gates and ``run()``. 

7 

8Both models are frozen — once normalized, a wire message is immutable. 

9 

10``schema_version`` policy: 

11 - Add-only fields are non-breaking. Do not bump the version. 

12 - Field renames or removals require bumping ``schema_version`` AND a 

13 CHANGELOG entry. Consumers must tolerate the bump. 

14 

15A2A invocations: 

16 Agent-to-agent messaging no longer touches this schema. The old 

17 first-party ``private_chat`` tool — which reused this wire by forwarding 

18 the caller's originating message with ``slash_target`` / ``kind="slash"`` 

19 overridden to satisfy a now-deleted ``addressed_to_me`` gate, plus a 

20 companion ``caller_agent_id`` deps key — was removed in the calfkit 0.12 

21 migration. Peer consults and handoffs now run through calfkit's native 

22 A2A (``message_agent`` / ``Handoff``), which carries its own envelope. A 

23 :class:`WireMessage` here is only ever the bridge's typed projection of a 

24 *human* Discord event, handed to the agent via ``deps["discord"]``. 

25""" 

26

35class WireAuthor(BaseModel): 

36 """Resolved author identity for a Discord message. 

37 

38 ``agent_id`` is now unconditionally ``None`` — the display-name→agent 

39 registry that once populated it was removed in the calfkit 0.12 migration. 

40 A persona webhook's own turns are recognized downstream by their bot-owned 

41 ``webhook_id`` (R-A3), not by resolving a name here. ``is_human_owner`` is 

42 set when the author is the configured human owner user. ``avatar_url`` is 

43 the author's effective 

44 avatar URL (per-guild member avatar, falling back to user avatar, falling 

45 back to Discord's default) — used by downstream consumers that want to 

46 render the author visually (e.g. reply-embed icons). 

47 """

49 model_config = ConfigDict(frozen=True)

54 is_webhook: bool

61class WireMessage(BaseModel): 

62 """A Discord event (regular message or slash invocation) projected onto Kafka. 

63 

64 For ``kind="message"``, ``message_id`` is the user's Discord message ID. 

65 For ``kind="slash"``, ``message_id`` is the bridge's followup message ID 

66 (the visible echo posted via ``interaction.followup.send``). In both cases 

67 this is the ID an agent should pass to ``DiscordPersonaSender.send``'s 

68 ``reply_to_message_id`` parameter to render its reply as an inline reply 

69 to the source. 

70 """ 

71 

72 model_config = ConfigDict(frozen=True) 

73 

74 schema_version: int = 1 

75 event_id: str 

76 kind: Literal["message", "slash"] 

77 slash_target: str | None = None 

78 message_id: int 

79 channel_id: int 

80 source_channel_id: int | None = None 

81 """The actual Discord channel id the message landed in (thread or 

82 top-level). ``channel_id`` is the parent-channel id used for Kafka 

83 topic routing (the normalizer flattens threads to parent so all 

84 messages in a thread group share one topic); ``source_channel_id`` 

85 preserves the un-flattened id so the bridge's 

86 :class:`~calfcord.bridge.history.ChannelHistoryFetcher` 

87 fetches the right channel's history (the thread itself, not the 

88 parent). ``None`` means the wire predates this field; callers fall 

89 back to ``channel_id`` (correct for non-thread messages; a thread 

90 wire from before the deploy loses one cycle of accurate history).""" 

91 guild_id: int 

92 content: str 

93 author: WireAuthor 

94 created_at: datetime

96 @model_validator(mode="after") 

97 def _check_slash_target(self) -> WireMessage: 

98 if self.kind == "slash" and self.slash_target is None: 

99 raise ValueError("slash_target is required when kind='slash'") 

100 if self.kind == "message" and self.slash_target is not None: 

101 raise ValueError("slash_target must be None when kind='message'") 

102 return self 

103 

104 @property 

105 def thread_id(self) -> int | None: 

106 """The thread this event originated in, or ``None`` for a top-level message. 

107 

108 ``channel_id`` is the flattened parent channel (the persona webhook's 

109 host and the Kafka topic key); ``source_channel_id`` is the 

110 un-flattened origin. They differ exactly when the event came from a 

111 Discord thread — which is precisely when an agent reply or live-step 

112 progress message must be posted *into* the thread rather than the 

113 parent. Callers pass this as the ``thread_id`` argument to 

114 :class:`~calfcord.discord.persona.DiscordPersonaSender`. 

115 """ 

116 if self.source_channel_id is not None and self.source_channel_id != self.channel_id: 

117 return self.source_channel_id 

118 return None
