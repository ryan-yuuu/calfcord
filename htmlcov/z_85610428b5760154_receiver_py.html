1"""Gateway-based Discord receiver. 

2 

3Holds a long-lived WebSocket connection to Discord, normalizes incoming 

4``discord.Message`` events into ``IncomingMessage``, and fans them out 

5to registered handlers. 

6 

7Handlers are dispatched sequentially per message; one slow handler will 

8delay later handlers for the same message but not for subsequent 

9messages. Exceptions raised inside a handler are logged and isolated so 

10they don't poison sibling handlers. 

11""" 

12 

13from __future__ import annotations 

14 

15import logging 

16from collections.abc import Awaitable, Callable 

17 

18import discord 

19 

20from calfcord.discord.messages import IncomingMessage 

21from calfcord.discord.settings import DiscordSettings 

22 

23logger = logging.getLogger(__name__) 

24 

25type MessageHandler = Callable[[IncomingMessage], Awaitable[None]] 

26 

27 

28class DiscordReceiver: 

29 """Long-lived gateway consumer for Discord messages. 

30 

31 Example:: 

32 

33 receiver = DiscordReceiver(settings) 

34 

35 @receiver.on_message 

36 async def handle(msg: IncomingMessage) -> None: 

37 print(msg.author_name, msg.content) 

38 

39 await receiver.start() # blocks until cancelled 

40 """ 

41 

42 def __init__(self, settings: DiscordSettings) -> None: 

43 self._settings = settings 

44 self._handlers: list[MessageHandler] = [] 

45 self._client = _DispatchingClient(receiver=self, intents=self._build_intents()) 

46 

47 @staticmethod 

48 def _build_intents() -> discord.Intents: 

49 # message_content + members are privileged and must be enabled 

50 # in the Developer Portal under Bot → Privileged Gateway Intents. 

51 intents = discord.Intents.default() 

52 intents.message_content = True 

53 intents.members = True 

54 return intents 

55 

56 def on_message(self, handler: MessageHandler) -> MessageHandler: 

57 """Register an async handler for every received message. 

58 

59 Returns the handler unchanged so the call doubles as a decorator. 

60 """ 

61 self._handlers.append(handler) 

62 return handler 

63 

64 async def _dispatch(self, message: discord.Message) -> None: 

65 bot_user = self._client.user 

66 is_from_self = bot_user is not None and message.author.id == bot_user.id 

67 

68 incoming = IncomingMessage( 

69 id=message.id, 

70 channel_id=message.channel.id, 

71 guild_id=message.guild.id if message.guild is not None else None, 

72 author_id=message.author.id, 

73 author_name=message.author.name, 

74 content=message.content, 

75 created_at=message.created_at, 

76 is_from_self=is_from_self, 

77 is_bot=message.author.bot, 

78 ) 

79 

80 for handler in self._handlers: 

81 try: 

82 await handler(incoming) 

83 except Exception: 

84 logger.exception( 

85 "handler %s raised on message id=%s", 

86 getattr(handler, "__name__", repr(handler)), 

87 incoming.id, 

88 ) 

89 

90 async def start(self) -> None: 

91 """Connect to the gateway and run until cancelled. Blocking.""" 

92 token = self._settings.bot_token.get_secret_value() 

93 logger.info("DiscordReceiver connecting to gateway") 

94 await self._client.start(token) 

95 

96 async def close(self) -> None: 

97 """Disconnect from the gateway gracefully. Idempotent.""" 

98 if self._client.is_closed(): 

99 return 

100 await self._client.close() 

101 

102 

103class _DispatchingClient(discord.Client): 

104 """Discord client that forwards message events to a ``DiscordReceiver``.""" 

105 

106 def __init__(self, receiver: DiscordReceiver, *, intents: discord.Intents) -> None: 

107 super().__init__(intents=intents) 

108 self._receiver = receiver 

109 

110 async def on_ready(self) -> None: 

111 user_id = self.user.id if self.user is not None else None 

112 logger.info("DiscordReceiver ready as %s (id=%s)", self.user, user_id) 

113 

114 async def on_message(self, message: discord.Message) -> None: 

115 await self._receiver._dispatch(message)
