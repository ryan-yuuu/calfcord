1"""Discord ingress gateway daemon and CLI entry point. 

2 

3Holds the long-lived gateway WebSocket and wires the per-``@mention`` orchestration 

4together on the calfkit 0.12 **caller surface**. Run via:: 

5 

6 uv run calfkit-bridge 

7 

8The daemon depends on a running Kafka broker reachable at ``CALF_HOST_URL`` 

9(defaults to ``localhost``) and a Discord bot configured via the ``DISCORD_*`` 

10environment variables (see ``.env.example``). 

11 

12The bridge is a pure calfkit :class:`~calfkit.client.Client` (no embedded Worker, 

13no consumers). For each ``@mention`` it builds a :class:`MentionRequest` and runs 

14:class:`~calfcord.bridge.mention_handler.MentionHandler.handle` as a tracked task: 

15the handler resolves the target against the live mesh roster, ``start()``s the 

16agent, drains its run ``stream()`` (live progress + A2A projection), and posts the 

17terminal reply under the responding agent's persona. Non-``@mention`` ("ambient") 

18messages go unanswered (C2). The bridge owns SIGINT/SIGTERM for its foreground 

19(the Discord gateway) and tears down by cancelling in-flight handler tasks then 

20closing the client. 

21"""

144async def _prune_on_startup(store: TranscriptStoreLike, settings: DiscordSettings) -> None: 

145 """Best-effort startup sweep: drop transcript rows past the retention window. 

146 

147 The bridge is the sole writer and restarts on every deploy, so a startup 

148 prune bounds the store's growth without a background task. Disabled when 

149 ``transcript_retention_days <= 0`` (keep forever). Best-effort: a prune 

150 failure must NEVER abort startup — any exception is logged and swallowed. 

151 """ 

152 if settings.transcript_retention_days <= 0: 

153 return 

154 try: 

155 cutoff = int(time.time()) - settings.transcript_retention_days * 86400 

156 pruned = await store.prune_older_than(cutoff) 

157 if pruned: 

158 logger.info("pruned %d transcript row(s) older than %d days", pruned, settings.transcript_retention_days) 

159 except Exception: 

160 logger.exception("transcript retention prune failed at startup; continuing")

342 def _spawn_handle(self, req: MentionRequest) -> None: 

343 """Run one ``@mention`` as a tracked background task. 

344 

345 Each mention is independent and may be long-running (an agent run + A2A 

346 consults), so it must not block the Discord event loop. The task is tracked 

347 in :attr:`_inflight` so shutdown can cancel it, and its result is reaped by 

348 :meth:`_on_handle_done` (which surfaces an unexpected crash to the user). 

349 """ 

350 task = asyncio.create_task(self._run_handler(req)) 

351 self._inflight.add(task) 

352 task.add_done_callback(self._inflight.discard)

356

423 async def _run() -> None: 

424 # The bridge owns its persona sender (it posts on behalf of every agent). 

425 # The calfkit Client is a pure caller surface: a durable inbox + the mesh 

426 # view; no Worker, no consumers. Nested ``async with`` (not combined) keeps 

427 # each context's rationale attached to it. 

428 async with DiscordPersonaSender(settings) as persona_sender: # noqa: SIM117 

429 async with Client.connect( 

430 server_urls, 

431 inbox_topic=_BRIDGE_INBOX_TOPIC, 

432 provisioning=PROVISIONING, 

433 mesh_config=MeshViewConfig(stale_after=_MESH_STALE_AFTER_SECONDS), 

434 ) as calfkit_client: 

435 # No eager broker pre-start here (D-11 revisited): the first 

436 # ``client.agent(name).start(...)`` self-ensures the broker BEFORE it 

437 # publishes — calfkit's ``AgentGateway.start`` awaits 

438 # ``_ensure_started`` (→ ``broker.start()``) ahead of ``_publish_call``, 

439 # and with ``provisioning=PROVISIONING`` that provisions the durable 

440 # inbox and starts its groupless reply subscriber consuming before the 

441 # request is sent, so a reply can't land on an unprovisioned/unconsumed 

442 # inbox (calfkit's ``_ensure_started`` docstring calls out exactly the 

443 # provisioning-enabled case). Nothing between here and the first mention 

444 # publishes to the broker, and the mesh roster read opens its own 

445 # independent reader — so there is nothing left to pre-start. (The CLI 

446 # probes DO keep an ``events()`` pre-start, but for a different reason: 

447 # there it doubles as the broker-reachability check.) 

448 async with _open_transcript_store(settings) as transcript_store: 

449 await _prune_on_startup(transcript_store, settings) 

450 

451 typing_notifier = TypingNotifier(persona_sender.client) 

452 overrides = EffortOverrides(transcript_store) 

453 await overrides.hydrate() # restore /thinking-effort overrides across restarts (D-8) 

454 # A2A audit projection: the resolver only uses the sender's 

455 # ``.client`` (a REST login), which the persona sender provides. 

456 resolver = A2AChannelResolver( 

457 persona_sender, 

458 settings.guild_id, 

459 channel_name=_a2a_channel_name(), 

460 category_name=_a2a_category_name(), 

461 ) 

462 gateway = DiscordIngressGateway( 

463 settings, 

464 calfkit_client=calfkit_client, 

465 persona_sender=persona_sender, 

466 transcript_store=transcript_store, 

467 roster=MeshRoster(calfkit_client), 

468 overrides=overrides, 

469 a2a=A2AProjector(resolver, persona_sender), 

470 progress=ProgressRenderer(persona_sender, typing_notifier), 

471 reply=ReplyPoster(persona_sender, transcript_store), 

472 memory_deps=MemoryPromptDeps(), 

473 ) 

474 try: 

475 stop = asyncio.Event() 

476 loop = asyncio.get_running_loop() 

477 for sig in (signal.SIGINT, signal.SIGTERM): 

478 loop.add_signal_handler(sig, stop.set) 

479 

480 gateway_task = asyncio.create_task(gateway.start()) 

481 stop_task = asyncio.create_task(stop.wait()) 

482 # Refresh the bridge heartbeat on a timer, gated on the live 

483 # Discord connection so a dropped gateway ages the beat (§12.1). 

484 refresher_task = asyncio.create_task( 

485 run_refresher( 

486 _resolve_health_home(), 

487 _HEALTH_COMPONENT, 

488 is_healthy=lambda: gateway.connected, 

489 identity=lambda: gateway.bot_identity, 

490 ) 

491 ) 

492 try: 

493 done, _ = await asyncio.wait( 

494 {gateway_task, stop_task}, 

495 return_when=asyncio.FIRST_COMPLETED, 

496 ) 

497 # A fatal gateway crash (not a signal) must surface as a 

498 # non-zero exit so the supervisor restarts us; asyncio.wait 

499 # does not propagate a task's exception. 

500 if gateway_task in done and not gateway_task.cancelled(): 

501 exc = gateway_task.exception() 

502 if exc is not None: 

503 raise exc 

504 finally: 

505 for t in (gateway_task, stop_task, refresher_task): 

506 if not t.done(): 

507 t.cancel() 

508 await asyncio.gather(refresher_task, return_exceptions=True) 

509 await gateway.close() 

510 finally: 

511 # Cancel in-flight handler tasks BEFORE the client context 

512 # exits (which closes the broker), then close typing — a 

513 # cancelled run can still have fired a typing task. 

514 await gateway.drain_inflight() 

515 await typing_notifier.aclose()
