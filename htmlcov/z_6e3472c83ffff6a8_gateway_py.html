1"""Discord ingress gateway daemon and CLI entry point. 

2 

3Holds the long-lived gateway WebSocket, wires the slash command manager 

4and the agent ingress publisher together, and exposes ``main()`` as the 

5script entry point. Run via:: 

6 

7 uv run calfkit-bridge 

8 

9The daemon depends on a running Kafka broker reachable at ``CALF_HOST_URL`` 

10(defaults to ``localhost``) and a Discord bot configured via the 

11``DISCORD_*`` environment variables (see ``.env.example``). 

12 

13The bridge process does both halves of the Discord I/O: 

14 

15* **Ingress** — Discord events normalize into :class:`WireMessage` and 

16 fire-and-forget through :class:`BridgeIngress` onto Kafka. 

17* **Egress** — every agent reply landing on ``discord.outbox`` is posted 

18 to Discord by a long-lived calfkit consumer 

19 (:func:`build_outbox_consumer`). Replies are no longer awaited 

20 inline; this is what lets multiple agents respond to the same 

21 inbound event without losing all but the fastest reply (calfkit's 

22 reply dispatcher dedupes by ``correlation_id``, so the request/reply 

23 shape we used before was inherently single-agent). 

24 

25The consumer's handler is registered on the same calfkit 

26:class:`~calfkit.Client` connection that the ingress publishes through. 

27The bridge is calfcord's single deliberate **embedded** Worker variant: it 

28co-runs the Discord gateway (a foreground WebSocket) and owns SIGINT/SIGTERM, 

29so it drives the worker via :meth:`Worker.start` / :meth:`Worker.stop` 

30(signals opted OUT) rather than :meth:`Worker.run` (which would own the 

31foreground and install a colliding signal set) — see :func:`main`. 

32"""

104# A plaintext ``/task`` command: a message whose content is the bare token 

105# ``/task`` optionally followed by the task text. Case-insensitive; DOTALL so 

106# a multi-line task body is captured whole. Anchored at the start so it is a 

107# real command, not a ``/task`` mentioned mid-sentence; ``/taskfoo`` does NOT 

108# match (the command must be the bare token ``/task``). A bare ``/task`` — 

109# nothing after it, or only whitespace — matches with an empty body, which 

110# :func:`_parse_task_command` reports as ``None``. 

111_TASK_COMMAND_RE = re.compile(r"^/task(?:\s+(?P<body>.*))?$", re.IGNORECASE | re.DOTALL)

143def _parse_task_command(content: str) -> tuple[bool, str | None]: 

144 """Classify a message body as a plaintext ``/task`` command. 

145 

146 Returns ``(is_task, body)``: 

147 

148 - ``(False, None)`` — not a ``/task`` command; route the message normally. 

149 - ``(True, None)`` — a bare ``/task`` with no task text (reply with usage). 

150 - ``(True, "<text>")`` — a ``/task`` carrying task text (open a thread). 

151 

152 The returned ``body`` is whitespace-stripped and used only to derive the 

153 thread title; the routed wire keeps the full original message content. 

154 """ 

155 match = _TASK_COMMAND_RE.match(content) 

156 if match is None: 

157 return False, None 

158 body = match.group("body") 

159 if body is not None: 

160 body = body.strip() 

161 return True, (body or None)

195async def _prune_on_startup(store: TranscriptStoreLike, settings: DiscordSettings) -> None: 

196 """Best-effort startup sweep: drop transcript rows past the retention window. 

197 

198 The bridge is the sole writer and restarts on every deploy, so a 

199 startup prune bounds the store's growth without a background task.

201

475 async def _maybe_handle_task(self, message: discord.Message) -> bool: 

476 """Handle a plaintext ``/task`` command; return whether it was owned. 

477 

478 Returns ``True`` when the message is a ``/task`` command this method 

479 has fully handled (so the caller must stop and NOT route it normally), 

480 and ``False`` when the message is not a task and should flow through 

481 the usual ambient/slash routing. 

482 

483 ``/task`` is a plaintext command rather than a Discord slash command on 

484 purpose: a slash invocation can only post the thread's opening message 

485 as the bot or a webhook, whereas this rides the user's own message — so 

486 the task's anchor is genuinely authored by the user. Discord's built-in 

487 ``/thread`` reads as the user for the same reason (it runs as the 

488 user's own client); a bot has no API to post a message as someone else, 

489 so the user must author the anchor and we thread off it. 

490 

491 Flow once a task is recognised (every branch returns ``True``): 

492 

493 1. Only humans start tasks — an agent-persona webhook (or any bot) that 

494 posts ``/task`` falls through to normal routing instead. 

495 2. Require a top-level text channel: reject inside an existing thread 

496 (Discord can't nest threads) or a non-text channel (forum/voice — 

497 the persona webhook needs a parent text channel anyway). 

498 3. A bare ``/task`` (no task text) gets a usage hint. 

499 4. Open a public thread anchored on the user's own message (the thread 

500 shares the message id) and route an ambient, thread-scoped wire so 

501 the router summons the agents and their replies + live-step progress 

502 post into the thread. 

503 

504 Every Discord call is best-effort: a failure logs and inline-replies 

505 explaining what went wrong. 

506 """ 

507 is_task, body = _parse_task_command(message.content) 

508 if not is_task: 

509 return False 

510 # Only genuine humans spawn tasks. Persona webhooks (and any other bot) 

511 # that happen to post "/task ..." flow through to normal routing so 

512 # peers still see the message — they don't open task threads. 

513 if message.webhook_id is not None or message.author.bot: 

514 return False 

515 

516 logger.info( 

517 "task command invoked channel_id=%s user_id=%s message_id=%s", 

518 getattr(message.channel, "id", None), 

519 message.author.id, 

520 message.id, 

521 ) 

522 

523 channel = message.channel 

524 if not isinstance(channel, discord.TextChannel): 

525 await self._reply_best_effort( 

526 message, 

527 "`/task` only works in a top-level text channel — not inside a " 

528 "thread (threads can't be nested) or a forum/voice channel.", 

529 ) 

530 return True 

531 

532 if body is None: 

533 await self._reply_best_effort( 

534 message, 

535 "Usage: `/task <what you want done>` — describe the task and I'll " 

536 "open a thread for the agents to work in.", 

537 ) 

538 return True 

539 

540 if self._message_normalizer is None: 

541 # Pre-ready guard. In practice unreachable via _on_message (which 

542 # returns early when the normalizer is unset, before reaching here), 

543 # but enforced defensively because we have already claimed the 

544 # message. Make the drop loud rather than silent: log it and tell 

545 # the user to retry. No thread has been created yet, so a retry is
