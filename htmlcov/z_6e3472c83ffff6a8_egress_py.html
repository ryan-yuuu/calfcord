1"""Unified agent-to-agent audit channel resolution. 

2 

3Provides a small helper that resolves (and lazily creates) the single 

4Discord text channel where every agent-to-agent conversation lives, plus 

5a helper for anchoring a public thread on a specific message in that 

6channel. There is one such channel per guild, sourced from 

7``CALFKIT_A2A_CHANNEL_NAME`` (default ``private-a2a-chats``) — collapsing the 

8former per-pair ``a2a-{x}-{y}`` design to a flat directory of threads 

9inside one channel. 

10 

11The unified design eliminates the O(N²) channel-count growth of the 

12per-pair scheme: ten agents now require one channel plus one permission 

13overwrite, not 45 channels with 45 overwrites. Each A2A call posts its 

14caller's request as a normal message in the unified channel, then 

15anchors a thread on that message; subsequent turns in the same 

16conversation post into the thread. Humans see a clean per-channel 

17directory of conversation starters and can drill into any thread to 

18audit. 

19 

20When constructed with a ``category_name`` (sourced from 

21``CALFKIT_A2A_CHANNEL_CATEGORY`` by the bridge, whose gateway constructs the 

22resolver), the unified 

23channel is placed under that Discord category on first creation — and 

24the category itself is created lazily on the first miss. The child 

25channel inherits the category's permission overwrites, so locking down 

26audit visibility is a one-time Discord-UI operation on the category 

27rather than a per-channel chore. An existing channel with the 

28configured name is reused regardless of its current category, so 

29operator overrides and migrations are non-disruptive. 

30 

31The resolver intentionally does not validate agent identities: identity 

32validation is the caller's job. The bridge is now its only caller — it 

33validates ids against the live mesh roster before reaching here. 

34""" 

35

48 """Resolves the unified A2A audit channel and anchors A2A threads on it."""

51 self, 

52 sender: DiscordSender, 

53 guild_id: int, 

54 *, 

55 channel_name: str, 

56 category_name: str | None = None,

65 async def resolve_unified_channel(self) -> int: 

66 """Return the Discord channel ID of the unified A2A audit channel. 

67 

68 Cached after the first resolution. On a full cache miss, scans 

69 the guild for an existing channel matching ``channel_name``; on 

70 full miss, creates one (under ``category_name`` if configured). 

71 

72 Raises: 

73 discord.Forbidden: If the bot lacks ``Manage Channels`` in 

74 the guild and the channel needs to be created. 

75 discord.HTTPException: On Discord 5xx during discovery or 

76 creation. 

77 """ 

78 if self._unified_channel_id is not None: 

79 return self._unified_channel_id 

80 

81 channel_id = await self._discover() 

82 if channel_id is None: 

83 channel_id = await self._create() 

84 self._unified_channel_id = channel_id 

85 return channel_id

87 async def create_anchored_thread( 

88 self, 

89 channel_id: int, 

90 anchor_message_id: int, 

91 *, 

92 name: str, 

93 ) -> int: 

94 """Create a public thread anchored on a message in the unified channel. 

95 

96 The thread is rooted at ``anchor_message_id`` — the message 

97 becomes the thread's "starter" so anyone scrolling the parent 

98 channel sees the conversation's first line and can click into 

99 the thread for the rest. ``discord.py``'s 

100 :meth:`TextChannel.create_thread` accepts any 

101 :class:`discord.abc.Snowflake` as ``message=``, so we pass a 

102 synthetic :class:`discord.Object` (no ``fetch_message`` 

103 round-trip needed). 

104 

105 Args: 

106 channel_id: ID of the unified A2A channel (typically the 

107 value returned by :meth:`resolve_unified_channel`). 

108 anchor_message_id: ID of the message to anchor the thread 

109 on — typically the caller's just-posted request. 

110 name: Thread title. The caller is responsible for keeping 

111 this within Discord's 100-character limit. 

112 

113 Returns: 

114 The newly-created thread's snowflake ID. 

115 

116 Raises: 

117 discord.Forbidden: If the bot lacks ``Create Public Threads`` 

118 on the parent channel. 

119 discord.NotFound: If ``anchor_message_id`` no longer exists 

120 (race: the message was deleted between post and anchor). 

121 discord.HTTPException: On any other Discord-side failure. 

122 

123 Each error is logged at WARN with ``channel_id`` / 

124 ``anchor_message_id`` / ``name`` context before re-raising so 

125 operators have an actionable record without having to 

126 cross-reference call sites. 

127 """ 

128 try: 

129 channel = await self._sender.client.fetch_channel(channel_id) 

130 except discord.DiscordException: 

131 logger.warning( 

132 "a2a anchor thread channel fetch failed channel_id=%d anchor_message_id=%d name=%r", 

133 channel_id, 

134 anchor_message_id, 

135 name, 

136 exc_info=True, 

137 ) 

138 raise 

139 

140 if not isinstance(channel, discord.TextChannel): 

141 logger.warning( 

142 "a2a anchor thread target is not a TextChannel channel_id=%d type=%s anchor_message_id=%d name=%r", 

143 channel_id, 

144 type(channel).__name__, 

145 anchor_message_id, 

146 name, 

147 ) 

148 raise TypeError(f"channel_id={channel_id} resolved to {type(channel).__name__}, expected TextChannel") 

149 

150 try: 

151 thread = await channel.create_thread( 

152 name=name, 

153 message=discord.Object(id=anchor_message_id), 

154 ) 

155 except discord.DiscordException: 

156 logger.warning( 

157 "a2a anchor thread creation failed channel_id=%d anchor_message_id=%d name=%r", 

158 channel_id, 

159 anchor_message_id, 

160 name, 

161 exc_info=True, 

162 ) 

163 raise 

164 

165 logger.info( 

166 "created a2a anchored thread channel_id=%d anchor_message_id=%d thread_id=%d name=%r", 

167 channel_id, 

168 anchor_message_id, 

169 thread.id, 

170 name, 

171 ) 

172 return thread.id

174 async def _discover(self) -> int | None: 

175 """Look for an existing unified channel by name. Returns its ID or None. 

176 

177 Discord errors (e.g. ``discord.Forbidden`` if the bot loses 

178 guild access, ``HTTPException`` on Discord 5xx) propagate to 

179 the caller; we log them here so the failure surfaces in 

180 resolver-side logs, not just at the downstream call site. 

181 """ 

182 try: 

183 guild = await self._sender.client.fetch_guild(self._guild_id) 

184 channels = await guild.fetch_channels() 

185 except discord.DiscordException: 

186 logger.warning( 

187 "a2a unified-channel discovery failed name=%s", 

188 self._channel_name, 

189 exc_info=True, 

190 ) 

191 raise 

192 for channel in channels: 

193 if isinstance(channel, discord.TextChannel) and channel.name == self._channel_name: 

194 logger.info( 

195 "resolved a2a unified channel name=%s id=%s", 

196 self._channel_name, 

197 channel.id, 

198 ) 

199 return channel.id 

200 return None

202 async def _create(self) -> int: 

203 """Create the unified channel with default permissions. 

204 

205 ``discord.Forbidden`` (no Manage Channels permission), 

206 ``HTTPException``, and any other Discord error propagate to the 

207 caller. Logged here so the resolver-side log records the cause, 

208 mirroring :meth:`_discover`. 

209 

210 If ``category_name`` was configured, the channel is placed under 

211 the resolved category (created lazily on first miss). 

212 """ 

213 try: 

214 category = await self._resolve_category() 

215 guild = await self._sender.client.fetch_guild(self._guild_id) 

216 channel = await guild.create_text_channel( 

217 name=self._channel_name, 

218 category=category, 

219 reason="calfkit unified a2a audit channel", 

220 ) 

221 except discord.DiscordException: 

222 logger.warning( 

223 "a2a unified-channel creation failed name=%s", 

224 self._channel_name, 

225 exc_info=True, 

226 ) 

227 raise 

228 logger.info( 

229 "created a2a unified channel name=%s id=%s category_id=%s", 

230 self._channel_name, 

231 channel.id, 

232 category.id if category else None, 

233 ) 

234 return channel.id

236 async def _resolve_category(self) -> discord.CategoryChannel | None: 

237 """Return the configured A2A category, discovering or creating it. 

238 

239 Returns ``None`` (and short-circuits with no Discord I/O) when 

240 no ``category_name`` was supplied at construction — the original 

241 "uncategorized at root" behavior. 

242 

243 On first miss, scans the guild for a :class:`discord.CategoryChannel` 

244 whose name matches, creating it if none exists. The result is 

245 cached for the resolver's lifetime: a missing category is created 

246 at most once per process, and subsequent channel creations reuse 

247 the cached object with no additional Discord roundtrips. 

248 

249 Discord errors (``Forbidden`` if the bot lacks Manage Channels, 

250 ``HTTPException`` on 5xx) propagate to the caller and are logged 

251 here, mirroring :meth:`_discover` and :meth:`_create`. 

252 """ 

253 if self._category_name is None: 

254 return None 

255 if self._category is not None: 

256 return self._category 

257 try: 

258 guild = await self._sender.client.fetch_guild(self._guild_id) 

259 channels = await guild.fetch_channels() 

260 for channel in channels: 

261 if isinstance(channel, discord.CategoryChannel) and channel.name == self._category_name: 

262 logger.info( 

263 "resolved a2a category name=%s id=%s", 

264 self._category_name, 

265 channel.id, 

266 ) 

267 self._category = channel 

268 return channel 

269 category = await guild.create_category( 

270 name=self._category_name, 

271 reason="calfkit a2a channel category", 

272 ) 

273 except discord.DiscordException: 

274 logger.warning( 

275 "a2a category resolution failed name=%s", 

276 self._category_name, 

277 exc_info=True, 

278 ) 

279 raise 

280 logger.info( 

281 "created a2a category name=%s id=%s", 

282 self._category_name, 

283 category.id, 

284 ) 

285 self._category = category 

286 return category
