1"""Shared worker-lifetime helper: run a :class:`Worker` until a signal. 

2 

3Every roster runner (agents / tools) shares one 

4shutdown contract: bring the worker up, wait for either it to fail or a 

5SIGINT/SIGTERM, and treat *any* worker exit that is **not** preceded by a 

6shutdown signal as a crash worth surfacing to the process supervisor. This 

7module is the single source of truth for that contract so the runners cannot 

8drift apart. 

9 

10It is intentionally transport-agnostic and dependency-light (it imports 

11only :class:`~calfkit.worker.Worker` for typing), so both the 

12credential-free tools runner and the bridge runner can share it without 

13either pulling the other's dependencies. 

14 

15Why the embedded ``start()``/``stop()`` surface, not ``run()`` 

16-------------------------------------------------------------- 

17 

18This helper drives the worker via :meth:`Worker.start` / :meth:`Worker.stop` 

19and owns SIGINT/SIGTERM with its *own* handlers — the same embedded pattern 

20the bridge uses. It deliberately does **not** use :meth:`Worker.run`. 

21 

22``Worker.run()`` delegates to FastStream's ``run()``, whose first act is 

23``set_exit(...)`` → ``loop.add_signal_handler(SIGINT/SIGTERM, ...)``. Because 

24``add_signal_handler`` *replaces* the existing handler, calling ``run()`` 

25after we installed our own handlers silently clobbers ours: a commanded 

26``SIGTERM`` would then fire only FastStream's handler, draining the worker and 

27returning ``run()`` cleanly — but our stop event would never be set, so the 

28clean return reads as the "returned unexpectedly" crash below and the process 

29exits non-zero (a false crash that can spuriously restart under 

30``Restart=on-failure``). The embedded surface installs no signal handlers, so 

31ours survive and a commanded signal drains cleanly (exit 0). 

32 

33The supervisor invariant 

34------------------------- 

35 

36A worker that finishes *without* a shutdown signal is unexpected — under 

37normal operation the only way out is a signal. If we let that case exit 0, a 

38supervisor configured for ``Restart=on-failure`` would leave the process 

39down. So a signal-less worker exit (a boot crash, or a serving loop that ends 

40on its own) is converted into a re-raised exception — the worker's own crash, 

41or a synthesized :class:`RuntimeError` for a clean-but-signal-less exit — to 

42force a non-zero exit and a restart. 

43""" 

44

56def _select_exit_exception( 

57 worker_task: asyncio.Future, 

58 stop_task: asyncio.Future, 

59 drain_label: str, 

60) -> BaseException | None: 

61 """Decide the process-exit outcome after the run/stop race resolves. 

62 

63 Returns the exception to re-raise (forcing a non-zero exit so a 

64 supervisor restarts us), or ``None`` for a clean drain. Factored out as 

65 a pure decision function so the otherwise timing-dependent race between a 

66 worker crash and a shutdown signal is deterministically testable — 

67 callers (tests) can pass pre-completed futures. 

68 

69 A done ``worker_task`` always means the worker exited on its own: the 

70 signal handlers only ``stop.set()`` (completing ``stop_task``), never 

71 ``worker_task``. So the precedence is: 

72 

73 1. ``worker_task`` done **with an exception** → surface it (a crash wins, 

74 even if ``stop_task`` is also done from a signal on the same wakeup); 

75 2. done **cleanly, with a signal** → benign shutdown, drain; 

76 3. done **cleanly, no signal** → unexpected; synthesize a 

77 :class:`RuntimeError` (the supervisor invariant in the module docstring); 

78 4. ``worker_task`` **not** done → only the signal fired, drain. 

79 

80 Here ``worker_task`` wraps :meth:`Worker.start` followed by an indefinite 

81 serve (see :func:`run_worker_until_signal`): a boot failure completes it 

82 with the raised exception (case 1); a successful boot leaves it pending so 

83 only the signal can complete the race (case 4). The clean-no-signal case 

84 (3) cannot arise from the normal embedded path — the serve never returns on 

85 its own — but the contract is preserved verbatim so any future surface that 

86 *can* return cleanly still trips the supervisor invariant rather than 

87 exiting 0. 

88 """ 

89 if not worker_task.done(): 

90 logger.info("shutdown signal received, draining %s", drain_label) 

91 return None 

92 worker_exc = worker_task.exception() 

93 if worker_exc is not None: 

94 logger.error("worker crashed during runtime; exiting non-zero", exc_info=worker_exc) 

95 return worker_exc 

96 if stop_task.done(): 

97 logger.info("shutdown signal received, draining %s", drain_label) 

98 return None 

99 runtime_error = RuntimeError("worker serve returned unexpectedly without a shutdown signal") 

100 logger.error("%s; exiting non-zero", runtime_error) 

101 return runtime_error

104async def _serve(worker: Worker) -> None: 

105 """Boot the worker (managed surface) and then serve until cancelled. 

106 

107 :meth:`Worker.start` returns as soon as the broker is up and the nodes are 

108 consuming in the background; the serve is event-driven from there with no 

109 awaitable to block on. We model the worker's *serving lifetime* as a task 

110 that boots and then parks indefinitely, so the caller can race it against 

111 the stop event: 

112 

113 * a boot failure completes this task with the raised exception — surfaced 

114 as a crash by :func:`_select_exit_exception`; 

115 * a successful boot parks here forever, so only a shutdown signal can 

116 complete the race; the caller then drains via :meth:`Worker.stop`. 

117 

118 The park (``asyncio.Event().wait()`` on a never-set event) holds the 

119 serving open until the caller cancels this task in its ``finally``. 

120 """ 

121 await worker.start() 

122 await asyncio.Event().wait() # park until cancelled by the caller's drain

125async def run_worker_until_signal(worker: Worker, *, drain_label: str = "worker") -> None: 

126 """Run ``worker`` until SIGINT/SIGTERM, then drain cleanly. 

127 

128 Drives the worker via the embedded :meth:`Worker.start` / :meth:`Worker.stop` 

129 surface (signals stay OFF, so FastStream never re-registers — and so never 

130 clobbers — our handlers; see the module docstring) and owns SIGINT/SIGTERM 

131 with its own handlers. A serve task (:func:`_serve`) boots the worker and 

132 then parks; it is raced against a stop-event task armed by the signal 

133 handlers, and whichever completes first decides the outcome: 

134 

135 * **Signal first** — log "shutdown signal received" and drain: cancel the 

136 still-parked serve task and :meth:`Worker.stop` the worker in ``finally``. 

137 * **Serve finished first** — inspect why. A propagated exception is re-raised 

138 (a boot crash), and that takes precedence even if a shutdown signal 

139 arrived on the same event-loop wakeup. A clean finish with no pending 

140 signal is *unexpected* and synthesized into a :class:`RuntimeError` (see 

141 the module docstring's supervisor invariant), which is then re-raised; a 

142 clean finish that coincides with a signal is a benign shutdown and drains 

143 normally. See :func:`_select_exit_exception`. 

144 

145 The worker is always drained via :meth:`Worker.stop` in ``finally`` — a 

146 no-op if it never started, idempotent otherwise — so neither a failed boot 

147 nor a clean run ever leaks the broker connection. 

148 

149 Args: 

150 worker: The calfkit :class:`~calfkit.worker.Worker` to run. 

151 drain_label: Human-readable label for the worker, used in the 

152 "shutdown signal received, draining {drain_label}" log line so 

153 a multi-process deployment's logs identify which worker is 

154 draining (e.g. ``"tools worker"``, ``"agents worker"``). 

155 

156 Raises: 

157 BaseException: Whatever :meth:`Worker.start` raised, if boot crashed; or 

158 a :class:`RuntimeError` if the serve finished cleanly without a 

159 shutdown signal. 

160 """ 

161 stop = asyncio.Event() 

162 loop = asyncio.get_running_loop() 

163 # Install OUR handlers before booting. ``Worker.start()`` never touches 

164 # signals, so these survive the whole run — a SIGTERM during a slow boot 

165 # still sets the stop event and the boot is then cancelled+drained cleanly. 

166 for sig in (signal.SIGINT, signal.SIGTERM): 

167 loop.add_signal_handler(sig, stop.set) 

168 

169 worker_task = asyncio.create_task(_serve(worker)) 

170 stop_task = asyncio.create_task(stop.wait()) 

171 worker_exc: BaseException | None = None 

172 try: 

173 await asyncio.wait( 

174 {worker_task, stop_task}, 

175 return_when=asyncio.FIRST_COMPLETED, 

176 ) 

177 # A done ``worker_task`` always means the worker exited on its own — 

178 # a signal only sets the stop event, it never completes that task. 

179 # ``_select_exit_exception`` therefore surfaces a boot crash even 

180 # when a SIGINT/SIGTERM lands on the same event-loop wakeup; a clean 

181 # exit 0 would otherwise mask the crash from the process supervisor. 

182 worker_exc = _select_exit_exception(worker_task, stop_task, drain_label) 

183 finally: 

184 for t in (worker_task, stop_task): 

185 if not t.done(): 

186 t.cancel() 

187 # Gather the cancelled serve/stop tasks (retrieve their results so no 

188 # "Task was destroyed but it is pending" warning leaks), then drain the 

189 # worker. ``stop()`` is safe on every path — clean signal, boot crash, or 

190 # unexpected serve return — because it is idempotent: even after a failed 

191 # boot (where ``start()`` built the app before raising and already ran its 

192 # own cleanup), this second ``stop()`` re-runs an app shutdown that no-ops 

193 # against the already-disconnected broker / torn-down resource stack. 

194 await asyncio.gather(worker_task, stop_task, return_exceptions=True) 

195 await worker.stop() 

196 

197 if worker_exc is not None: 

198 raise worker_exc
