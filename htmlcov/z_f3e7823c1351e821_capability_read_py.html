1"""Best-effort CLI read of the live MCP capability view. 

2 

3The ``mcp.capabilities`` compacted topic is the org-wide source of truth for 

4which MCP tools exist *right now* — including servers hosted by other 

5machines that this host's ``mcp.json`` knows nothing about. The tools 

6editor reads it to offer per-tool ``mcp/<server>/<tool>`` rows. 

7 

8Strictly best-effort: the CLI must work offline (broker down, workspace 

9closed, dev laptop on a plane), so every failure path degrades to an empty 

10snapshot — the editor then falls back to server-level rows from the local 

11``mcp.json``. The short catch-up timeout keeps the editor snappy; a partial 

12catch-up just means fewer rows, never an error. 

13""" 

14 

15from __future__ import annotations 

16 

17import asyncio 

18import logging 

19 

20logger = logging.getLogger(__name__) 

21 

22CAPABILITY_TOPIC = "mcp.capabilities" 

23# Bounded so the interactive editor stays snappy on a slow/filtered broker; 

24# a healthy local broker replays the tiny compacted topic well inside this. 

25_DEFAULT_TIMEOUT_SECONDS = 1.5 

26 

27 

28def snapshot_capability_tools( 

29 server_urls: str, *, timeout: float = _DEFAULT_TIMEOUT_SECONDS 

30) -> dict[str, list[str]] | None: 

31 """``{server: [tool, ...]}`` from the live capability view, or ``None``. 

32 

33 Replays the compacted topic with a ``timeout``-bounded catch-up and 

34 returns each advertised toolbox's tool names (sorted). Any failure — 

35 unreachable broker, missing topic, replay timeout — returns ``None`` 

36 (NOT ``{}``): callers can tell "the view answered and is empty" apart 

37 from "the view was unreachable" and tell the operator which one 

38 happened. 

39 """ 

40 try: 

41 return asyncio.run(_snapshot(server_urls, timeout)) 

42 except Exception as exc: 

43 logger.debug("capability view unavailable (%s); offline rows only", exc) 

44 return None 

45 

46 

47async def _snapshot(server_urls: str, timeout: float) -> dict[str, list[str]]: 

48 from calfkit.models.capability import CapabilityRecord 

49 from ktables import KafkaTable 

50 

51 table: KafkaTable[CapabilityRecord] = KafkaTable.json( 

52 bootstrap_servers=server_urls, 

53 topic=CAPABILITY_TOPIC, 

54 model=CapabilityRecord, 

55 catchup_timeout=timeout, 

56 # Never create the topic from a read-only CLI peek: a missing topic 

57 # means no toolbox has ever advertised, which IS the answer ({}). 

58 ensure_topic=False, 

59 ) 

60 await table.start() 

61 try: 

62 records = table.snapshot() 

63 finally: 

64 await table.stop() 

65 return { 

66 toolbox_id: sorted(tool.name for tool in record.tools) 

67 for toolbox_id, record in records.items() 

68 }
