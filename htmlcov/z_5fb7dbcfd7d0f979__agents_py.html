1"""Shared agent-directory inspection and ``.md`` write helpers for the CLI. 

2 

3Both ``disco init`` (which *reports* the agents an install would load) and 

4``disco agent tools`` (which *picks* one to edit) need the same answer to 

5"which ``.md`` files are live agents?". Factoring :func:`detect_agents` here 

6keeps that one definition from drifting between the two callers — a mismatch 

7would let ``init`` report an agent the editor can't open, or vice versa. 

8 

9This module also owns the agent-file *write* and *identity* helpers that both 

10``disco init`` (first-run setup) and ``disco agent create`` build on: 

11slugifying a typed name into a valid stem, deriving a default display name and 

12body, the create/update :func:`write_agent` path, and the tools-checkbox 

13builder :func:`pick_tools`. They live here rather than in ``init`` so the two 

14commands share one implementation and can't drift. 

15 

16The skip rules in :func:`detect_agents` mirror the loader's 

17(:func:`calfcord.agents.loader.load_agents_dir`): dot-prefixed files and 

18``*.template.md`` reference templates are not live agents, so the names returned 

19here match exactly what ``calfkit-agent`` would run. 

20 

21:func:`pick_tools` defers its ``TOOL_REGISTRY`` import into the function body so 

22the rest of the module (e.g. :func:`detect_agents`) stays light — importing the 

23registry eagerly composes the tool surface (importing the vendored 

24``calfkit-tools`` nodes). 

25"""

48# The starter agent's name and the *exact* description the installer seeds it 

49# with. The prune-pristine check in :func:`write_agent` keys off this string: 

50# an ``assistant.md`` still carrying it is an untouched seed (safe to remove 

51# when the operator names a different agent); any other description means the 

52# operator customized it and it must be preserved. 

53STARTER_AGENT_NAME = "assistant" 

54DEFAULT_DESCRIPTION = "General-purpose AI teammate — answers questions and helps with tasks." 

55 

56# Tools that grant code-execution or filesystem-write reach into the 

57# ``calfkit-tools`` host. Selecting any of them drives the one-line security 

58# caution, because anyone who can @mention the agent can then drive them. 

59# ``terminal`` and ``execute_code`` run arbitrary code on the host (see 

60# docs/adr/0005); ``write_file``/``patch`` mutate the shared workspace. 

61_DANGEROUS_TOOLS = frozenset( 

62 {"terminal", "process", "execute_code", "write_file", "patch"} 

63) 

64 

65# Permitted characters for an agent-name *stem*. The on-disk identifier must 

66# satisfy ``AgentDefinition.agent_id`` (``[a-z0-9_-]{1,32}``); we slugify toward 

67# that here so a friendly typed name ("My Helper") becomes a valid stem 

68# ("my_helper") rather than failing validation at write time. 

69_STEM_INVALID = re.compile(r"[^a-z0-9_-]+")

72def detect_agents(agents_dir: Path) -> list[str]: 

73 """Return the agent names (``.md`` stems) ``agents_dir`` would load, sorted. 

74 

75 Returns an empty list when ``agents_dir`` is not an existing directory, so 

76 callers can treat "no dir" and "empty dir" identically — both mean "no 

77 agents to act on". Dotfiles and ``*.template.md`` templates are skipped to 

78 match the loader; the result is sorted for deterministic prompts/output. 

79 """ 

80 if not agents_dir.is_dir(): 

81 return [] 

82 return sorted( 

83 p.stem 

84 for p in agents_dir.glob("*.md") 

85 if not p.name.startswith(".") and not p.name.endswith(".template.md") 

86 )

89def slug_stem(raw: str) -> str: 

90 """Coerce a typed agent name into a safe ``.md`` filename stem. 

91 

92 The frontmatter ``name`` (and thus the filename) must match 

93 ``[a-z0-9_-]{1,32}``; an operator typing "My Helper" should not hit a 

94 validation error, so we lowercase, turn runs of disallowed characters into 

95 single underscores, and trim the result. A name that slugifies to nothing 

96 (e.g. all punctuation) falls back to the starter name rather than producing 

97 an empty, invalid stem — keeping the wizard moving instead of aborting. 

98 """ 

99 slug = _STEM_INVALID.sub("_", raw.strip().lower()).strip("_-") 

100 slug = slug[:32].strip("_-") 

101 result = slug or STARTER_AGENT_NAME 

102 # Postcondition: the stem is a valid agent id, so callers can write it as a 

103 # filename without a second validation. The fallback guarantees non-empty, 

104 # and slugification guarantees the charset/length, so this can only fire on 

105 # a bug in those rules — assert rather than silently emit an invalid stem. 

106 assert AGENT_ID_PATTERN.fullmatch(result), f"slugified stem {result!r} is not a valid agent id" 

107 return result

110def existing_agent(agents_dir: Path, name: str) -> AgentDefinition | None: 

111 """Return the parsed agent at ``agents_dir/<name>.md`` if it parses, else ``None``. 

112 

113 Used purely to pre-fill the description/model prompts with the operator's 

114 current values on a re-run that targets an existing agent. A missing or 

115 malformed file is not an error here — we simply offer the defaults — so the 

116 parse is guarded; the actual write path validates strictly. 

117 """ 

118 target = agents_dir / f"{name}.md" 

119 if not target.is_file(): 

120 return None 

121 try: 

122 return parse_agent_md(target) 

123 except (ValueError, OSError): 

124 return None

127def agent_body(name: str) -> str: 

128 """Render the generic system-prompt body for a brand-new agent. 

129 

130 A minimal, generic prompt so the agent answers sensibly from the first boot 

131 without further editing. Addresses the agent by a human-friendly rendering of 

132 its slug ``name`` (underscores/dashes title-cased, "my_helper" → "My Helper") 

133 so the greeting reads naturally. Kept separate from the frontmatter so the 

134 create path can serialize identity fields through ``frontmatter.dumps`` (which 

135 YAML-quotes free-text values safely) rather than string interpolation. 

136 """ 

137 human = name.replace("_", " ").replace("-", " ").title() 

138 return ( 

139 f"You are {human}, a helpful AI teammate in this Discord workspace. Answer\n" 

140 "questions and help with tasks clearly and concisely. If you don't know something,\n" 

141 "say so rather than guessing.\n" 

142 )

145def atomic_write(path: Path, payload: str) -> None: 

146 """Write ``payload`` to ``path`` via a same-dir tmp file + atomic rename. 

147 

148 A partial agent file would make the next ``calfkit-agent`` boot fail to 

149 load the directory, so the create path must never leave a half-written file 

150 behind on error. ``path.parent`` is created first because a fresh install's 

151 ``agents/`` directory may not exist yet (unlike :mod:`md_writer`'s in-place 

152 rewrite, which can assume the file — and so the dir — already exists). 

153 """ 

154 path.parent.mkdir(parents=True, exist_ok=True) 

155 fd, tmp_name = tempfile.mkstemp(dir=path.parent, prefix=f".{path.name}.", suffix=".tmp") 

156 tmp_path = Path(tmp_name) 

157 try: 

158 with os.fdopen(fd, "w", encoding="utf-8") as tmp: 

159 tmp.write(payload) 

160 tmp.flush() 

161 os.fsync(tmp.fileno()) 

162 os.replace(tmp_path, path) 

163 except BaseException: 

164 tmp_path.unlink(missing_ok=True) 

165 raise

168def is_pristine_seed(agents_dir: Path) -> bool: 

169 """True when ``agents_dir/assistant.md`` is the untouched seeded starter. 

170 

171 "Untouched" is detected by the seed's two stable identity markers: its 

172 ``agent_id`` is still ``assistant`` and its description is still the exact 

173 seed default. If the operator customized the description, both halves no 

174 longer hold and the file must be preserved. A missing or malformed 

175 ``assistant.md`` is treated as "not a pristine seed" (nothing safe to 

176 prune), so the parse is guarded — a broken file is never deleted on a guess. 

177 """ 

178 seed = agents_dir / f"{STARTER_AGENT_NAME}.md" 

179 if not seed.is_file(): 

180 return False 

181 try: 

182 parsed = parse_agent_md(seed) 

183 except (ValueError, OSError): 

184 return False 

185 return parsed.agent_id == STARTER_AGENT_NAME and parsed.description == DEFAULT_DESCRIPTION

188def write_agent( 

189 agents_dir: Path, 

190 *, 

191 name: str, 

192 description: str, 

193 provider: str, 

194 model: str, 

195 tools: list[str], 

196 prune_seed: bool = False, 

197) -> Path: 

198 """Create or update ``agents_dir/<name>.md`` for the wizard's agent. 

199 

200 Two paths, both validate-before-write so a bad value never lands on disk: 

201 

202 * **Target exists** — update the agent in place, preserving its body: 

203 rewrite ``description``/``provider``/``model`` via 

204 :func:`md_writer._update_fields`, then the tool list via 

205 :func:`md_writer.update_tools`. Both are validated-atomic, so a bad value 

206 leaves the file untouched. 

207 * **Target missing** — build the frontmatter as a mapping and serialize it 

208 with :func:`frontmatter.dumps` (NOT string interpolation), which 

209 YAML-quotes free-text values so a description like ``"Calendar: book 

210 meetings"`` or one carrying quotes/``#``/leading punctuation can't corrupt 

211 the file. The synthetic :class:`AgentDefinition` is built *first* (mirroring 

212 :func:`md_writer._update_fields`), so an invalid value raises before any 

213 disk write. After the atomic write, when ``prune_seed`` is set and the 

214 operator named a *different* agent (``name != "assistant"``) on an install 

215 still carrying the *pristine* seeded ``assistant.md``, that seed is deleted 

216 so they end with one clean agent. ``init`` opts in for its first-run setup; 

217 ``agent create`` leaves it off so a second agent never removes the starter. 

218 A *customized* ``assistant.md`` (or naming the agent ``assistant`` itself) 

219 is never deleted. 

220 

221 Raises: 

222 ValueError: a field value fails :class:`AgentDefinition` validation 

223 (create path) or the existing ``.md``/new value is invalid (update 

224 path). No partial file is written. 

225 OSError: a filesystem error during the atomic write. No partial file is 

226 written. 

227 """ 

228 target = agents_dir / f"{name}.md" 

229 

230 if target.exists(): 

231 md_writer._update_fields(target, {"description": description, "provider": provider, "model": model}) 

232 md_writer.update_tools(target, tools) 

233 return target 

234 

235 body = agent_body(name) 

236 metadata = { 

237 "name": name, 

238 "description": description, 

239 "provider": provider, 

240 "model": model, 

241 "tools": list(tools), 

242 } 

243 # Validate the full definition in memory FIRST (mirrors 

244 # md_writer._update_fields): a bad free-text value raises here, before any 

245 # bytes touch disk, so the create path can never leave an unloadable file. 

246 AgentDefinition(**{**metadata, "system_prompt": body, "source_path": target}) 

247 

248 payload = frontmatter.dumps(frontmatter.Post(body, **metadata)) 

249 if not payload.endswith("\n"): 249 ↛ 251line 249 didn't jump to line 251 because the condition on line 249 was always true

250 payload += "\n" 

251 atomic_write(target, payload) 

252 

253 # Prune the pristine starter only when the caller opted in (``init``'s 

254 # first-run "one clean agent" goal) and a *different* agent was created; 

255 # naming the agent ``assistant`` would have hit the update path above. 

256 # ``agent create`` leaves ``prune_seed`` False so adding a second agent 

257 # never deletes the operator's starter. 

258 if prune_seed and name != STARTER_AGENT_NAME and is_pristine_seed(agents_dir): 

259 (agents_dir / f"{STARTER_AGENT_NAME}.md").unlink(missing_ok=True) 

260 logger.info("pruned pristine seed assistant.md after creating %s", target) 

261 

262 return target

265def pick_tools( 

266 prompter: Prompter, 

267 name: str, 

268 *, 

269 mcp_servers_fn: Callable[[], list[str]] | None = None, 

270 live_tools_fn: Callable[[], dict[str, list[str]]] | None = None, 

271) -> list[str]: 

272 """Prompt for the agent's tools and return the selected tokens. 

273 

274 Every builtin (sorted :data:`calfcord.tools.TOOL_REGISTRY`) is offered 

275 pre-checked so the default is the same "all builtins" set a frontmatter that 

276 omits ``tools:`` would expand to; MCP rows (``mcp/<server>`` from mcp.json 

277 plus live-discovered per-tool rows) are offered UNCHECKED because MCP is an 

278 explicit grant that never rides that default. Row building is shared with 

279 the ``agent tools`` editor (:func:`calfcord.cli.agent_tools._build_choices`) 

280 so the two surfaces can't drift. If a write/shell tool ends up selected we 

281 print the security caution, because anyone who can @mention the agent can 

282 then drive it. 

283 """ 

284 from calfcord.cli.agent_tools import ( 

285 _build_choices, 

286 _default_live_tools, 

287 _default_mcp_servers, 

288 ) 

289 from calfcord.tools import TOOL_REGISTRY 

290 

291 choices = _build_choices( 

292 set(TOOL_REGISTRY), # builtins pre-checked; MCP rows start unchecked 

293 mcp_servers=(mcp_servers_fn or _default_mcp_servers)(), 

294 live_tools=(live_tools_fn or _default_live_tools)(), 

295 ) 

296 

297 selected = prompter.checkbox( 

298 f"Tools for {name} (all selected — deselect any you don't want):", 

299 choices, 

300 instruction="space toggles, enter confirms", 

301 ) 

302 

303 if _DANGEROUS_TOOLS.intersection(selected): 

304 print( 

305 "note: these tools include code execution + file write access in the " 

306 "calfkit-tools launch dir, drivable by anyone who can @mention this agent " 

307 "— keep the bot off public Discord (docs/security.md §3.4)." 

308 ) 

309 

310 return selected
