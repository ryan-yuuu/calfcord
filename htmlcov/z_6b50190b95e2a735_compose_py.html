1"""Render the Process Compose project that supervises a calfcord host. 

2 

3The Process Compose YAML is *derived state* — calfcord generates it from the 

4agents on disk and config; the user never edits it (design §3.1). This module is 

5the pure heart of that generation: agent ids + home dir + launcher prefix in, a 

6process-compose project ``dict`` out. No filesystem, no broker, no network — so 

7the structure is fully golden-testable and the same generator works whether the 

8host runs detached, in dev, or under a future supervisor swap. 

9 

10The shape is the §13.2 Phase-0 contract, pinned against Process Compose 

11``v1.110.0`` (config schema ``version: "0.5"``): 

12 

13* **Substrate** (``broker``, ``bridge``) autostarts under ``disco start``; 

14 **roster** (every agent + ``tools``) is declared 

15 ``disabled`` and waits for an explicit ``... start`` (a ``POST /process/start``). 

16 "Nothing runs that the user did not start" is a trust property, so the split is 

17 encoded here, not left to the launcher. 

18* Every ``command`` invokes the calfcord *launcher* (the shim) rather than a 

19 reconstructed ``uv run`` line, so the venv + ``--env-file`` + default env come 

20 from one place and no secret literal is ever inlined into the YAML (design 

21 §12.3). The launcher prefix is a parameter so this generator is mode-agnostic 

22 and unit-testable. 

23* ``bridge`` and roster members gate on the broker via ``depends_on`` 

24 (``process_healthy``); readiness is an ``exec`` probe (the bridge has no HTTP 

25 server) calling ``<launcher> _healthcheck <component>``. 

26* ``restart: always`` for the substrate (``broker``, ``bridge``); ``on_failure`` 

27 for the whole roster — every agent *and* ``tools`` — which 

28 now all run via ``run_worker_until_signal`` and therefore force a non-zero exit 

29 on any uncommanded exit (a crash *or* a clean signal-less return), while an 

30 operator-commanded ``stop`` is suppressed from restart by Process Compose 

31 regardless of exit code. (The bridge stays ``always``: it owns its own signals 

32 and exits 0 on a clean shutdown, so ``on_failure`` would never fire to recover 

33 it from an uncommanded clean return.) Never ``exit_on_failure``. 

34 

35The REST port is intentionally absent: it is a flag to ``process-compose up`` 

36(``-p <PC_PORT>``), not a field in the project file (design §13.2). 

37"""

46# Process Compose config-file schema version (NOT the binary version). v1.110.0 

47# reads the "0.5" schema; confirmed against the process-compose docs. 

48COMPOSE_SCHEMA_VERSION = "0.5" 

49 

50# Readiness-probe cadence — the §13.2 pinned values. An exec probe (the bridge 

51# exposes no HTTP endpoint) shells out to the launcher's internal healthcheck. 

52_PROBE_INITIAL_DELAY_SECONDS = 2 

53_PROBE_PERIOD_SECONDS = 3 

54_PROBE_TIMEOUT_SECONDS = 5 

55_PROBE_SUCCESS_THRESHOLD = 1 

56_PROBE_FAILURE_THRESHOLD = 3 

57 

58# Autorestart backoff shared by both restart policies (always for the substrate, 

59# on_failure for the roster); max_restarts 0 == unlimited retries in Process 

60# Compose, applied uniformly. 

61_RESTART_BACKOFF_SECONDS = 2 

62_RESTART_MAX_RESTARTS = 0 

63 

64# Per-host log rotation (project-level); §13.2. 

65_LOG_MAX_SIZE_MB = 10 

66_LOG_MAX_AGE_DAYS = 7 

67_LOG_MAX_BACKUPS = 5 

68_LOG_COMPRESS = True 

69 

70# Graceful shutdown: SIGTERM with a 10s grace window — comfortably above the 

71# ~2s an agent needs to publish its AgentDepartureEvent — signalling the whole 

72# group (parent_only: false) so child workers under the shim also stop. 

73_SHUTDOWN_SIGNAL = 15 

74_SHUTDOWN_TIMEOUT_SECONDS = 10 

75_SHUTDOWN_PARENT_ONLY = False 

76 

77_HEALTHY = "process_healthy"

79# The filename stem of the supervisor's *own* log (``process-compose up -L 

80# <stem>.log``). It is not a process the generator declares, but it sits beside 

81# the per-process logs and is a legitimate tail target, so its name must agree 

82# across the modules that write it (``lifecycle._SUPERVISOR_LOG_FILENAME``) and 

83# read it (``cli.logs``). Both derive from this one stem so the literal can never 

84# drift: lifecycle reconstructs the filename as ``SUPERVISOR_LOG_STEM + ".log"``, 

85# and ``cli.logs`` passes the stem to ``_log_location`` (which appends ``.log``). 

86# Homed here because both consumers already import ``compose`` and ``compose`` is 

87# import-light, keeping the logs CLI's decoupling intact. 

88SUPERVISOR_LOG_STEM = "process-compose" 

89 

90# Process names owned by the substrate + non-agent components. An agent id equal 

91# to one of these would silently overwrite that process's entry in the shared 

92# `processes` dict (corrupting the substrate), so the generator rejects it. 

93_RESERVED_PROCESS_NAMES = frozenset({"broker", "bridge", "tools"}) 

94 

95# The slot-name convention for MCP servers, homed here (like 

96# SUPERVISOR_LOG_STEM) because compose declares the slots and the roster 

97# drives them — both must agree on the literal, and compose is the 

98# import-light module the supervisor package already leans on. 

99MCP_SLOT_PREFIX = "mcp-"

102def mcp_slot_name(server: str) -> str: 

103 """The Process Compose slot for MCP server ``server`` (``mcp-<server>``).""" 

104 return f"{MCP_SLOT_PREFIX}{server}"

106 

107# Hosts whose broker URL means "a local broker calfcord itself supervises": the 

108# native install runs Tansu as the compose-managed ``broker`` process bound to 

109# loopback. Anything else is an EXTERNAL broker the operator runs elsewhere, so 

110# calfcord must neither declare a local broker slot for it nor start one. The 

111# empty host (":9092" / bare "") is treated as loopback (no host == localhost). 

112_LOOPBACK_HOSTS = frozenset({"", "localhost", "::1"})

115def _host_of(url: str) -> str: 

116 """Extract the host from a bare ``host[:port]`` broker URL (no scheme). 

117 

118 Handles the bracketed IPv6 form (``[::1]:9092`` → ``::1``) and a bare IPv6 

119 literal (multiple colons, unbracketed, no port — kept whole); a single colon 

120 is the ordinary ``host:port`` separator. 

121 """ 

122 url = url.strip() 

123 if url.startswith("["): 

124 end = url.find("]") 

125 if end != -1: 125 ↛ 127line 125 didn't jump to line 127 because the condition on line 125 was always true

126 return url[1:end] 

127 if url.count(":") == 1: 

128 return url.rsplit(":", 1)[0] 

129 return url

132def _is_loopback_host(host: str) -> bool: 

133 """Whether ``host`` names this machine's loopback (localhost / 127.0.0.0/8 / ::1).""" 

134 host = host.strip().lower() 

135 return host in _LOOPBACK_HOSTS or host.startswith("127.")

138def broker_is_compose_managed(server_urls: str) -> bool: 

139 """Whether ``server_urls`` designates a local broker calfcord supervises. 

140 

141 This is the single predicate that keeps two decisions from diverging (design 

142 §13.2): whether the rendered project declares a local ``broker`` process, and 

143 whether :func:`lifecycle.start` runs its pre-launch broker fast-fail probe. 

144 

145 A **loopback** URL (``localhost`` / ``127.0.0.0/8`` / ``::1``, with or without 

146 a port) means calfcord runs Tansu as the compose-managed ``broker`` process: 

147 ``process-compose up`` launches it (with its own readiness probe + depends_on 

148 graph), so `start` must NOT pre-probe it — the broker is cold until `up` starts 

149 it. A non-loopback (external) URL means the operator's broker lives elsewhere, 

150 so no local slot is declared and the pre-launch fast-fail stands. A 

151 comma-separated bootstrap list is compose-managed only if EVERY host is 

152 loopback; an empty string is external (unknown → keep the fast-fail). 

153 """ 

154 hosts = [h for h in (part.strip() for part in server_urls.split(",")) if h] 

155 if not hosts: 

156 return False 

157 return all(_is_loopback_host(_host_of(h)) for h in hosts)

173def _readiness_probe(launcher: str, component: str) -> dict: 

174 """An exec readiness probe driving ``depends_on: process_healthy``.""" 

175 return { 

176 "exec": {"command": f"{launcher} _healthcheck {component}"},

185def _process( 

186 *, 

187 command: str, 

188 home: str, 

189 name: str, 

190 disabled: bool, 

191 restart_policy: str, 

192 depends_on: Mapping[str, str] | None = None, 

193 readiness_probe: dict | None = None, 

194) -> dict: 

195 proc: dict = { 

196 "command": command, 

197 "disabled": disabled, 

198 "availability": _restart(restart_policy), 

199 "shutdown": { 

200 "signal": _SHUTDOWN_SIGNAL, 

201 "timeout_seconds": _SHUTDOWN_TIMEOUT_SECONDS, 

202 "parent_only": _SHUTDOWN_PARENT_ONLY, 

203 }, 

204 "log_location": _log_location(home, name), 

205 } 

206 if depends_on is not None: 

207 proc["depends_on"] = {dep: {"condition": condition} for dep, condition in depends_on.items()} 

208 if readiness_probe is not None: 

209 proc["readiness_probe"] = readiness_probe 

210 return proc

213def build_compose_project( 

214 *, 

215 agent_ids: Iterable[str], 

216 home: str, 

217 launcher: str, 

218 mcp_servers: Iterable[str] = (), 

219 broker_managed: bool = True, 

220) -> dict: 

221 """Build the Process Compose project that supervises one calfcord host. 

222 

223 ``agent_ids`` are the roster agents to declare (the caller enumerates the 

224 host's ``agents/*.md``; this function never reads the filesystem), and 

225 ``mcp_servers`` the MCP server names to declare (the caller enumerates 

226 ``mcp.json`` via the no-secrets ``list_server_names`` reader) — each gets 

227 its own ``mcp-<server>`` roster slot, so one broken server entry can 

228 never take down sibling servers and each restarts independently. ``home`` 

229 is ``$CALFCORD_HOME`` — only the per-process ``state/logs/<name>.log`` paths 

230 use it. ``launcher`` is the shim prefix every ``command`` is built on 

231 (e.g. ``$CALFCORD_HOME/shims/disco``); the generator never reconstructs 

232 ``uv run`` flags or inlines secrets. 

233 

234 ``broker_managed`` (default ``True``, the native local install) declares Tansu 

235 as the autostart ``broker`` process and health-gates the bridge + roster on it. 

236 Set ``False`` for an EXTERNAL broker (the operator runs Kafka elsewhere): no 

237 local ``broker`` process is declared — starting an ephemeral broker nobody 

238 talks to is wrong — and nothing carries a ``depends_on`` the broker (a 

239 dependency on an undeclared process would make process-compose reject the 

240 project). The caller derives this flag from 

241 :func:`broker_is_compose_managed` so manifest inclusion and `start`'s 

242 pre-launch probe cannot diverge (design §13.2). 

243 

244 Returns a plain ``dict`` (serialize with :func:`render_compose`). See the 

245 module docstring for the substrate/roster, restart, depends_on, and probe 

246 contract this encodes. Raises :class:`ValueError` if an ``agent_id`` collides 

247 with a reserved substrate/component process name or an MCP server slot. 

248 """ 

249 agent_ids = list(agent_ids) 

250 mcp_servers = list(mcp_servers) 

251 mcp_slots = {mcp_slot_name(server): server for server in mcp_servers} 

252 reserved = sorted(_RESERVED_PROCESS_NAMES.intersection(agent_ids)) 

253 if reserved: 

254 raise ValueError( 

255 f"agent id(s) {reserved} collide with reserved process name(s) " 

256 f"{sorted(_RESERVED_PROCESS_NAMES)}; rename the agent(s)" 

257 ) 

258 slot_collisions = sorted(set(mcp_slots).intersection(agent_ids)) 

259 if slot_collisions: 

260 raise ValueError( 

261 f"agent id(s) {slot_collisions} collide with MCP server process " 

262 f"slot(s) (one per mcp.json server, named mcp-<server>); rename the " 

263 f"agent(s) or the server(s)" 

264 ) 

265 

266 processes: dict[str, dict] = {} 

267 

268 # The broker gate every downstream process shares — present only when the 

269 # broker is a local compose process. For an external broker there is no local 

270 # ``broker`` process to health-gate on, so the dependency is dropped entirely 

271 # (process-compose rejects a depends_on to an undeclared process). 

272 broker_dep = {"broker": _HEALTHY} if broker_managed else None 

273 

274 # Substrate — autostarts, health-gated. The broker (when compose-managed) has a 

275 # readiness probe checking metadata reachability (not bare TCP); the bridge's 

276 # checks the Discord heartbeat. `start` gates downstream on the bridge. 

277 if broker_managed: 

278 processes["broker"] = _process( 

279 command=f"{launcher} broker", 

280 home=home, 

281 name="broker", 

282 disabled=False, 

283 restart_policy="always", 

284 readiness_probe=_readiness_probe(launcher, "broker"), 

285 ) 

286 processes["bridge"] = _process( 

287 command=f"{launcher} run bridge", 

288 home=home, 

289 name="bridge", 

290 disabled=False, 

291 restart_policy="always", 

292 depends_on=broker_dep, 

293 readiness_probe=_readiness_probe(launcher, "bridge"), 

294 ) 

295 

296 # Roster — declared disabled; each member clocks in on an explicit start and 

297 # gates on the broker being healthy (when it is a local process). Every roster 

298 # member (agents *and* tools) runs via run_worker_until_signal, which forces a 

299 # non-zero exit on any uncommanded exit, so on_failure restarts a crash while an 

300 # operator-commanded stop is suppressed from restart by Process Compose. 

301 for agent_id in agent_ids: 

302 processes[agent_id] = _process( 

303 command=f"{launcher} run agent {agent_id}", 

304 home=home, 

305 name=agent_id, 

306 disabled=True, 

307 restart_policy="on_failure", 

308 depends_on=broker_dep, 

309 ) 

310 

311 for component in ("tools",): 

312 processes[component] = _process( 

313 command=f"{launcher} run {component}", 

314 home=home, 

315 name=component, 

316 disabled=True, 

317 restart_policy="on_failure", 

318 depends_on=broker_dep, 

319 ) 

320 

321 # MCP servers — roster members like agents (disabled, on_failure, 

322 # broker-gated when local), one slot per mcp.json server for failure isolation. 

323 for slot, server in mcp_slots.items(): 

324 processes[slot] = _process( 

325 command=f"{launcher} run mcp {server}", 

326 home=home, 

327 name=slot, 

328 disabled=True, 

329 restart_policy="on_failure", 

330 depends_on=broker_dep, 

331 ) 

332 

333 return { 

334 "version": COMPOSE_SCHEMA_VERSION, 

335 "log_configuration": { 

336 "rotation": { 

337 "max_size_mb": _LOG_MAX_SIZE_MB, 

338 "max_age_days": _LOG_MAX_AGE_DAYS, 

339 "max_backups": _LOG_MAX_BACKUPS, 

340 "compress": _LOG_COMPRESS, 

341 } 

342 }, 

343 "processes": processes, 

344 }

347def render_compose( 

348 *, 

349 agent_ids: Iterable[str], 

350 home: str, 

351 launcher: str, 

352 mcp_servers: Iterable[str] = (), 

353 broker_managed: bool = True, 

354) -> str: 

355 """Render the Process Compose project as a YAML string. 

356 

357 Thin serializer over :func:`build_compose_project` — ``sort_keys=False`` keeps 

358 the substrate-before-roster ordering the builder emits, which makes the 

359 generated file readable even though the user never edits it. ``broker_managed`` 

360 is threaded through unchanged (see :func:`build_compose_project`). 

361 """ 

362 project = build_compose_project( 

363 agent_ids=agent_ids, 

364 home=home, 

365 launcher=launcher, 

366 mcp_servers=mcp_servers, 

367 broker_managed=broker_managed, 

368 ) 

369 return yaml.safe_dump(project, sort_keys=False)
