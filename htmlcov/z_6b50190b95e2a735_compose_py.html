1"""Render the Process Compose project that supervises a calfcord host. 

2 

3The Process Compose YAML is *derived state* — calfcord generates it from the 

4agents on disk and config; the user never edits it (design §3.1). This module is 

5the pure heart of that generation: agent ids + home dir + launcher prefix in, a 

6process-compose project ``dict`` out. No filesystem, no broker, no network — so 

7the structure is fully golden-testable and the same generator works whether the 

8host runs detached, in dev, or under a future supervisor swap. 

9 

10The shape is the §13.2 Phase-0 contract, pinned against Process Compose 

11``v1.110.0`` (config schema ``version: "0.5"``): 

12 

13* **Substrate** (``broker``, ``bridge``) autostarts under ``calfcord start``; 

14 **roster** (every agent + ``tools`` / ``router``) is declared 

15 ``disabled`` and waits for an explicit ``... start`` (a ``POST /process/start``). 

16 "Nothing runs that the user did not start" is a trust property, so the split is 

17 encoded here, not left to the launcher. 

18* Every ``command`` invokes the calfcord *launcher* (the shim) rather than a 

19 reconstructed ``uv run`` line, so the venv + ``--env-file`` + default env come 

20 from one place and no secret literal is ever inlined into the YAML (design 

21 §12.3). The launcher prefix is a parameter so this generator is mode-agnostic 

22 and unit-testable. 

23* ``bridge`` and roster members gate on the broker via ``depends_on`` 

24 (``process_healthy``); readiness is an ``exec`` probe (the bridge has no HTTP 

25 server) calling ``<launcher> _healthcheck <component>``. 

26* ``restart: always`` for the substrate (``broker``, ``bridge``); ``on_failure`` 

27 for the whole roster — every agent *and* ``tools`` / ``router`` — which 

28 now all run via ``run_worker_until_signal`` and therefore force a non-zero exit 

29 on any uncommanded exit (a crash *or* a clean signal-less return), while an 

30 operator-commanded ``stop`` is suppressed from restart by Process Compose 

31 regardless of exit code. (The bridge stays ``always``: it owns its own signals 

32 and exits 0 on a clean shutdown, so ``on_failure`` would never fire to recover 

33 it from an uncommanded clean return.) Never ``exit_on_failure``. 

34 

35The REST port is intentionally absent: it is a flag to ``process-compose up`` 

36(``-p <PC_PORT>``), not a field in the project file (design §13.2). 

37""" 

38 

39from __future__ import annotations 

40 

41import os 

42from collections.abc import Iterable, Mapping 

43 

44import yaml 

45 

46# Process Compose config-file schema version (NOT the binary version). v1.110.0 

47# reads the "0.5" schema; confirmed against the process-compose docs. 

48COMPOSE_SCHEMA_VERSION = "0.5" 

49 

50# Readiness-probe cadence — the §13.2 pinned values. An exec probe (the bridge 

51# exposes no HTTP endpoint) shells out to the launcher's internal healthcheck. 

52_PROBE_INITIAL_DELAY_SECONDS = 2 

53_PROBE_PERIOD_SECONDS = 3 

54_PROBE_TIMEOUT_SECONDS = 5 

55_PROBE_SUCCESS_THRESHOLD = 1 

56_PROBE_FAILURE_THRESHOLD = 3 

57 

58# Autorestart backoff shared by both restart policies (always for the substrate, 

59# on_failure for the roster); max_restarts 0 == unlimited retries in Process 

60# Compose, applied uniformly. 

61_RESTART_BACKOFF_SECONDS = 2 

62_RESTART_MAX_RESTARTS = 0 

63 

64# Per-host log rotation (project-level); §13.2. 

65_LOG_MAX_SIZE_MB = 10 

66_LOG_MAX_AGE_DAYS = 7 

67_LOG_MAX_BACKUPS = 5 

68_LOG_COMPRESS = True 

69 

70# Graceful shutdown: SIGTERM with a 10s grace window — comfortably above the 

71# ~2s an agent needs to publish its AgentDepartureEvent — signalling the whole 

72# group (parent_only: false) so child workers under the shim also stop. 

73_SHUTDOWN_SIGNAL = 15 

74_SHUTDOWN_TIMEOUT_SECONDS = 10 

75_SHUTDOWN_PARENT_ONLY = False 

76 

77_HEALTHY = "process_healthy" 

78 

79# The filename stem of the supervisor's *own* log (``process-compose up -L 

80# <stem>.log``). It is not a process the generator declares, but it sits beside 

81# the per-process logs and is a legitimate tail target, so its name must agree 

82# across the modules that write it (``lifecycle._SUPERVISOR_LOG_FILENAME``) and 

83# read it (``cli.logs``). Both derive from this one stem so the literal can never 

84# drift: lifecycle reconstructs the filename as ``SUPERVISOR_LOG_STEM + ".log"``, 

85# and ``cli.logs`` passes the stem to ``_log_location`` (which appends ``.log``). 

86# Homed here because both consumers already import ``compose`` and ``compose`` is 

87# import-light, keeping the logs CLI's decoupling intact. 

88SUPERVISOR_LOG_STEM = "process-compose" 

89 

90# Process names owned by the substrate + non-agent components. An agent id equal 

91# to one of these would silently overwrite that process's entry in the shared 

92# `processes` dict (corrupting the substrate), so the generator rejects it. 

93_RESERVED_PROCESS_NAMES = frozenset({"broker", "bridge", "tools", "router"}) 

94 

95# The slot-name convention for MCP servers, homed here (like 

96# SUPERVISOR_LOG_STEM) because compose declares the slots and the roster 

97# drives them — both must agree on the literal, and compose is the 

98# import-light module the supervisor package already leans on. 

99MCP_SLOT_PREFIX = "mcp-" 

100 

101 

102def mcp_slot_name(server: str) -> str: 

103 """The Process Compose slot for MCP server ``server`` (``mcp-<server>``).""" 

104 return f"{MCP_SLOT_PREFIX}{server}" 

105

107def _log_location(home: str, name: str) -> str: 

108 return os.path.join(home, "state", "logs", f"{name}.log") 

109

111def _restart(policy: str) -> dict: 

112 """An ``availability`` block for ``always`` or ``on_failure``.""" 

113 return { 

114 "restart": policy, 

115 "backoff_seconds": _RESTART_BACKOFF_SECONDS, 

116 "max_restarts": _RESTART_MAX_RESTARTS, 

117 } 

118 

119 

120def _readiness_probe(launcher: str, component: str) -> dict: 

121 """An exec readiness probe driving ``depends_on: process_healthy``.""" 

122 return { 

123 "exec": {"command": f"{launcher} _healthcheck {component}"}, 

124 "initial_delay_seconds": _PROBE_INITIAL_DELAY_SECONDS, 

125 "period_seconds": _PROBE_PERIOD_SECONDS, 

126 "timeout_seconds": _PROBE_TIMEOUT_SECONDS, 

127 "success_threshold": _PROBE_SUCCESS_THRESHOLD, 

128 "failure_threshold": _PROBE_FAILURE_THRESHOLD, 

129 }

132def _process( 

133 *, 

134 command: str, 

135 home: str, 

136 name: str, 

137 disabled: bool, 

138 restart_policy: str, 

139 depends_on: Mapping[str, str] | None = None, 

140 readiness_probe: dict | None = None, 

141) -> dict: 

142 proc: dict = { 

143 "command": command, 

144 "disabled": disabled, 

145 "availability": _restart(restart_policy), 

146 "shutdown": { 

147 "signal": _SHUTDOWN_SIGNAL, 

148 "timeout_seconds": _SHUTDOWN_TIMEOUT_SECONDS, 

149 "parent_only": _SHUTDOWN_PARENT_ONLY, 

150 }, 

151 "log_location": _log_location(home, name), 

152 } 

153 if depends_on is not None: 

154 proc["depends_on"] = { 

155 dep: {"condition": condition} for dep, condition in depends_on.items() 

156 } 

157 if readiness_probe is not None: 

158 proc["readiness_probe"] = readiness_probe 

159 return proc

162def build_compose_project( 

163 *, agent_ids: Iterable[str], home: str, launcher: str, mcp_servers: Iterable[str] = () 

164) -> dict: 

165 """Build the Process Compose project that supervises one calfcord host. 

166 

167 ``agent_ids`` are the roster agents to declare (the caller enumerates the 

168 host's ``agents/*.md``; this function never reads the filesystem), and 

169 ``mcp_servers`` the MCP server names to declare (the caller enumerates 

170 ``mcp.json`` via the no-secrets ``list_server_names`` reader) — each gets 

171 its own ``mcp-<server>`` roster slot, so one broken server entry can 

172 never take down sibling servers and each restarts independently. ``home`` 

173 is ``$CALFCORD_HOME`` — only the per-process ``state/logs/<name>.log`` paths 

174 use it. ``launcher`` is the shim prefix every ``command`` is built on 

175 (e.g. ``$CALFCORD_HOME/shims/calfcord``); the generator never reconstructs 

176 ``uv run`` flags or inlines secrets. 

177 

178 Returns a plain ``dict`` (serialize with :func:`render_compose`). See the 

179 module docstring for the substrate/roster, restart, depends_on, and probe 

180 contract this encodes. Raises :class:`ValueError` if an ``agent_id`` collides 

181 with a reserved substrate/component process name or an MCP server slot. 

182 """ 

183 agent_ids = list(agent_ids) 

184 mcp_servers = list(mcp_servers) 

185 mcp_slots = {mcp_slot_name(server): server for server in mcp_servers} 

186 reserved = sorted(_RESERVED_PROCESS_NAMES.intersection(agent_ids)) 

187 if reserved: 

188 raise ValueError( 

189 f"agent id(s) {reserved} collide with reserved process name(s) " 

190 f"{sorted(_RESERVED_PROCESS_NAMES)}; rename the agent(s)" 

191 ) 

192 slot_collisions = sorted(set(mcp_slots).intersection(agent_ids)) 

193 if slot_collisions: 

194 raise ValueError( 

195 f"agent id(s) {slot_collisions} collide with MCP server process " 

196 f"slot(s) (one per mcp.json server, named mcp-<server>); rename the " 

197 f"agent(s) or the server(s)" 

198 ) 

199 

200 processes: dict[str, dict] = {} 

201 

202 # Substrate — autostarts, health-gated. The broker's readiness probe checks 

203 # metadata reachability (not bare TCP); the bridge's checks the Discord 

204 # heartbeat. `start` gates downstream on the bridge. 

205 processes["broker"] = _process( 

206 command=f"{launcher} broker", 

207 home=home, 

208 name="broker", 

209 disabled=False, 

210 restart_policy="always", 

211 readiness_probe=_readiness_probe(launcher, "broker"), 

212 ) 

213 processes["bridge"] = _process( 

214 command=f"{launcher} run bridge", 

215 home=home, 

216 name="bridge", 

217 disabled=False, 

218 restart_policy="always", 

219 depends_on={"broker": _HEALTHY}, 

220 readiness_probe=_readiness_probe(launcher, "bridge"), 

221 ) 

222 

223 # Roster — declared disabled; each member clocks in on an explicit start and 

224 # gates on the broker being healthy. Every roster member (agents *and* 

225 # tools/router) runs via run_worker_until_signal, which forces a non-zero 

226 # exit on any uncommanded exit, so on_failure restarts a crash while an 

227 # operator-commanded stop is suppressed from restart by Process Compose. 

228 for agent_id in agent_ids: 

229 processes[agent_id] = _process( 

230 command=f"{launcher} run agent {agent_id}", 

231 home=home, 

232 name=agent_id, 

233 disabled=True, 

234 restart_policy="on_failure", 

235 depends_on={"broker": _HEALTHY}, 

236 ) 

237 

238 for component in ("tools", "router"): 

239 processes[component] = _process( 

240 command=f"{launcher} run {component}", 

241 home=home, 

242 name=component, 

243 disabled=True, 

244 restart_policy="on_failure", 

245 depends_on={"broker": _HEALTHY}, 

246 ) 

247 

248 # MCP servers — roster members like agents (disabled, on_failure, 

249 # broker-gated), one slot per mcp.json server for failure isolation. 

250 for slot, server in mcp_slots.items(): 

251 processes[slot] = _process( 

252 command=f"{launcher} run mcp {server}", 

253 home=home, 

254 name=slot, 

255 disabled=True, 

256 restart_policy="on_failure", 

257 depends_on={"broker": _HEALTHY}, 

258 ) 

259 

260 return { 

261 "version": COMPOSE_SCHEMA_VERSION, 

262 "log_configuration": { 

263 "rotation": { 

264 "max_size_mb": _LOG_MAX_SIZE_MB, 

265 "max_age_days": _LOG_MAX_AGE_DAYS, 

266 "max_backups": _LOG_MAX_BACKUPS, 

267 "compress": _LOG_COMPRESS, 

268 } 

269 }, 

270 "processes": processes, 

271 } 

272 

273 

274def render_compose( 

275 *, agent_ids: Iterable[str], home: str, launcher: str, mcp_servers: Iterable[str] = () 

276) -> str: 

277 """Render the Process Compose project as a YAML string. 

278 

279 Thin serializer over :func:`build_compose_project` — ``sort_keys=False`` keeps 

280 the substrate-before-roster ordering the builder emits, which makes the 

281 generated file readable even though the user never edits it. 

282 """ 

283 project = build_compose_project( 

284 agent_ids=agent_ids, home=home, launcher=launcher, mcp_servers=mcp_servers 

285 ) 

286 return yaml.safe_dump(project, sort_keys=False)
